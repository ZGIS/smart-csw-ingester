34 class QueryController @Inject()(luceneService: LuceneService) extends Controller with ClassnameLogger {
35 
36   implicit val geoJSONFeatureCollectionWrite = GeoJSONFeatureCollectionWriter
37   private lazy val DEFAULT_QUERY = "*:*"
38 
39   /**
40     * Action that passes the query from URL to the [[services.LuceneService]].
41     *
42     * @param query
43     * @return
44     * @see services.LuceneService
45     */
46   def query(query: Option[String],
47             bboxWkt: Option[String],
48             fromDateStr: Option[String],
49             toDateStr: Option[String]): Action[AnyContent] = Action {
50     logger.info(s"Query CSW: ${query.getOrElse("NONE")}")
51     //TODO SR make tuple expression out of this
52     val fromDate = fromDateStr match {
53       case None => Some(LocalDate.ofEpochDay(0))
54       case _ => Some(MdMetadataSet.dateFromStrings(List(fromDateStr.get)))
55     }
56 
57     val toDate = toDateStr match {
58       case None => Some(LocalDate.now())
59       case _ => Some(MdMetadataSet.dateFromStrings(List(toDateStr.get)))
60     }
61 
62     //FIXME SR error handling
63     val featureCollection = luceneService.query(query.getOrElse(DEFAULT_QUERY), bboxWkt, fromDate, toDate)
64 
65     // TODO AK here we could insert the query string again if needed
66     val resultJson = Json.toJson(featureCollection)
67 
68     Ok(resultJson).as(JSON)
69   }
70 }
