38   /**
39     * Action that passes the query from URL to the [[services.LuceneService]].
40     *
41     * @param query
42     * @return
43     * @see services.LuceneService
44     */
45   def query(query: Option[String],
46             bboxWkt: Option[String],
47             fromDateStr: Option[String],
48             toDateStr: Option[String]): Action[AnyContent] = Action {
49 
50     //TODO SR make tuple expression out of this
51     val fromDate = fromDateStr match {
52       case None => Some(LocalDate.ofEpochDay(0))
53       case _ => Some(MdMetadataSet.dateFromStrings(List(fromDateStr.get)))
54     }
55 
56     val toDate = toDateStr match {
57       case None => Some(LocalDate.now())
58       case _ => Some(MdMetadataSet.dateFromStrings(List(toDateStr.get)))
59     }
60 
61     //FIXME SR error handling
62     val featureCollection = luceneService.query(query.getOrElse(DEFAULT_QUERY), bboxWkt, fromDate, toDate)
63 
64     // TODO AK here we could insert the query string again if needed
65     val resultJson = Json.toJson(featureCollection)
66 
67     Ok(resultJson).as(JSON)
68   }
