88   /**
89     * Action that passes the query from URL to the [[services.LuceneService]].
90     *
91     * @param query              Option[String] that contains the Lucene query. Defaults to [[QueryController.DEFAULT_QUERY]]
92     * @param bboxWkt            Option[String] that contains the bbox (as WKT) to search in.
93     * @param fromDateStr        Option[String] containing an ISO-DATE for lower date bound
94     * @param toDateStr          Option[String] containing an ISO-DATE for upper date bound
95     * @param maxNumberOfResults Option[Int] containing the max number of documents to return
96     * @param contentType        Option[String] containing the type, of the result document. Valid values: "GeoJson" and "OwcContext". Defaults to "GeoJson"
97     * @see [[services.LuceneService]]
98     * @return
99     */
100   def query(query: Option[String],
101             bboxWkt: Option[String],
102             fromDateStr: Option[String],
103             toDateStr: Option[String],
104             maxNumberOfResults: Option[Int],
105             contentType: Option[String]): Action[AnyContent] = Action { implicit request =>
106     logger.info(s"Query CSW: ${query.getOrElse("NONE")}")
107     // TODO SR make tuple expression out of this
108     val fromDate = fromDateStr match {
109       case None => Some(LocalDate.ofEpochDay(0))
110       case _ => Some(MdMetadataSet.dateFromStrings(List(fromDateStr.get)))
111     }
112 
113     val toDate = toDateStr match {
114       case None => Some(LocalDate.now())
115       case _ => Some(MdMetadataSet.dateFromStrings(List(toDateStr.get)))
116     }
117 
118     try {
119       val searchResult = luceneService.query(query.getOrElse(DEFAULT_QUERY), bboxWkt, fromDate, toDate,
120         maxNumberOfResults)
121       val featureCollection = searchResult.documents
122 
123       contentType.getOrElse("GeoJson") match {
124         case "GeoJson" => {
125           // TODO AK here we could insert the query string again if needed
126           val jsonTransformer = __.json.update(
127             (__ \ 'countMatched).json.put(JsNumber(searchResult.numberOfMatchingDocuments)))
128           val resultJson: JsValue = Json.toJson(featureCollection)
129 
130           Ok(resultJson.transform(jsonTransformer).get).as(JSON)
131         }
132         case "OwcContext" => {
133           // FIXME SR we should use the standardized "FORWARDED" header. Or better both? I hate HTTP(S)!
134           // See: https://developer.mozilla.org/en-US/docs/Web/HTTP/Headers/Forwarded
135           val proto = request.headers.get("X-Forwarded-Proto").getOrElse("http")
136           val host = request.headers.get("X-Forwarded-Host").getOrElse(request.headers("host"))
137           val url = s"${proto}://${request.headers("host")}${request.uri}"
138           logger.debug(s"Requested URL by client: ${url}")
139           val resultJson = toOwcContextJsonWithScores(featureCollection = featureCollection, url = url)
140           Ok(resultJson).as(JSON)
141         }
142         case _ => BadRequest(Json.toJson(ErrorResult(s"Unknown content type: ${contentType}", None))).as(JSON)
143       }
144     }
145     catch {
146       case e: ParseException => {
147         logger.error("Exception parsing query: " + e.getMessage, e);
148         val error: ErrorResult = ErrorResult(s"Could not parse query '${query.get}' ", Some(e.getMessage))
149         InternalServerError(Json.toJson(error)).as(JSON)
150       }
151     }
152   }
