36 
37   implicit val geoJSONFeatureCollectionWrite = GeoJSONFeatureCollectionWriter
38   private lazy val DEFAULT_QUERY = "*:*"
39 
40   /**
41     * Action that passes the query from URL to the [[services.LuceneService]].
42     *
43     * @param query
44     * @return
45     * @see services.LuceneService
46     */
47   def query(query: Option[String],
48             bboxWkt: Option[String],
49             fromDateStr: Option[String],
50             toDateStr: Option[String]): Action[AnyContent] = Action {
51     logger.info(s"Query CSW: ${query.getOrElse("NONE")}")
52     //TODO SR make tuple expression out of this
53     val fromDate = fromDateStr match {
54       case None => Some(LocalDate.ofEpochDay(0))
55       case _ => Some(MdMetadataSet.dateFromStrings(List(fromDateStr.get)))
56     }
57 
58     val toDate = toDateStr match {
59       case None => Some(LocalDate.now())
60       case _ => Some(MdMetadataSet.dateFromStrings(List(toDateStr.get)))
61     }
62 
63     //FIXME SR error handling
64     val featureCollection = luceneService.query(query.getOrElse(DEFAULT_QUERY), bboxWkt, fromDate, toDate)
65 
66     // TODO AK here we could insert the query string again if needed
67     val resultJson = Json.toJson(featureCollection)
68 
69     Ok(resultJson).as(JSON)
70   }
71 
72   def buildIndexFor(catalogueName: String): Action[AnyContent] = Action {
73     logger.info(s"Request for building index for $catalogueName")
74     val result = luceneService.buildIndex(catalogueName)
75     Ok(s"building index for $catalogueName")
76   }
