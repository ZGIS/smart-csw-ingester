42   /**
43     * Action that passes the query from URL to the [[services.LuceneService]].
44     *
45     * @param query
46     * @return
47     * @see services.LuceneService
48     */
49   def query(query: Option[String],
50             bboxWkt: Option[String],
51             fromDateStr: Option[String],
52             toDateStr: Option[String],
53             maxNumberOfResults: Option[Int]): Action[AnyContent] = Action {
54     logger.info(s"Query CSW: ${query.getOrElse("NONE")}")
55     // TODO SR make tuple expression out of this
56     val fromDate = fromDateStr match {
57       case None => Some(LocalDate.ofEpochDay(0))
58       case _ => Some(MdMetadataSet.dateFromStrings(List(fromDateStr.get)))
59     }
60 
61     val toDate = toDateStr match {
62       case None => Some(LocalDate.now())
63       case _ => Some(MdMetadataSet.dateFromStrings(List(toDateStr.get)))
64     }
65 
66     try {
67       val searchResult = luceneService.query(query.getOrElse(DEFAULT_QUERY), bboxWkt, fromDate, toDate,
68         maxNumberOfResults)
69       val featureCollection = searchResult.documents
70 
71       // TODO AK here we could insert the query string again if needed
72       val jsonTransformer = __.json.update(
73         (__ \ 'countMatched).json.put(JsNumber(searchResult.numberOfMatchingDocuments)))
74       val resultJson: JsValue = Json.toJson(featureCollection)
75 
76       Ok(resultJson.transform(jsonTransformer).get).as(JSON)
77     }
78     catch {
79       case e: ParseException => { // TODO SR generally catch Exception is BAD(tm) but can we live with that here?
80         logger.error("Exception parsing query: " + e.getMessage, e);
81         val error: ErrorResult = ErrorResult(s"Could not parse query '${query.get}' ", Some(e.getMessage))
82         InternalServerError(Json.toJson(error)).as(JSON)
83       }
84     }
85   }
86 
87   def buildIndexFor(catalogueName: String): Action[AnyContent] = Action {
88     logger.info(s"Request for building index for $catalogueName")
89     luceneService.buildIndex(catalogueName)
90     Ok(s"building index for $catalogueName")
91   }
