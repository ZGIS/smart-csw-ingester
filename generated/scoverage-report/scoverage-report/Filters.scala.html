27 /**
28  * This class configures filters that run on every request. This
29  * class is queried by Play to get a list of filters.
30  *
31  * Play will automatically use filters from any class called
32  * `Filters` that is placed the root package. You can load filters
33  * from a different class by adding a `play.http.filters` setting to
34  * the `application.conf` configuration file.
35  *
36  * @param env Basic environment settings for the current application.
37  * @param exampleFilter A demonstration filter that adds a header to
38  * each response.
39  */
40 @deprecated(message = "DEMO FILE - DELETE FOR PRODUCTION", since = "0.0.0")
41 @Singleton
42 class Filters @Inject() (
43   env: Environment,
44   exampleFilter: ExampleFilter) extends HttpFilters {
45 
46   override val filters = {
47     // Use the example filter if we're running development mode. If
48     // we're running in production or test mode then don't use any
49     // filters at all.
50     if (env.mode == Mode.Dev) Seq(exampleFilter) else Seq.empty
51   }
52 
53 }
