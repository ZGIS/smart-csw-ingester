147   /**
148     * Builds a seq of Lucene Documents from a given CswGetRecordsResponse object
149     *
150     * @param cswGetRecordsResponse
151     * @param catalogueName
152     * @return
153     */
154   private def extractLuceneDocsFromResponse(cswGetRecordsResponse: CswGetRecordsResponse, catalogueName: String): Map[String, Document] = {
155     logger.info(s"Start building (partial) index for ${catalogueName}...")
156     val mdMetadataSet = (cswGetRecordsResponse.xml \\ "MD_Metadata").map(mdMetadataNode => {
157       logger.debug(f"Preparing($catalogueName): ${(mdMetadataNode \ "fileIdentifier" \ "CharacterString").text}")
158       logger.trace(mdMetadataNode.toString)
159       MdMetadataSet.fromXml(mdMetadataNode, catalogueName, luceneService.getCatalogueUrl(catalogueName))
160     }).filter(item => item.isDefined).map(item => item.get) //filter out all None values
161 
162     mdMetadataSet.map(md => (md.fileIdentifier, md.asLuceneDocument)).toMap
163   }

166     * Builds a Lucene Index from a given CswGetRecordsResponse object

170     * @return [[Directory]] containing the index

172   @deprecated

174 
175     val directory = new RAMDirectory()
176     val config = new IndexWriterConfig()
177     config.setCommitOnClose(true)

179     //FIXME SR use SCALA_ARM (automated resource management)?

182       iwriter.deleteAll()
183       iwriter.commit()

194   /**
195     * it all happens here
196     *
197     * @param catalogueName
198     * @param catalogueUrl
199     * @param startDocument
200     * @param documentsToFetch
201     */
202   private def queryCatalogue(catalogueName: String, catalogueUrl: String, startDocument: Int,
203                              documentsToFetch: Int): Unit = {
204     val wsClientResponseFuture =
205       wsClient.url(catalogueUrl)
206 //        .withRequestTimeout(20.seconds)
207         .withHeaders("Content-Type" -> "application/xml")
208         .post(CswGetRecordsRequest(startDocument, documentsToFetch))
209 
210     wsClientResponseFuture.onComplete {
211       case Success(wsClientResponse) => {
212         wsClientResponse.status match {
213           case Status.OK => {
214             logger.debug(f"Response Content Type: ${wsClientResponse.allHeaders.getOrElse("Content-Type", "Unknown")}")
215             logger.debug(f"Response-Length: ${wsClientResponse.body.length}")
216             logger.trace(f"Response-Body: ${wsClientResponse.body.toString}")
217             wsClientResponse.xml.label match {
218               case "ExceptionReport" => {
219                 logger.warn(
220                   f"Got XML Exception Response. Text: ${(wsClientResponse.xml \ "Exception" \ "ExceptionText").text}")
221               }
222               case "GetRecordsResponse" => {
223                 val cswGetRecResp = CswGetRecordsResponse(wsClientResponse.xml)
224                 logger.info(
225                   f"c: $catalogueName nextRecord: ${cswGetRecResp.nextRecord}, " +
226                     f"numberOfRec ${cswGetRecResp.numberOfRecordsMatched}, " +
227                     f"recordsReturned ${cswGetRecResp.numberOfRecordsReturned}")
228                 if ((cswGetRecResp.nextRecord > cswGetRecResp.numberOfRecordsMatched) ||
229                   (cswGetRecResp.nextRecord == 0) ||
230                   (cswGetRecResp.numberOfRecordsReturned == 0)) {
231                   logger.info(s"Sending IndexResponseDocument - is last! ($catalogueName)")
232                   self ! IndexResponseDocument(cswGetRecResp, catalogueName, true)
233                 }
234                 else {
235                   logger.info(s"Sending IndexResponseDocument - and start another query round. ($catalogueName)")
236                   self ! IndexResponseDocument(cswGetRecResp, catalogueName, false)
237                   queryCatalogue(catalogueName, catalogueUrl, cswGetRecResp.nextRecord, documentsToFetch)
238                 }
239               }
240               case _ => {
241                 logger.warn(f"Unknown response content. Body: ${wsClientResponse.xml.toString} ($catalogueName)")
242                 self ! ShutdownActor(catalogueName)
243               }
244             }
245           }
246           case _ => {
247             self ! ShutdownActor(catalogueName)
248           }
249         }
250       }
251       case Failure(ex) => {
252         logger.warn(s"Exception while querying CSW $catalogueName (${ex.getClass.getCanonicalName}): ${ex.getMessage}", ex)
253         self ! ShutdownActor(catalogueName)
254       }
255     }
256   }
