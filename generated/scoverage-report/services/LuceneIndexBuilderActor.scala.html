59 /**
60   * Builds a Lucene Index from a given CswGetRecordsResponse object
61   */
62 class LuceneIndexBuilderActor @Inject()(configuration: Configuration, wsClient: WSClient, luceneService: LuceneService)
63                                        (implicit ec: ExecutionContext) extends Actor with
64   ClassnameLogger {
65 
66   logger.debug("Reading configuration: csw.maxDocs ")
67   private val CSW_MAX_RECORDS = 500
68   private val maxDocsPerFetch = configuration.getInt("csw.maxDocs").getOrElse(CSW_MAX_RECORDS)
69 
70   private val catalogueDirectory = new RAMDirectory()
71 
72 
73   /**
74     * Defines the initial Actor behaviour. Following behaviours are implemented:
75     *   - [[IndexCatalogue]]
76     *
77     * @return
78     */
79   override def receive: Receive = {
80 
81     case IndexCatalogue(catalogueName, catalogueUrl) => {
82       logger.info("querying catalogue")
83       context.become(indexing)
84       self ! QueryCatalogue(catalogueName, catalogueUrl)
85     }
86     //FIXME some kind of default behaviour?
87   }
88 
89   def indexing: Receive = {
90     case QueryCatalogue(catalogueName, catalogueUrl) => {
91       logger.info(s"Querying $catalogueName on url $catalogueUrl")
92       queryCatalogue(catalogueName, catalogueUrl, 1, maxDocsPerFetch)
93     }
94 
95     case IndexResponseDocument(cswGetRecordsResponse, catalogueName, isLast) => {
96       val directory = buildIndexFor(cswGetRecordsResponse, catalogueName)
97       sender() ! MergeLuceneIndex(directory, catalogueName)
98 
99       //its important to send the shutdown from here, otherwise the last merge will not be executed, because shutdown
100       //will be received before this merge message
101       if (isLast) {
102         logger.info(s"Last response indexed. Sending shutdown for $catalogueName")
103         sender() ! ShutdownActor(catalogueName)
104       }
105     }
106 
107     case MergeLuceneIndex(directory, catalogueName) => {
108       logger.info(s"Merging index in actor for $catalogueName")
109       val config = new IndexWriterConfig()
110       config.setCommitOnClose(true)
111 
112       val iwriter = new IndexWriter(catalogueDirectory, config)
113       try {
114         iwriter.addIndexes(directory)
115         iwriter.commit()
116         logger.info("Done merging index")
117       }
118       finally {
119         iwriter.close()
120       }
121     }
122 
123     case ShutdownActor(catalogueName) => {
124       if (catalogueDirectory.listAll().length > 0) {
125         logger.info("Merging index back to Service")
126         luceneService.mergeIndex(catalogueDirectory, catalogueName)
127       }
128 
129       logger.info(s"Shutting down actor for $catalogueName")
130       context.stop(self)
131     }
132     //FIXME some kind of default behaviour?
133   }
134 
135   /**
136     * Builds a Lucene Index from a given CswGetRecordsResponse object
137     *
138     * @param cswGetRecordsResponse [[CswGetRecordsResponse]] to build an index from
139     * @param catalogueName         String containing the catalogue name
140     * @return [[Directory]] containing the index
141     */
142   private def buildIndexFor(cswGetRecordsResponse: CswGetRecordsResponse, catalogueName: String): Directory = {
143     logger.info(s"Start building (partial) index for ${catalogueName}...")
144     val mdMetadataSet = (cswGetRecordsResponse.xml \\ "MD_Metadata").map(mdMetadataNode => {
145       logger.debug(f"Preparing($catalogueName): ${(mdMetadataNode \ "fileIdentifier" \ "CharacterString").text}")
146       logger.trace(mdMetadataNode.toString)
147       MdMetadataSet.fromXml(mdMetadataNode, catalogueName, luceneService.getCatalogueUrl(catalogueName))
148     }).filter(item => item.isDefined).map(item => item.get) //filter out all None values
149 
150     val luceneDocuments = mdMetadataSet.map(_.asLuceneDocument)
151 
152     val directory = new RAMDirectory()
153     val config = new IndexWriterConfig()
154     config.setCommitOnClose(true)
155 
156     //FIXME SR use SCALA_ARM (automated resource management)?
157     val iwriter = new IndexWriter(directory, config)
158     try {
159       iwriter.deleteAll()
160       iwriter.commit()
161       luceneDocuments.foreach(iwriter.addDocument(_))
162       iwriter.commit()
163       logger.info(s"Partial index ready for $catalogueName")
164     }
165     finally {
166       iwriter.close()
167     }
168     directory
169   }
170 
171   private def queryCatalogue(catalogueName: String, catalogueUrl: String, startDocument: Int,
172                              documentsToFetch: Int): Unit = {
173     val wsClientResponseFuture =
174       wsClient.url(catalogueUrl)
175 //        .withRequestTimeout(20.seconds)
176         .withHeaders("Content-Type" -> "application/xml")
177         .post(CswGetRecordsRequest(startDocument, documentsToFetch))
178 
179     wsClientResponseFuture.onComplete {
180       case Success(wsClientResponse) => {
181         wsClientResponse.status match {
182           case Status.OK => {
183             logger.debug(f"Response Content Type: ${wsClientResponse.allHeaders.getOrElse("Content-Type", "Unknown")}")
184             logger.debug(f"Response-Length: ${wsClientResponse.body.length}")
185             logger.trace(f"Response-Body: ${wsClientResponse.body.toString}")
186             wsClientResponse.xml.label match {
187               case "ExceptionReport" => {
188                 logger.warn(
189                   f"Got XML Exception Response. Text: ${(wsClientResponse.xml \ "Exception" \ "ExceptionText").text}")
190               }
191               case "GetRecordsResponse" => {
192                 val cswGetRecResp = CswGetRecordsResponse(wsClientResponse.xml)
193                 logger.info(
194                   f"nextRecord: ${cswGetRecResp.nextRecord}, " +
195                     f"numberOfRec ${cswGetRecResp.numberOfRecordsMatched}, " +
196                     f"recordsReturned ${cswGetRecResp.numberOfRecordsReturned}")
197                 if ((cswGetRecResp.nextRecord > cswGetRecResp.numberOfRecordsMatched) ||
198                   (cswGetRecResp.nextRecord == 0) ||
199                   (cswGetRecResp.numberOfRecordsReturned == 0)) {
200                   logger.info("Sending IndexResponseDocument - is last!")
201                   self ! IndexResponseDocument(cswGetRecResp, catalogueName, true)
202                 }
203                 else {
204                   logger.info("Sending IndexResponseDocument - and start another query round.")
205                   self ! IndexResponseDocument(cswGetRecResp, catalogueName, false)
206                   queryCatalogue(catalogueName, catalogueUrl, cswGetRecResp.nextRecord, documentsToFetch)
207                 }
208               }
209               case _ => {
210                 logger.warn(f"Unknown response content. Body: ${wsClientResponse.xml.toString}")
211                 self ! ShutdownActor(catalogueName)
212               }
213             }
214           }
215           case _ => {
216             self ! ShutdownActor(catalogueName)
217           }
218         }
219       }
220       case Failure(ex) => {
221         logger.warn(s"Exception while querying CSW (${ex.getClass.getCanonicalName}): ${ex.getMessage}", ex)
222         self ! ShutdownActor(catalogueName)
223       }
224     }
225   }
