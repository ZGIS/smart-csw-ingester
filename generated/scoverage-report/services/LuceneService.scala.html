103   def buildIndex(): Unit = {

105 
106     val gmdElemSetsFutures = Future.sequence(catalogues.map {
107       case (catName, url) => {
108         queryCatalogue(catName, url)
109       }

112     //FIXME SR 15 minutes seem quite arbitraty... what is a good timeout here?
113     val gmdElemSets = Await.result(gmdElemSetsFutures, 15.minutes).flatten

116     val analyzer = new StandardAnalyzer()
117     val config = new IndexWriterConfig(analyzer)
118     config.setCommitOnClose(true)
119     //FIXME SR use SCALA_ARM (automated resource management)?
120     val iwriter = new IndexWriter(directory, config)
121     try {
122       iwriter.deleteAll()
123       iwriter.commit()
124       gmdElemSets.foreach(searchDocument => iwriter.addDocument(searchDocument.asLuceneDocument()))

126       iwriter.commit()

128     }
129     finally {

131     }

134   /**
135     * Sends a GetRecords POST request to a catalogue and returns the response as a future.
136     *
137     * @param url           String containing URL to catalogue
138     * @param startPosition Int with the first element to grab (pagination)
139     * @param maxDocuments  Int with the max number of documents to return
140     */
141   //TODO SR maybe we should also return the exception object to give more meaningful error messages in the GUI?
142   private def postGetRecordsRequest(url: String, startPosition: Int,
143                                     maxDocuments: Int,
144                                     f: Future[List[CswGetRecordsResponse]]): Future[List[CswGetRecordsResponse]] = {
145     logger.info(f"Sending GetRecords request to $url")
146 
147     val wsClientResponseFuture =
148       wsClient.url(url)
149         //        .withRequestTimeout(20.seconds)
150         .withHeaders("Content-Type" -> "application/xml")
151         .post(CswGetRecordsRequest(startPosition, maxDocuments))
152 
153     val cswGetRecordsResponseListFuture = wsClientResponseFuture.flatMap { wsClientResponse =>
154       logger.info(f"Response status: ${wsClientResponse.status} - ${wsClientResponse.statusText} ($url)")
155       wsClientResponse.status match {
156         case Status.OK => {
157           logger.debug(f"Response Content Type: ${wsClientResponse.allHeaders.getOrElse("Content-Type", "Unknown")}")
158           logger.debug(f"Response-Length: ${wsClientResponse.body.length}")
159           logger.trace(f"Response-Body: ${wsClientResponse.body.toString}")
160           wsClientResponse.xml.label match {
161             case "ExceptionReport" => {
162               logger.warn(
163                 f"Got Exception Response. Text: ${(wsClientResponse.xml \ "Exception" \ "ExceptionText").text}")
164               Future.successful(List())
165             }
166             case "GetRecordsResponse" => {
167               val cswGetRecResp = CswGetRecordsResponse(wsClientResponse.xml)
168               logger.info(
169                 f"nextRecord: ${cswGetRecResp.nextRecord}, numberOfRec ${cswGetRecResp.numberOfRecordsMatched}")
170               if ((cswGetRecResp.nextRecord > cswGetRecResp.numberOfRecordsMatched) ||
171                 (cswGetRecResp.nextRecord == 0)) {
172                 f.flatMap(l => {
173                   Future.successful(cswGetRecResp :: l)
174                 })
175               }
176               else {
177                 f.flatMap(l => {
178                   postGetRecordsRequest(url, cswGetRecResp.nextRecord, maxDocuments,
179                     Future.successful(cswGetRecResp :: l))
180                 })
181               }
182             }
183             case _ => {
184               logger.warn(f"Unknown response content. Body: ${wsClientResponse.xml.toString}")
185               Future.successful(List())
186             }
187           }
188         }
189         case _ => {
190           logger.warn(f"Error while executing CSW query.")
191           logger.debug(f"Response body: ${wsClientResponse.body}")
192           Future.successful(List())
193         }
194       }
195     } recover {
196       case e => {
197         logger.warn(f"Exception on $url", e)
198         List()
199       }
200     }
201     cswGetRecordsResponseListFuture
202   }

205     *
206     * @param catalogueName
207     * @param url
208     * @return
209     */
210   private def queryCatalogue(catalogueName: String, url: String): Future[List[MdMetadataSet]] = {
211     val getRecordsFuture = postGetRecordsRequest(url, 1, maxDocsPerFetch, Future.successful(Nil))
212     val gmdElementsFuture = getRecordsFuture.map(cswGetRecordsResponses => {
213       cswGetRecordsResponses.flatMap(cswGetRecordsResponse => {
214         (cswGetRecordsResponse.xml \\ "MD_Metadata").map(mdMetadataNode => {
215           logger.debug(f"Preparing($catalogueName): ${(mdMetadataNode \ "fileIdentifier" \ "CharacterString").text}")
216           logger.trace(mdMetadataNode.toString)
217           MdMetadataSet.fromXml(mdMetadataNode, catalogueName)
218         })
219       }).filter(item => item.isDefined).map(item => item.get) //filter out all None values
220     })
221     gmdElementsFuture
222   }

241   /**
242     * creates a query, that should find anything inside a BBOX
243     *
244     * @param bboxWkt
245     * @return
246     */
247   private def parseBboxQuery(bboxWkt: String) = {
248     logger.error(s"create query for $bboxWkt")
249 
250     val envelopeWkt = "ENVELOPE\\(([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+)\\)".r
251 
252     val shape = bboxWkt.replaceAll("\\s", "") match {
253       case envelopeWkt(minX, maxX, minY, maxY) => {
254         MdMetadataSet.bboxFromCoords(minX.toFloat, maxX.toFloat, minY.toFloat, maxY.toFloat)
255       }
256     }
257 
258 
259     logger.error(s"parsed shape is ${shape.toString}")
260     val ctx = SpatialContext.GEO
261     val bboxStrategy: BBoxStrategy = BBoxStrategy.newInstance(ctx, "bbox")
262     bboxStrategy.makeQuery(new SpatialArgs(SpatialOperation.IsWithin, shape))
263   }

265   /**
266     * Queries the Search Index
267     *
268     * @param query
269     * @return
270     */
271   def query(query: String,
272             bboxWtk: Option[String] = None,
273             fromDate: Option[LocalDate] = None,
274             toDate: Option[LocalDate] = None): List[MdMetadataSet] = {
275 
276     val textQuery = parseQueryString(query)
277 
278     val dateQuery = LongPoint.newRangeQuery("dateStamp",
279       fromDate.getOrElse(LocalDate.ofEpochDay(0)).toEpochDay(),
280       toDate.getOrElse(LocalDate.now()).toEpochDay())
281 
282     val bboxQuery = parseBboxQuery(bboxWtk.filterNot(_.isEmpty).getOrElse(WORLD_WKT))
283 
284     //FIXME SR when index empty, the next call will fail
285     val searcherManager = new SearcherManager(directory, null)
286     searcherManager.maybeRefreshBlocking()
287     val isearcher = searcherManager.acquire()
288 
289     // TODO TBD AK The way you built it, could omit dates or BBOX query constraint in this
290     // this builder if not provided through query params (but particularly date should NOT be limited to 1970 to now?)
291     val booleanQueryBuilder = new BooleanQuery.Builder()
292     booleanQueryBuilder.add(textQuery, BooleanClause.Occur.MUST)
293     booleanQueryBuilder.add(dateQuery, BooleanClause.Occur.MUST)
294     booleanQueryBuilder.add(bboxQuery, BooleanClause.Occur.MUST)
295     val luceneQuery = booleanQueryBuilder.build()
296 
297     val search = isearcher.search(luceneQuery, isearcher.getIndexReader.numDocs())
298     val scoreDocs = search.scoreDocs
299 
300     val results = scoreDocs.map(scoreDoc => {
301       val doc = isearcher.doc(scoreDoc.doc)
302       MdMetadataSet.fromLuceneDoc(doc)
303     }).toList
304 
305     //FIXME SR use ARM --> possible mem leak
306     searcherManager.release(isearcher)
307 
308     results
309 
310   }
