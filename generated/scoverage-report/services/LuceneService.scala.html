137   def buildIndex(catalogueName: String): Unit = {

139     val catalogueUrl = catalogues(catalogueName)

141     val indexBuilder = (indexBuilderMaster ? GetSpecificIndexBuilder(catalogueName)).mapTo[ActorRef]

148   def mergeIndex(directory: Directory, catalogueName: String): Unit = {
149     catalogueIndexes += (catalogueName -> directory)

151     val config = new IndexWriterConfig()
152     config.setCommitOnClose(true)
153 
154     val iwriter = new IndexWriter(indexRamDirectory, config)
155     try {
156       logger.info(s"Merging ${catalogueIndexes.keys} into master index")
157       iwriter.deleteAll()
158       iwriter.commit()
159       iwriter.addIndexes(catalogueIndexes.values.toArray : _*)

163     }

169   /**
170     * Queries the Search Index
171     *
172     * @param query
173     * @return
174     */
175   def query(query: String,
176             bboxWtk: Option[String] = None,
177             fromDate: Option[LocalDate] = None,
178             toDate: Option[LocalDate] = None,
179             maxNumberOfResults: Option[Int] = None): SearchResult = {
180 
181     val textQuery = parseQueryString(query)
182 
183     val dateQuery = LongPoint.newRangeQuery("dateStamp",
184       fromDate.getOrElse(LocalDate.ofEpochDay(0)).toEpochDay(),
185       toDate.getOrElse(LocalDate.now()).toEpochDay())
186 
187     val bboxQuery = parseBboxQuery(bboxWtk.filterNot(_.isEmpty).getOrElse(WORLD_WKT))
188 
189     searcherManager.maybeRefreshBlocking()
190     val isearcher = searcherManager.acquire()
191 
192     // TODO TBD AK The way you built it, could omit dates or BBOX query constraint in this
193     // this builder if not provided through query params (but particularly date should NOT be limited to 1970 to now?)
194     val booleanQueryBuilder = new BooleanQuery.Builder()
195     booleanQueryBuilder.add(textQuery, BooleanClause.Occur.MUST)
196     booleanQueryBuilder.add(dateQuery, BooleanClause.Occur.MUST)
197     booleanQueryBuilder.add(bboxQuery, BooleanClause.Occur.MUST)
198     val finalLuceneQuery = booleanQueryBuilder.build()
199 
200     val numOfMatchingDocuments = isearcher.count(finalLuceneQuery);
201 
202     // TODO SR is "all" a good default, when queried for 0 or less documents?
203     val maxDocuments = maxNumberOfResults match {
204       case Some(x) if maxNumberOfResults.get <= 0 => numOfMatchingDocuments + 1 //+1 just in case 0 docs match
205       case _ => maxNumberOfResults.getOrElse(defaultMaxDocuments)
206     }
207     val search = isearcher.search(finalLuceneQuery, maxDocuments);
208     val scoreDocs = search.scoreDocs
209 
210     val results = scoreDocs.map(scoreDoc => {
211       val doc = isearcher.doc(scoreDoc.doc)
212       MdMetadataSet.fromLuceneDoc(doc, scoreDoc.score)
213     }).toList
214 
215     //FIXME SR use ARM --> possible mem leak
216     searcherManager.release(isearcher)
217 
218     new SearchResult {
219       override val luceneQuery: Query = finalLuceneQuery
220       override val numberOfMatchingDocuments: Int = numOfMatchingDocuments
221       override val documents: List[MdMetadataSet] = results
222     }
223   }

240   private def parseQueryString(queryString: String) = {
241     queryString.trim() match {
242       case "" => new MatchAllDocsQuery()
243       case _ => {
244         val parser = new QueryParser("catch_all", new StandardAnalyzer())
245         //FIXME errorhandling
246         parser.parse(queryString)
247       }
248     }
249   }

251   /**
252     * creates a query, that should find anything inside a BBOX
253     *
254     * @param bboxWkt
255     * @return
256     */
257   private def parseBboxQuery(bboxWkt: String) = {
258     logger.debug(s"create query for $bboxWkt")
259 
260     val envelopeWktRegex =
261       "ENVELOPE\\(([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+)\\)".r
262 
263     val shape = bboxWkt.replaceAll("\\s", "") match {
264       case envelopeWktRegex(minX, maxX, minY, maxY) => {
265         MdMetadataSet.bboxFromCoords(minX.toFloat, maxX.toFloat, minY.toFloat, maxY.toFloat)
266       }
267       case _ => {
268         logger.warn(s"Could not parse WKT '$bboxWkt'. Using WORLD as query BBOX.");
269         MdMetadataSet.bboxFromCoords(-180, 180, -90, 90);
270       }
271     }
272 
273     logger.debug(s"parsed shape is ${shape.toString}")
274     val bboxStrategy: BBoxStrategy = BBoxStrategy.newInstance(SPATIAL_CONTEXT, "bbox")
275     bboxStrategy.makeQuery(new SpatialArgs(SpatialOperation.IsWithin, shape))
276   }

278   /**

280     *
281     * @param directory
282     */
283   private def prepareEmptyIndex(directory: Directory): Directory = {
284     val analyzer = new StandardAnalyzer()
285     val config = new IndexWriterConfig(analyzer)
286     config.setCommitOnClose(true)
287     //FIXME SR use SCALA_ARM (automated resource management)?
288     val iwriter = new IndexWriter(directory, config)
289     try {
290       iwriter.deleteAll()
291       iwriter.addDocument(new Document)
292       iwriter.commit()
293       logger.info("Empty Index ready")
294     }
295     finally {
296       iwriter.close()
297     }
298     directory
299   }
