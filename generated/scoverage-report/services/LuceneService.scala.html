92     * Builds the Search index
93     */
94   //FIXME SR use something like lib-lucene-sugar. Lib unfortunately seems very old/outdated? https://github.com/gilt/lib-lucene-sugar
95   def buildIndex(): Unit = {
96     logger.info("Building Lucene Index")
97 
98     val gmdElemSetsFutures = Future.sequence(catalogues.map {
99       case (catName, url) => {
100         queryCatalogue(catName, url)
101       }
102     }.toList)
103 
104     //FIXME SR 5 minutes seem quite arbitraty... what is a good timeout here?
105     val gmdElemSets = Await.result(gmdElemSetsFutures, 5.minutes).flatten
106     logger.info(f"Loaded ${gmdElemSets.size} documents from CSW.")
107 
108     val analyzer = new StandardAnalyzer()
109     val config = new IndexWriterConfig(analyzer)
110     config.setCommitOnClose(true)
111     //FIXME SR use SCALA_ARM (automated resource management)?
112     val iwriter = new IndexWriter(directory, config)
113     try {
114       iwriter.deleteAll()
115       iwriter.commit()
116       gmdElemSets.foreach(searchDocument => iwriter.addDocument(searchDocument.asLuceneDocument()))
117 
118       iwriter.commit()
119       logger.info("Index ready")
120     }
121     finally {
122       iwriter.close()
123     }
124   }

126   /**
127     * Sends a GetRecords POST request to a catalogue and returns the response as a future.
128     *
129     * @param url           String containing URL to catalogue
130     * @param startPosition Int with the first element to grab (pagination)
131     * @param maxDocuments  Int with the max number of documents to return
132     */
133   //TODO SR maybe we should also return the exception object to give more meaningful error messages in the GUI?
134   private def postGetRecordsRequest(url: String, startPosition: Int,
135                                     maxDocuments: Int,
136                                     f: Future[List[CswGetRecordsResponse]]): Future[List[CswGetRecordsResponse]] = {
137     logger.info(f"Sending GetRecords request to $url")
138 
139     val wsClientResponseFuture =
140       wsClient.url(url)
141         .withRequestTimeout(20.seconds)
142         .withHeaders("Content-Type" -> "application/xml")
143         .post(CswGetRecordsRequest(startPosition, maxDocuments))
144 
145     val cswGetRecordsResponseListFuture = wsClientResponseFuture.flatMap { wsClientResponse =>
146       logger.info(f"Response status: ${wsClientResponse.status} - ${wsClientResponse.statusText} ($url)")
147       wsClientResponse.status match {
148         case 200 => {
149           logger.debug(f"Response Content Type: ${wsClientResponse.allHeaders.getOrElse("Content-Type", "Unknown")}")
150           logger.debug(f"Response-Length: ${wsClientResponse.body.length}")
151           logger.trace(f"Response-Body: ${wsClientResponse.body.toString}")
152           wsClientResponse.xml.label match {
153             case "ExceptionReport" => {
154               logger.warn(
155                 f"Got Exception Response. Text: ${(wsClientResponse.xml \ "Exception" \ "ExceptionText").text}")
156               Future.successful(List())
157             }
158             case "GetRecordsResponse" => {
159               val cswGetRecResp = CswGetRecordsResponse(wsClientResponse.xml)
160               logger.info(f"nextRecord: ${cswGetRecResp.nextRecord}, numberOfRec ${cswGetRecResp.numberOfRecordsMatched}")
161               if ((cswGetRecResp.nextRecord > cswGetRecResp.numberOfRecordsMatched) ||
162                 (cswGetRecResp.nextRecord == 0))
163                 f.flatMap(l => {
164                   Future.successful(cswGetRecResp :: l)
165                 })
166               else
167                 f.flatMap(l => {
168                   postGetRecordsRequest(url, cswGetRecResp.nextRecord, maxDocuments,
169                     Future.successful(cswGetRecResp :: l))
170                 })
171             }
172             case _ => {
173               logger.warn(f"Unknown response content. Body: ${wsClientResponse.xml.toString}")
174               Future.successful(List())
175             }
176           }
177         }
178         case _ => {
179           logger.warn(f"Error while executing CSW query.")
180           logger.debug(f"Response body: ${wsClientResponse.body}")
181           Future.successful(List())
182         }
183       }
184     } recover {
185         case e => {
186           logger.warn(f"Exception on $url", e)
187           List()
188         }
189     }
190     cswGetRecordsResponseListFuture
191   }

193   /**
194     *
195     * @param catalogueName
196     * @param url
197     * @return
198     */
199   private def queryCatalogue(catalogueName: String, url: String): Future[List[MdMetadataSet]] = {
200     val getRecordsFuture = postGetRecordsRequest(url, 1, maxDocsPerFetch, Future.successful(Nil))
201     val gmdElementsFuture = getRecordsFuture.map(cswGetRecordsResponses => {
202       cswGetRecordsResponses.flatMap(cswGetRecordsResponse => {
203         (cswGetRecordsResponse.xml \\ "MD_Metadata").map(mdMetadataNode => {
204           logger.debug(f"Preparing($catalogueName): ${(mdMetadataNode \ "fileIdentifier" \ "CharacterString").text}")
205           logger.trace(mdMetadataNode.toString)
206           MdMetadataSet.fromXml(mdMetadataNode, catalogueName)
207         })
208       }).filter(item => item.isDefined).map(item => item.get) //filter out all None values
209     })
210     gmdElementsFuture
211   }

213   /**
214     * Queries the Search Index
215     *
216     * @param query
217     * @return
218     */
219   def query(query: String): SearchResult = {
220 
221     val ctx = SpatialContext.GEO
222 
223     val bboxStrategy: BBoxStrategy = BBoxStrategy.newInstance(ctx, "bboxStrategy")
224     // Query = LongPoint.newRangeQuery("dateStampCompare", localDate1.toEpochDay, localDate2.toEpochDay)
225 
226     val luceneQuery = query.trim() match {
227       case "" => new MatchAllDocsQuery()
228       case _ => {
229         val parser = new QueryParser("catch_all", new StandardAnalyzer())
230         //FIXME errorhandling
231         parser.parse(query)
232       }
233     }
234 
235     //FIXME SR when index empty, the next call will fail
236     val searcherManager = new SearcherManager(directory, null)
237     searcherManager.maybeRefreshBlocking()
238     val isearcher = searcherManager.acquire()
239 
240     val search = isearcher.search(luceneQuery, isearcher.getIndexReader.numDocs())
241     val scoreDocs = search.scoreDocs
242 
243     val header = SearchResultHeader(search.totalHits, luceneQuery.toString())
244     val results = scoreDocs.map(scoreDoc => {
245       val doc = isearcher.doc(scoreDoc.doc)
246       MdMetadataSet.fromLuceneDoc(doc)
247     })
248     //FIXME SR use ARM --> possible mem leak
249     searcherManager.release(isearcher)
250     SearchResult(header, results.toList)
251   }
