96     * Builds the Search index
97     */
98   //FIXME SR use something like lib-lucene-sugar. Lib unfortunately seems very old/outdated? https://github.com/gilt/lib-lucene-sugar
99   def buildIndex(): Unit = {
100     logger.info("Building Lucene Index")
101 
102     val gmdElemSetsFutures = Future.sequence(catalogues.map {
103       case (catName, url) => {
104         queryCatalogue(catName, url)
105       }
106     }.toList)
107 
108     //FIXME SR 5 minutes seem quite arbitraty... what is a good timeout here?
109     val gmdElemSets = Await.result(gmdElemSetsFutures, 5.minutes).flatten
110     logger.info(f"Loaded ${gmdElemSets.size} documents from CSW.")
111 
112     val analyzer = new StandardAnalyzer()
113     val config = new IndexWriterConfig(analyzer)
114     config.setCommitOnClose(true)
115     //FIXME SR use SCALA_ARM (automated resource management)?
116     val iwriter = new IndexWriter(directory, config)
117     try {
118       iwriter.deleteAll()
119       iwriter.commit()
120       gmdElemSets.foreach(searchDocument => iwriter.addDocument(searchDocument.asLuceneDocument()))
121 
122       iwriter.commit()
123       logger.info("Index ready")
124     }
125     finally {
126       iwriter.close()
127     }
128   }

130   /**
131     * Sends a GetRecords POST request to a catalogue and returns the response as a future.
132     *
133     * @param url           String containing URL to catalogue
134     * @param startPosition Int with the first element to grab (pagination)
135     * @param maxDocuments  Int with the max number of documents to return
136     */
137   //TODO SR maybe we should also return the exception object to give more meaningful error messages in the GUI?
138   private def postGetRecordsRequest(url: String, startPosition: Int,
139                                     maxDocuments: Int,
140                                     f: Future[List[CswGetRecordsResponse]]): Future[List[CswGetRecordsResponse]] = {
141     logger.info(f"Sending GetRecords request to $url")
142 
143     val wsClientResponseFuture =
144       wsClient.url(url)
145         .withRequestTimeout(20.seconds)
146         .withHeaders("Content-Type" -> "application/xml")
147         .post(CswGetRecordsRequest(startPosition, maxDocuments))
148 
149     val cswGetRecordsResponseListFuture = wsClientResponseFuture.flatMap { wsClientResponse =>
150       logger.info(f"Response status: ${wsClientResponse.status} - ${wsClientResponse.statusText} ($url)")
151       wsClientResponse.status match {
152         case Status.OK => {
153           logger.debug(f"Response Content Type: ${wsClientResponse.allHeaders.getOrElse("Content-Type", "Unknown")}")
154           logger.debug(f"Response-Length: ${wsClientResponse.body.length}")
155           logger.trace(f"Response-Body: ${wsClientResponse.body.toString}")
156           wsClientResponse.xml.label match {
157             case "ExceptionReport" => {
158               logger.warn(
159                 f"Got Exception Response. Text: ${(wsClientResponse.xml \ "Exception" \ "ExceptionText").text}")
160               Future.successful(List())
161             }
162             case "GetRecordsResponse" => {
163               val cswGetRecResp = CswGetRecordsResponse(wsClientResponse.xml)
164               logger.info(f"nextRecord: ${cswGetRecResp.nextRecord}, numberOfRec ${cswGetRecResp.numberOfRecordsMatched}")
165               if ((cswGetRecResp.nextRecord > cswGetRecResp.numberOfRecordsMatched) ||
166                 (cswGetRecResp.nextRecord == 0)) {
167                 f.flatMap(l => {
168                   Future.successful(cswGetRecResp :: l)
169                 })
170               }
171               else {
172                 f.flatMap(l => {
173                   postGetRecordsRequest(url, cswGetRecResp.nextRecord, maxDocuments,
174                     Future.successful(cswGetRecResp :: l))
175                 })
176               }
177             }
178             case _ => {
179               logger.warn(f"Unknown response content. Body: ${wsClientResponse.xml.toString}")
180               Future.successful(List())
181             }
182           }
183         }
184         case _ => {
185           logger.warn(f"Error while executing CSW query.")
186           logger.debug(f"Response body: ${wsClientResponse.body}")
187           Future.successful(List())
188         }
189       }
190     } recover {
191         case e => {
192           logger.warn(f"Exception on $url", e)
193           List()
194         }
195     }
196     cswGetRecordsResponseListFuture
197   }

199   /**
200     *
201     * @param catalogueName
202     * @param url
203     * @return
204     */
205   private def queryCatalogue(catalogueName: String, url: String): Future[List[MdMetadataSet]] = {
206     val getRecordsFuture = postGetRecordsRequest(url, 1, maxDocsPerFetch, Future.successful(Nil))
207     val gmdElementsFuture = getRecordsFuture.map(cswGetRecordsResponses => {
208       cswGetRecordsResponses.flatMap(cswGetRecordsResponse => {
209         (cswGetRecordsResponse.xml \\ "MD_Metadata").map(mdMetadataNode => {
210           logger.debug(f"Preparing($catalogueName): ${(mdMetadataNode \ "fileIdentifier" \ "CharacterString").text}")
211           logger.trace(mdMetadataNode.toString)
212           MdMetadataSet.fromXml(mdMetadataNode, catalogueName)
213         })
214       }).filter(item => item.isDefined).map(item => item.get) //filter out all None values
215     })
216     gmdElementsFuture
217   }

219   /**
220     * Queries the Search Index
221     *
222     * @param query
223     * @return
224     */
225   def query(query: String): SearchResult = {
226 
227     // val ctx = SpatialContext.GEO
228     // val bboxStrategy: BBoxStrategy = BBoxStrategy.newInstance(ctx, "bboxStrategy")
229     // Query = LongPoint.newRangeQuery("dateStampCompare", localDate1.toEpochDay, localDate2.toEpochDay)
230 
231     val luceneQuery = query.trim() match {
232       case "" => new MatchAllDocsQuery()
233       case _ => {
234         val parser = new QueryParser("catch_all", new StandardAnalyzer())
235         //FIXME errorhandling
236         parser.parse(query)
237       }
238     }
239 
240     //FIXME SR when index empty, the next call will fail
241     val searcherManager = new SearcherManager(directory, null)
242     searcherManager.maybeRefreshBlocking()
243     val isearcher = searcherManager.acquire()
244 
245     val search = isearcher.search(luceneQuery, isearcher.getIndexReader.numDocs())
246     val scoreDocs = search.scoreDocs
247 
248     val header = SearchResultHeader(search.totalHits, luceneQuery.toString())
249     val results = scoreDocs.map(scoreDoc => {
250       val doc = isearcher.doc(scoreDoc.doc)
251       MdMetadataSet.fromLuceneDoc(doc)
252     })
253     //FIXME SR use ARM --> possible mem leak
254     searcherManager.release(isearcher)
255     SearchResult(header, results.toList)
256   }
