111     *
112     */
113   def buildIndex(catalogueName: String): Unit = {
114     logger.info(s"Building Lucene Index (AKKA VERSION) for $catalogueName")
115     val catalogueUrl = catalogues(catalogueName)
116     implicit val timeout: akka.util.Timeout = 5.seconds
117     val indexBuilder = (indexBuilderMaster ? GetSpecificIndexBuilder(catalogueName)).mapTo[ActorRef]
118     indexBuilder.onComplete({
119       case Success(indexBuilder) => indexBuilder ! IndexCatalogue(catalogueName, catalogueUrl)
120       case Failure(ex) => logger.warn(s"Exception during index build ${ex.getMessage}", ex)
121     })
122   }

124   def mergeIndex(directory: Directory, catalogueName: String): Unit = {
125     catalogueIndexes += (catalogueName -> directory)
126 
127     val config = new IndexWriterConfig()
128     config.setCommitOnClose(true)
129 
130     val iwriter = new IndexWriter(indexRamDirectory, config)
131     try {
132       logger.info(s"Merging ${catalogueIndexes.keys} into master index")
133       iwriter.deleteAll()
134       iwriter.commit()
135       iwriter.addIndexes(catalogueIndexes.values.toArray : _*)

145   /**
146     * Queries the Search Index
147     *
148     * @param query
149     * @return
150     */
151   def query(query: String,
152             bboxWtk: Option[String] = None,
153             fromDate: Option[LocalDate] = None,
154             toDate: Option[LocalDate] = None): List[MdMetadataSet] = {
155 
156     val textQuery = parseQueryString(query)
157 
158     val dateQuery = LongPoint.newRangeQuery("dateStamp",
159       fromDate.getOrElse(LocalDate.ofEpochDay(0)).toEpochDay(),
160       toDate.getOrElse(LocalDate.now()).toEpochDay())
161 
162     val bboxQuery = parseBboxQuery(bboxWtk.filterNot(_.isEmpty).getOrElse(WORLD_WKT))
163 
164     searcherManager.maybeRefreshBlocking()
165     val isearcher = searcherManager.acquire()
166 
167     // TODO TBD AK The way you built it, could omit dates or BBOX query constraint in this
168     // this builder if not provided through query params (but particularly date should NOT be limited to 1970 to now?)
169     val booleanQueryBuilder = new BooleanQuery.Builder()
170     booleanQueryBuilder.add(textQuery, BooleanClause.Occur.MUST)
171     booleanQueryBuilder.add(dateQuery, BooleanClause.Occur.MUST)
172     booleanQueryBuilder.add(bboxQuery, BooleanClause.Occur.MUST)
173     val luceneQuery = booleanQueryBuilder.build()
174 
175     val search = isearcher.search(luceneQuery, isearcher.getIndexReader.numDocs())
176     val scoreDocs = search.scoreDocs
177 
178     val results = scoreDocs.map(scoreDoc => {
179       val doc = isearcher.doc(scoreDoc.doc)
180       MdMetadataSet.fromLuceneDoc(doc)
181     }).toList
182 
183     //FIXME SR use ARM --> possible mem leak
184     searcherManager.release(isearcher)
185 
186     results
187   }

189   /**
190     * parses the query text
191     *
192     * @param queryString
193     * @return
194     */
195   private def parseQueryString(queryString: String) = {
196     queryString.trim() match {
197       case "" => new MatchAllDocsQuery()
198       case _ => {
199         val parser = new QueryParser("catch_all", new StandardAnalyzer())
200         //FIXME errorhandling
201         parser.parse(queryString)
202       }
203     }
204   }

206   /**
207     * creates a query, that should find anything inside a BBOX
208     *
209     * @param bboxWkt
210     * @return
211     */
212   private def parseBboxQuery(bboxWkt: String) = {
213     logger.debug(s"create query for $bboxWkt")
214 
215     val envelopeWktRegex =
216       "ENVELOPE\\(([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+)\\)".r
217 
218     val shape = bboxWkt.replaceAll("\\s", "") match {
219       case envelopeWktRegex(minX, maxX, minY, maxY) => {
220         MdMetadataSet.bboxFromCoords(minX.toFloat, maxX.toFloat, minY.toFloat, maxY.toFloat)
221       }
222       case _ => {
223         logger.warn(s"Could not parse WKT '$bboxWkt'. Using WORLD as query BBOX.");
224         MdMetadataSet.bboxFromCoords(-180, 180, -90, 90);
225       }
226     }
227 
228     logger.debug(s"parsed shape is ${shape.toString}")
229     val bboxStrategy: BBoxStrategy = BBoxStrategy.newInstance(SPATIAL_CONTEXT, "bbox")
230     bboxStrategy.makeQuery(new SpatialArgs(SpatialOperation.IsWithin, shape))
231   }

233   /**
234     * creates an empty index to make searches not crash before real documents are in the index.
235     *
236     * @param directory
237     */
238   private def prepareEmptyIndex(directory: Directory): Directory = {
239     val analyzer = new StandardAnalyzer()
240     val config = new IndexWriterConfig(analyzer)
241     config.setCommitOnClose(true)
242     //FIXME SR use SCALA_ARM (automated resource management)?
243     val iwriter = new IndexWriter(directory, config)
244     try {
245       iwriter.deleteAll()
246       iwriter.addDocument(new Document)
247       iwriter.commit()
248       logger.info("Empty Index ready")
249     }
250     finally {
251       iwriter.close()
252     }
253     directory
254   }
