236     *
237     * @param nodeSeq the provided xml
238     * @return a LoclDate from the list of extracted fields

240   def dateFromXml(nodeSeq: NodeSeq): LocalDate = {
241     val dateNodes = (nodeSeq \\ "dateStamp" \ "Date") ++
242       (nodeSeq \\ "dateStamp" \ "DateTime") ++
243       (nodeSeq \\ "date" \ "CI_Date" \ "date" \ "Date") ++
244       (nodeSeq \\ "date" \ "CI_Date" \ "date" \ "DateTime")

248   /**
249     * check if Time info in date/datestamp
250     * check decide dateStamp and/or/plus CI_Date date
251     * worst case create default date
252     *
253     * TODO AK ponder if precise ZoneInfo for parsedDate needed, alternatively either UTC or NZ TimeZone
254     *
255     * @param dateStrings
256     * @return
257     */
258   def dateFromStrings(dateStrings: List[String]): LocalDate = {
259     val yearMonthMatcher = """^\d\d\d\d-\d\d$""".r
260     val yearMatcher = """^\d\d\d\d$""".r
261     val isoInstantMatcher = """.*Z$""".r
262 
263     val acceptedDateFormats = List(
264       DateTimeFormatter.ISO_LOCAL_DATE_TIME, //2011-12-03T10:15:30
265       DateTimeFormatter.ISO_LOCAL_DATE, //2011-12-03
266       DateTimeFormatter.BASIC_ISO_DATE, //20111203
267       DateTimeFormatter.ISO_OFFSET_DATE, //2011-12-03+01:00
268       DateTimeFormatter.ISO_OFFSET_DATE_TIME //2011-12-03T10:15:30+01:00
269       //TODO SR DateTimeFormatter.ISO_INSTANT cannot be parsed into LocalDate but only LocalDateTime?
270       // DateTimeFormatter.ISO_INSTANT //2011-12-03T10:15:30Z
271     )
272 
273     val datesList = dateStrings.filter(!_.trim.isEmpty).flatMap(//iterate over parameter list
274       dateString => acceptedDateFormats.map(//try all parsers per date
275         df => {
276           try {
277             val correctedDateString = dateString match {
278               case yearMonthMatcher() => dateString.concat("-01")
279               case yearMatcher() => dateString.concat("-01-01")
280               case isoInstantMatcher() => dateString.dropRight(1)
281               case _ => dateString
282             }
283             Some(LocalDate.parse(correctedDateString, df))
284           }
285           catch {
286             case e: DateTimeParseException => None
287           }
288         }
289       ).filter(_.isDefined)).map(_.get) //filter None and remove the Option
290 
291     if (datesList.isEmpty) {
292       logger.warn(f"Could not parse any of the dates (${dateStrings.mkString(", ")})")
293       LocalDate.ofEpochDay(0) //1970-01-01
294     }
295     else {
296       if (datesList.size > 1) {
297         logger.info(f"Could parse ${datesList.size} of (${dateStrings.mkString(", ")}) only returning first success")
298       }
299       datesList.head
300     }
301   }

438     * @param nodeSeq xml nodeseq containing an extent element
439     * @return [[Rectangle]] the bounding box

441   def bboxFromXml(nodeSeq: NodeSeq): Rectangle = {
442     val bboxXml = (nodeSeq \\ "extent" \ "EX_Extent")
443     logger.trace(f"bboxFromXml: ${bboxXml}")
444     if (bboxXml.size == 1) {
445       val westText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "westBoundLongitude" \ "Decimal").text
446       val eastText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "eastBoundLongitude" \ "Decimal").text
447       val southText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "southBoundLatitude" \ "Decimal").text
448       val northText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "northBoundLatitude" \ "Decimal").text
449       val west = westText.toDoubleWithDefault(minLon)
450       val east = eastText.toDoubleWithDefault(maxLon)
451       val south = southText.toDoubleWithDefault(minLat)
452       val north = northText.toDoubleWithDefault(maxLat)
453       // Rectangle rect(double minX, double maxX, double minY, double maxY);
454       logger.debug(s"parsed bbox (${west}, ${east}, ${south}, ${north})")
455       bboxFromCoords(west, east, south, north)
456     }
457     else {
458       logger.warn(f"${(nodeSeq \\ "fileIdentifier" \ "CharacterString").text} has no BBOX")
459       bboxFromCoords(minLon, maxLon, minLat, maxLat)
460     }
461     //    bboxFromCoords(165, 360 - 175, -53.1, -28.8)

464   /**
465     * tries to build a bounding box rectangle as safely as possible from provided coordinates
466     * Rectangle rect(double minX, double maxX, double minY, double maxY);
467     * bboxFromCoords(west, east, south, north)
468     *
469     * @param west  most western value / minX
470     * @param east  most eastern value / maxX
471     * @param south most southern value / minY
472     * @param north most northern value / maxY
473     * @return the resulting bounding box
474     */
475   def bboxFromCoords(west: Double, east: Double, south: Double, north: Double): Rectangle = {
476     val (prunedWest, prunedEast) = pruneLongitudeValues(west, east)
477     val (prunedSouth, prunedNorth) = pruneLatitudeValues(south, north)
478 
479     val rect = ctx.getShapeFactory().rect(prunedWest, prunedEast, prunedSouth, prunedNorth)
480     logger.debug(s"parsed rect ${rect.toString}")
481     rect
482   }

484   /**
485     * tries to naively prune the provided coordinates into good shape for WSG84
486     * TODO DATE Line Wraps :-( ?
487     * Rectangle rect(double minX, double maxX, double minY, double maxY);
488     * bboxFromCoords(west, east, south, north)
489     *
490     * @param west most western value / minY
491     * @param east most eastern value / maxY
492     * @return tuple of viable coordinates in WSG84
493     */
494   def pruneLongitudeValues(west: Double, east: Double): (Double, Double) = {
495     if (math.abs(west - east) > math.abs(minLon - maxLon)) {
496       (minLon, maxLon) //in case the rectangle spans more than 360 deg make it world
497     }
498     else {
499       val result = List(west, east).map({ (value: Double) =>
500         value match {
501           case n if value >= minLon && value <= maxLon => n
502           case n if math.abs(value % math.abs(minLon - maxLon)) < maxLon => {
503             val result = value % maxLon
504             logger.info(f"changing longitude value $n to $result")
505             result
506           }
507           case _ => {
508             val result = math.signum(value) * minLon + (value % maxLon)
509             logger.info(f"changing longitude value $value to $result")
510             result
511           }
512         }
513       })
514       (result(0), result(1))
515     }
516   }

518   /**
519     * Cuts off latitudes outside of minLax / maxLat and swaps if south > north
520     *
521     * @param south most southern value / minY
522     * @param north most northern value / maxY
523     * @return tuple of viable coordinates
524     */
525   def pruneLatitudeValues(south: Double, north: Double): (Double, Double) = {
526     //min/max in tuples swaps north/south if necessary,
527     (Math.max(minLat, Math.min(south, north)),
528       Math.min(maxLat, Math.max(south, north)))
529   }

543   /**
544     * returns an Option[EmailAddress] object if parsing is successful
545     *
546     * @param emailString the string that might comprise of or contain an email address
547     * @return Option[EmailAddress]
548     */
549   def parseEmailStringtoEmailAddress(emailString: String): Option[EmailAddress] = {
550 
551     if (EmailAddress.isValid(emailString)) {
552       Some(EmailAddress(emailString))
553     } else {
554       if (emailString.contains(",")) {
555         val commaFree = emailString.split(",").head.trim
556 
557         Try {
558           EmailAddress(commaFree)
559         } match {
560           case Success(e) => Some(e)
561           case _ => None
562         }
563       } else {
