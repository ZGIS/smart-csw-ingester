37 /**
38   * Holds a subset of MDMetadata
39   *
40   * @param fileIdentifier the unique file identifier of the record
41   * @param dateStamp      a datestamp of the record
42   * @param title          title of the record
43   * @param abstrakt       abstract of the record
44   * @param keywords       list of keywords for the record
45   * @param topicCategories  list of ISO/ANZLIC topic categories of the record
46   * @param contactName    contact name for the record or dataset
47   * @param contactOrg     cantact organisation for the record or dataset
48   * @param contactEmail   contact email for the record or dataset
49   * @param license        licenses for the record or dataset
50   * @param bbox           bbox of the record
51   * @param origin         origin i.e. source catalogue where the record was loaded from

53 case class MdMetadataSet(fileIdentifier: String,
54                          dateStamp: LocalDate,
55                          title: String,
56                          abstrakt: String,
57                          keywords: List[String],
58                          topicCategories: List[String],
59                          contactName: String,
60                          contactOrg: String,
61                          contactEmail: String,
62                          license: String,
63                          bbox: Rectangle,
64                          origin: String) extends ClassnameLogger {
65   require(!fileIdentifier.trim.isEmpty, "FileIdentifier was empty")
66 
67   override def toString: String = {
68     f"""MdMetadataSet(
69         |$fileIdentifier,
70         |${dateStampAsIsoString},
71         |${title},
72         |${abstrakt},
73         |keywords(${keywords.mkString(", ")}),
74         |topicCategory(${topicCategories.mkString(", ")}),
75         |${contactName},
76         |${contactOrg},
77         |${contactEmail},
78         |${license},
79         |${bboxAsWkt},
80         |${origin})
81      """.stripMargin.replaceAll("\n", " ")
82   }

105     * returns the object as LuceneDocument
106     *
107     * @return LuceneDocument for index
108     */
109   def asLuceneDocument(): Document = {
110     val doc = new Document()

112     doc.add(new TextField("fileIdentifier", fileIdentifier, Field.Store.YES))
113     doc.add(new TextField("title", title, Field.Store.YES))

115     doc.add(new TextField("dateStampText", dateStampAsIsoString, Field.Store.YES))

117     val longDate = dateStamp.toEpochDay
118     doc.add(new LongPoint("dateStamp", longDate))

120     keywords.foreach(keyword => {
121       doc.add(new TextField("keywords", keyword, Field.Store.YES))

125       doc.add(new TextField("topicCategory", topicCategory, Field.Store.YES))

129     doc.add(new TextField("contactOrg", contactOrg, Field.Store.YES))
130     doc.add(new TextField("contactEmail", contactEmail, Field.Store.YES))
131     doc.add(new TextField("license", license, Field.Store.YES))

133     doc.add(new TextField("bboxText", bboxAsWkt, Field.Store.YES))

135     val ctx = SpatialContext.GEO
136     val bboxStrategy: BBoxStrategy = BBoxStrategy.newInstance(ctx, "bbox")
137     val bboxFields = bboxStrategy.createIndexableFields(bbox)
138     bboxFields.foreach(field => doc.add(field))
139 
140     doc.add(new StringField("origin", origin, Field.Store.YES))
141 
142     //FIXME decide if use catch_all field or how to build a query that queries all fields
143     doc.add(new TextField("catch_all", fileIdentifier, Field.Store.YES))
144     doc.add(new TextField("catch_all", title, Field.Store.YES))
145     doc.add(new TextField("catch_all", abstrakt, Field.Store.YES))
146     keywords.foreach(keyword => {
147       doc.add(new TextField("catch_all", keyword, Field.Store.YES))
148     })
149     topicCategories.foreach(topicCategor => {
150       doc.add(new TextField("catch_all", topicCategor, Field.Store.YES))
151     })
152     doc.add(new TextField("catch_all", contactName, Field.Store.YES))
153     doc.add(new TextField("catch_all", contactOrg, Field.Store.YES))
154     doc.add(new TextField("catch_all", contactEmail, Field.Store.YES))
155     doc.add(new TextField("catch_all", license, Field.Store.YES))
156     doc.add(new TextField("catch_all", origin, Field.Store.YES))
157 
158     doc
159   }

242   /**
243     * check if Time info in date/datestamp
244     * check decide dateStamp and/or/plus CI_Date date
245     * worst case create default date
246     *
247     * TODO AK ponder if precise ZoneInfo for parsedDate needed, alternatively either UTC or NZ TimeZone
248     *
249     * @param dateStrings
250     * @return
251     */
252   def dateFromStrings(dateStrings: List[String]): LocalDate = {
253     val yearMonthMatcher = """^\d\d\d\d-\d\d$""".r
254     val yearMatcher = """^\d\d\d\d$""".r
255     val isoInstantMatcher = """.*Z$""".r
256 
257     val acceptedDateFormats = List(
258       DateTimeFormatter.ISO_LOCAL_DATE_TIME, //2011-12-03T10:15:30
259       DateTimeFormatter.ISO_LOCAL_DATE, //2011-12-03
260       DateTimeFormatter.BASIC_ISO_DATE, //20111203
261       DateTimeFormatter.ISO_OFFSET_DATE, //2011-12-03+01:00
262       DateTimeFormatter.ISO_OFFSET_DATE_TIME //2011-12-03T10:15:30+01:00
263       //TODO SR DateTimeFormatter.ISO_INSTANT cannot be parsed into LocalDate but only LocalDateTime?
264       // DateTimeFormatter.ISO_INSTANT //2011-12-03T10:15:30Z
265     )
266 
267     val datesList = dateStrings.filter(!_.trim.isEmpty).flatMap(//iterate over parameter list
268       dateString => acceptedDateFormats.map(//try all parsers per date
269         df => {
270           try {
271             val correctedDateString = dateString match {
272               case yearMonthMatcher() => dateString.concat("-01")
273               case yearMatcher() => dateString.concat("-01-01")
274               case isoInstantMatcher() => dateString.dropRight(1)
275               case _ => dateString
276             }
277             Some(LocalDate.parse(correctedDateString, df))
278           }
279           catch {
280             case e: DateTimeParseException => None
281           }
282         }
283       ).filter(_.isDefined)).map(_.get) //filter None and remove the Option
284 
285     if (datesList.isEmpty) {
286       logger.warn(f"Could not parse and of the dates (${dateStrings.mkString(", ")})")
287       LocalDate.ofEpochDay(0) //1970-01-01
288     }
289     else {
290       if (datesList.size > 1) {
291         logger.warn(f"Could parse ${datesList.size} of (${dateStrings.mkString(", ")}) only returning first success")
292       }
293       datesList.head
294     }
295   }

380   /**
381     * tries to naively prune the provided coordinates into good shape for WSG84
382     * TODO DATE Line Wraps :-( ?
383     * Rectangle rect(double minX, double maxX, double minY, double maxY);
384     * bboxFromCoords(west, east, south, north)
385     *
386     * @param west most western value / minY
387     * @param east most eastern value / maxY
388     * @return tuple of viable coordinates in WSG84
389     */
390   def pruneLongitudeValues(west: Double, east: Double): (Double, Double) = {
391     if (math.abs(west - east) > math.abs(minLon - maxLon)) {
392       (minLon, maxLon) //in case the rectangle spans more than 360 deg make it world
393     }
394     else {
395       val result = List(west, east).map({ (value: Double) =>
396         value match {
397           case n if value >= minLon && value <= maxLon => n
398           case n if math.abs(value % math.abs(minLon - maxLon)) < maxLon => {
399             val result = value % maxLon
400             logger.warn(f"changing longitude value $n to $result")
401             result
402           }
403           case _ => {
404             val result = math.signum(value) * minLon + (value % maxLon)
405             logger.warn(f"changing longitude value $value to $result")
406             result
407           }
408         }
409       })
410       (result(0), result(1))
411     }
412 
413 
414   }

416   /**
417     * Cuts off latitudes outside of minLax / maxLat and swaps if south > north
418     *
419     * @param south most southern value / minY
420     * @param north most northern value / maxY
421     * @return tuple of viable coordinates
422     */
423   def pruneLatitudeValues(south: Double, north: Double): (Double, Double) = {
424     //min/max in tuples swaps north/south if necessary,
425     (Math.max(minLat, Math.min(south, north)),
426       Math.min(maxLat, Math.max(south, north)))
427   }

429   /**
430     * tries to build a bounding box rectangle as safely as possible from provided coordinates
431     * Rectangle rect(double minX, double maxX, double minY, double maxY);
432     * bboxFromCoords(west, east, south, north)
433     *
434     * @param west  most western value / minX
435     * @param east  most eastern value / maxX
436     * @param south most southern value / minY
437     * @param north most northern value / maxY
438     * @return the resulting bounding box
439     */
440   def bboxFromCoords(west: Double, east: Double, south: Double, north: Double): Rectangle = {
441     val (prunedWest, prunedEast) = pruneLongitudeValues(west, east)
442     val (prunedSouth, prunedNorth) = pruneLatitudeValues(south, north)
443 
444     val rect = ctx.getShapeFactory().rect(prunedWest, prunedEast, prunedSouth, prunedNorth)
445     logger.debug("PARSED RECT: " + rect.toString)
446     rect
447   }

463   /**
464     * tries to extract a valid and usable bounding box rectangle from xml
465     *
466     * @param nodeSeq xml nodeseq containing an extent element
467     * @return [[Rectangle]] the bounding box
468     */
469   def bboxFromXml(nodeSeq: NodeSeq): Rectangle = {
470     val bboxXml = (nodeSeq \\ "extent" \ "EX_Extent")
471     logger.trace(f"bboxFromXml: ${bboxXml}")
472     if (bboxXml.size == 1) {
473       val westText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "westBoundLongitude" \ "Decimal").text
474       val eastText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "eastBoundLongitude" \ "Decimal").text
475       val southText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "southBoundLatitude" \ "Decimal").text
476       val northText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "northBoundLatitude" \ "Decimal").text
477       val west = westText.toDoubleWithDefault(minLon)
478       val east = eastText.toDoubleWithDefault(maxLon)
479       val south = southText.toDoubleWithDefault(minLat)
480       val north = northText.toDoubleWithDefault(maxLat)
481       // Rectangle rect(double minX, double maxX, double minY, double maxY);
482       bboxFromCoords(west, east, south, north)
483     }
484     else {
485       logger.warn(f"${(nodeSeq \\ "fileIdentifier" \ "CharacterString").text} has no BBOX")
486       bboxFromCoords(minLon, maxLon, minLat, maxLat)
487     }
488 //    bboxFromCoords(165, 360 - 175, -53.1, -28.8)
489   }

498 object MdMetadataSetWriter extends Writes[MdMetadataSet] with ClassnameLogger {
499 
500   private lazy val ctx = SpatialContext.GEO
501   private lazy val jsWriter = ctx.getFormats().getWriter(ShapeIO.GeoJSON)
502 
503   /**
504     * provides the GeoJSON Polygon geometry from the feature's Rectangle bbox
505     *
506     * @param gmd MdMetadataSet
507     * @return JsValue with GeoJSON Polygon geometry
508     */
509   def getJsGeom(gmd: MdMetadataSet): JsValue = {
510     val geometry = jsWriter.toString(gmd.bbox)
511 
512     if (gmd.bbox.getCrossesDateLine) {
513       val newCoordinates = Json.arr(Json.arr(
514         Json.arr(gmd.bbox.getMinX, gmd.bbox.getMinY), //SW
515         Json.arr(gmd.bbox.getMinX, gmd.bbox.getMaxY), //NW
516         Json.arr(gmd.bbox.getMaxX+360, gmd.bbox.getMaxY), //NE
517         Json.arr(gmd.bbox.getMaxX+360, gmd.bbox.getMinY), //NW
518         Json.arr(gmd.bbox.getMinX, gmd.bbox.getMinY)  //SW
519       ))
520       Json.obj("type" -> "Polygon", "coordinates" -> newCoordinates)
521     }
522     else {
523       Json.parse(geometry)
524     }
525   }
526 
527   /**
528     * if bbox crosses dateline (east < west) then correct this for OL3
529  *
530     * @param bbox
531     * @return
532     */
533   def correctBbox(bbox: JsArray): JsArray = {
534     val west = bbox(0).asOpt[Double].get
535     val east = if (bbox(0).asOpt[Double].get > bbox(2).asOpt[Double].get) {
536                  bbox(2).asOpt[Double].get + 360
537                }
538                else {
539                  bbox(2).asOpt[Double].get
540                }
541     Json.arr(west, bbox(1).get, east, bbox(3).get)
542   }
543 
544   /**
545     * Converts [[MdMetadataSet]] object into [[JsObject]] as GeoJSON
546     */
547   def writes(gmd: MdMetadataSet): JsObject = {
548 
549     val properties = Json.obj(
550       "fileIdentifier" -> gmd.fileIdentifier,
551       "dateStamp" -> gmd.dateStampAsIsoString,
552       "title" -> gmd.title,
553       "abstrakt" -> gmd.abstrakt,
554       "keywords" -> gmd.keywords,
555       "topicCategories" -> gmd.topicCategories,
556       "contactName" -> gmd.contactName,
557       "contactOrg" -> gmd.contactOrg,
558       "contactEmail" -> gmd.contactEmail,
559       "license" -> gmd.license,
560       // extent is an array [10, 10, 40, 40] minX, maxX, maxY, minY
561       "bbox" -> correctBbox(Json.arr(
562         JsNumber(gmd.bbox.getMinX()),
563         JsNumber(gmd.bbox.getMinY()),
564         JsNumber(gmd.bbox.getMaxX()),
565         JsNumber(gmd.bbox.getMaxY()))
566       ),
567       "origin" -> gmd.origin
568     )
569 
570     Json.obj(
571       "type" -> "Feature",
572       "geometry" -> getJsGeom(gmd),
573       "properties" -> properties
574     )
575   }
576 }
