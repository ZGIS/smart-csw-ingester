37 /**
38   * Holds a subset of MDMetadata
39   *
40   * @param fileIdentifier the unique file identifier of the record
41   * @param dateStamp      a datestamp of the record
42   * @param title          title of the record
43   * @param abstrakt       abstract of the record
44   * @param keywords       list of keywords for the record
45   * @param topicCategories  list of ISO/ANZLIC topic categories of the record
46   * @param contactName    contact name for the record or dataset
47   * @param contactOrg     cantact organisation for the record or dataset
48   * @param contactEmail   contact email for the record or dataset
49   * @param license        licenses for the record or dataset
50   * @param bbox           bbox of the record
51   * @param origin         origin i.e. source catalogue where the record was loaded from
52   */
53 case class MdMetadataSet(fileIdentifier: String,
54                          dateStamp: LocalDate,
55                          title: String,
56                          abstrakt: String,
57                          keywords: List[String],
58                          topicCategories: List[String],
59                          contactName: String,
60                          contactOrg: String,
61                          contactEmail: String,
62                          license: String,
63                          bbox: Rectangle,
64                          linkage: List[String],
65                          origin: String) extends ClassnameLogger {
66   require(!fileIdentifier.trim.isEmpty, "FileIdentifier was empty")

70         |$fileIdentifier,
71         |${dateStampAsIsoString},
72         |${title},

75         |topicCategory(${topicCategories.mkString(", ")}),

81         |${linkage},

157     doc.add(new TextField("catch_all", contactOrg, Field.Store.YES))
158     doc.add(new TextField("catch_all", contactEmail, Field.Store.YES))
159     doc.add(new TextField("catch_all", license, Field.Store.YES))
160     doc.add(new TextField("catch_all", origin, Field.Store.YES))

163   }

248   /**
249     * check if Time info in date/datestamp
250     * check decide dateStamp and/or/plus CI_Date date
251     * worst case create default date
252     *
253     * TODO AK ponder if precise ZoneInfo for parsedDate needed, alternatively either UTC or NZ TimeZone
254     *
255     * @param dateStrings
256     * @return
257     */
258   def dateFromStrings(dateStrings: List[String]): LocalDate = {
259     val yearMonthMatcher = """^\d\d\d\d-\d\d$""".r
260     val yearMatcher = """^\d\d\d\d$""".r
261     val isoInstantMatcher = """.*Z$""".r
262 
263     val acceptedDateFormats = List(
264       DateTimeFormatter.ISO_LOCAL_DATE_TIME, //2011-12-03T10:15:30
265       DateTimeFormatter.ISO_LOCAL_DATE, //2011-12-03
266       DateTimeFormatter.BASIC_ISO_DATE, //20111203
267       DateTimeFormatter.ISO_OFFSET_DATE, //2011-12-03+01:00
268       DateTimeFormatter.ISO_OFFSET_DATE_TIME //2011-12-03T10:15:30+01:00
269       //TODO SR DateTimeFormatter.ISO_INSTANT cannot be parsed into LocalDate but only LocalDateTime?
270       // DateTimeFormatter.ISO_INSTANT //2011-12-03T10:15:30Z
271     )
272 
273     val datesList = dateStrings.filter(!_.trim.isEmpty).flatMap(//iterate over parameter list
274       dateString => acceptedDateFormats.map(//try all parsers per date
275         df => {
276           try {
277             val correctedDateString = dateString match {
278               case yearMonthMatcher() => dateString.concat("-01")
279               case yearMatcher() => dateString.concat("-01-01")
280               case isoInstantMatcher() => dateString.dropRight(1)
281               case _ => dateString
282             }
283             Some(LocalDate.parse(correctedDateString, df))
284           }
285           catch {
286             case e: DateTimeParseException => None
287           }
288         }
289       ).filter(_.isDefined)).map(_.get) //filter None and remove the Option
290 
291     if (datesList.isEmpty) {
292       logger.warn(f"Could not parse and of the dates (${dateStrings.mkString(", ")})")
293       LocalDate.ofEpochDay(0) //1970-01-01
294     }
295     else {
296       if (datesList.size > 1) {
297         logger.warn(f"Could parse ${datesList.size} of (${dateStrings.mkString(", ")}) only returning first success")
298       }
299       datesList.head
300     }
301   }

396   /**
397     * tries to naively prune the provided coordinates into good shape for WSG84
398     * TODO DATE Line Wraps :-( ?
399     * Rectangle rect(double minX, double maxX, double minY, double maxY);
400     * bboxFromCoords(west, east, south, north)
401     *
402     * @param west most western value / minY
403     * @param east most eastern value / maxY
404     * @return tuple of viable coordinates in WSG84
405     */
406   def pruneLongitudeValues(west: Double, east: Double): (Double, Double) = {
407     if (math.abs(west - east) > math.abs(minLon - maxLon)) {
408       (minLon, maxLon) //in case the rectangle spans more than 360 deg make it world
409     }
410     else {
411       val result = List(west, east).map({ (value: Double) =>
412         value match {
413           case n if value >= minLon && value <= maxLon => n
414           case n if math.abs(value % math.abs(minLon - maxLon)) < maxLon => {
415             val result = value % maxLon
416             logger.warn(f"changing longitude value $n to $result")
417             result
418           }
419           case _ => {
420             val result = math.signum(value) * minLon + (value % maxLon)
421             logger.warn(f"changing longitude value $value to $result")
422             result
423           }
424         }
425       })
426       (result(0), result(1))
427     }
428 
429 
430   }

432   /**
433     * Cuts off latitudes outside of minLax / maxLat and swaps if south > north
434     *
435     * @param south most southern value / minY
436     * @param north most northern value / maxY
437     * @return tuple of viable coordinates
438     */
439   def pruneLatitudeValues(south: Double, north: Double): (Double, Double) = {
440     //min/max in tuples swaps north/south if necessary,
441     (Math.max(minLat, Math.min(south, north)),
442       Math.min(maxLat, Math.max(south, north)))
443   }

445   /**
446     * tries to build a bounding box rectangle as safely as possible from provided coordinates
447     * Rectangle rect(double minX, double maxX, double minY, double maxY);
448     * bboxFromCoords(west, east, south, north)
449     *
450     * @param west  most western value / minX
451     * @param east  most eastern value / maxX
452     * @param south most southern value / minY
453     * @param north most northern value / maxY
454     * @return the resulting bounding box
455     */
456   def bboxFromCoords(west: Double, east: Double, south: Double, north: Double): Rectangle = {
457     val (prunedWest, prunedEast) = pruneLongitudeValues(west, east)
458     val (prunedSouth, prunedNorth) = pruneLatitudeValues(south, north)
459 
460     val rect = ctx.getShapeFactory().rect(prunedWest, prunedEast, prunedSouth, prunedNorth)
461     logger.debug(s"parsed rect ${rect.toString}")
462     rect
463   }

479   /**
480     * tries to extract a valid and usable bounding box rectangle from xml
481     *
482     * @param nodeSeq xml nodeseq containing an extent element
483     * @return [[Rectangle]] the bounding box
484     */
485   def bboxFromXml(nodeSeq: NodeSeq): Rectangle = {
486     val bboxXml = (nodeSeq \\ "extent" \ "EX_Extent")
487     logger.trace(f"bboxFromXml: ${bboxXml}")
488     if (bboxXml.size == 1) {
489       val westText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "westBoundLongitude" \ "Decimal").text
490       val eastText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "eastBoundLongitude" \ "Decimal").text
491       val southText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "southBoundLatitude" \ "Decimal").text
492       val northText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "northBoundLatitude" \ "Decimal").text
493       val west = westText.toDoubleWithDefault(minLon)
494       val east = eastText.toDoubleWithDefault(maxLon)
495       val south = southText.toDoubleWithDefault(minLat)
496       val north = northText.toDoubleWithDefault(maxLat)
497       // Rectangle rect(double minX, double maxX, double minY, double maxY);
498       logger.debug(s"parsed bbox (${west}, ${east}, ${south}, ${north})")
499       bboxFromCoords(west, east, south, north)
500     }
501     else {
502       logger.warn(f"${(nodeSeq \\ "fileIdentifier" \ "CharacterString").text} has no BBOX")
503       bboxFromCoords(minLon, maxLon, minLat, maxLat)
504     }
505 //    bboxFromCoords(165, 360 - 175, -53.1, -28.8)
506   }

510 /**
511   * GeoJSON writer for [[MdMetadataSet]]
512   *
513   * @see [[Writes]]
514   */
515 object MdMetadataSetWriter extends Writes[MdMetadataSet] with ClassnameLogger {
516 
517   private lazy val ctx = SpatialContext.GEO
518   private lazy val jsWriter = ctx.getFormats().getWriter(ShapeIO.GeoJSON)
519 
520   /**
521     * provides the GeoJSON Polygon geometry from the feature's Rectangle bbox
522     *
523     * @param gmd MdMetadataSet
524     * @return JsValue with GeoJSON Polygon geometry
525     */
526   def getJsGeom(gmd: MdMetadataSet): JsValue = {
527     val geometry = jsWriter.toString(gmd.bbox)
528 
529     if (gmd.bbox.getCrossesDateLine) {
530       val newCoordinates = Json.arr(Json.arr(
531         Json.arr(gmd.bbox.getMinX, gmd.bbox.getMinY), //SW
532         Json.arr(gmd.bbox.getMinX, gmd.bbox.getMaxY), //NW
533         Json.arr(gmd.bbox.getMaxX+360, gmd.bbox.getMaxY), //NE
534         Json.arr(gmd.bbox.getMaxX+360, gmd.bbox.getMinY), //NW
535         Json.arr(gmd.bbox.getMinX, gmd.bbox.getMinY)  //SW
536       ))
537       Json.obj("type" -> "Polygon", "coordinates" -> newCoordinates)
538     }
539     else {
540       Json.parse(geometry)
541     }
542   }
543 
544   /**
545     * if bbox crosses dateline (east < west) then correct this for OL3
546  *
547     * @param bbox
548     * @return
549     */
550   def correctBbox(bbox: JsArray): JsArray = {
551     val west = bbox(0).asOpt[Double].get
552     val east = if (bbox(0).asOpt[Double].get > bbox(2).asOpt[Double].get) {
553                  bbox(2).asOpt[Double].get + 360
554                }
555                else {
556                  bbox(2).asOpt[Double].get
557                }
558     Json.arr(west, bbox(1).get, east, bbox(3).get)
559   }
560 
561   /**
562     * Converts [[MdMetadataSet]] object into [[JsObject]] as GeoJSON
563     */
564   def writes(gmd: MdMetadataSet): JsObject = {
565 
566     val properties = Json.obj(
567       "fileIdentifier" -> gmd.fileIdentifier,
568       "dateStamp" -> gmd.dateStampAsIsoString,
569       "title" -> gmd.title,
570       "abstrakt" -> gmd.abstrakt,
571       "keywords" -> gmd.keywords,
572       "topicCategories" -> gmd.topicCategories,
573       "contactName" -> gmd.contactName,
574       "contactOrg" -> gmd.contactOrg,
575       "contactEmail" -> gmd.contactEmail,
576       "license" -> gmd.license,
577       // extent is an array [10, 10, 40, 40] minX, maxX, maxY, minY
578       "bbox" -> correctBbox(Json.arr(
579         JsNumber(gmd.bbox.getMinX()),
580         JsNumber(gmd.bbox.getMinY()),
581         JsNumber(gmd.bbox.getMaxX()),
582         JsNumber(gmd.bbox.getMaxY()))
583       ),
584       "linkage" -> gmd.linkage,
585       "origin" -> gmd.origin
586     )
587 
588     Json.obj(
589       "type" -> "Feature",
590       "geometry" -> getJsGeom(gmd),
591       "properties" -> properties
592     )
593   }
594 }
