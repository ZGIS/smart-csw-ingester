266   /**
267     * check if Time info in date/datestamp
268     * check decide dateStamp and/or/plus CI_Date date
269     * worst case create default date
270     *
271     * TODO AK ponder if precise ZoneInfo for parsedDate needed, alternatively either UTC or NZ TimeZone
272     *
273     * @param dateStrings
274     * @return
275     */
276   def dateFromStrings(dateStrings: List[String]): LocalDate = {
277     val yearMonthMatcher = """^\d\d\d\d-\d\d$""".r
278     val yearMatcher = """^\d\d\d\d$""".r
279     val isoInstantMatcher = """.*Z$""".r
280 
281     val acceptedDateFormats = List(
282       DateTimeFormatter.ISO_LOCAL_DATE_TIME, //2011-12-03T10:15:30
283       DateTimeFormatter.ISO_LOCAL_DATE, //2011-12-03
284       DateTimeFormatter.BASIC_ISO_DATE, //20111203
285       DateTimeFormatter.ISO_OFFSET_DATE, //2011-12-03+01:00
286       DateTimeFormatter.ISO_OFFSET_DATE_TIME //2011-12-03T10:15:30+01:00
287       //TODO SR DateTimeFormatter.ISO_INSTANT cannot be parsed into LocalDate but only LocalDateTime?
288       // DateTimeFormatter.ISO_INSTANT //2011-12-03T10:15:30Z
289     )
290 
291     val datesList = dateStrings.filter(!_.trim.isEmpty).flatMap(//iterate over parameter list
292       dateString => acceptedDateFormats.map(//try all parsers per date
293         df => {
294           try {
295             val correctedDateString = dateString match {
296               case yearMonthMatcher() => dateString.concat("-01")
297               case yearMatcher() => dateString.concat("-01-01")
298               case isoInstantMatcher() => dateString.dropRight(1)
299               case _ => dateString
300             }
301             Some(LocalDate.parse(correctedDateString, df))
302           }
303           catch {
304             case e: DateTimeParseException => None
305           }
306         }
307       ).filter(_.isDefined)).map(_.get) //filter None and remove the Option
308 
309     if (datesList.isEmpty) {
310       logger.warn(f"Could not parse and of the dates (${dateStrings.mkString(", ")})")
311       LocalDate.ofEpochDay(0) //1970-01-01
312     }
313     else {
314       if (datesList.size > 1) {
315         logger.info(f"Could parse ${datesList.size} of (${dateStrings.mkString(", ")}) only returning first success")
316       }
317       datesList.head
318     }
319   }

447   /**
448     * tries to naively prune the provided coordinates into good shape for WSG84
449     * TODO DATE Line Wraps :-( ?
450     * Rectangle rect(double minX, double maxX, double minY, double maxY);
451     * bboxFromCoords(west, east, south, north)
452     *
453     * @param west most western value / minY
454     * @param east most eastern value / maxY
455     * @return tuple of viable coordinates in WSG84
456     */
457   def pruneLongitudeValues(west: Double, east: Double): (Double, Double) = {
458     if (math.abs(west - east) > math.abs(minLon - maxLon)) {
459       (minLon, maxLon) //in case the rectangle spans more than 360 deg make it world
460     }
461     else {
462       val result = List(west, east).map({ (value: Double) =>
463         value match {
464           case n if value >= minLon && value <= maxLon => n
465           case n if math.abs(value % math.abs(minLon - maxLon)) < maxLon => {
466             val result = value % maxLon
467             logger.info(f"changing longitude value $n to $result")
468             result
469           }
470           case _ => {
471             val result = math.signum(value) * minLon + (value % maxLon)
472             logger.info(f"changing longitude value $value to $result")
473             result
474           }
475         }
476       })
477       (result(0), result(1))
478     }

483   /**
484     * Cuts off latitudes outside of minLax / maxLat and swaps if south > north
485     *
486     * @param south most southern value / minY
487     * @param north most northern value / maxY
488     * @return tuple of viable coordinates
489     */
490   def pruneLatitudeValues(south: Double, north: Double): (Double, Double) = {
491     //min/max in tuples swaps north/south if necessary,
492     (Math.max(minLat, Math.min(south, north)),
493       Math.min(maxLat, Math.max(south, north)))
494   }

496   /**
497     * tries to build a bounding box rectangle as safely as possible from provided coordinates
498     * Rectangle rect(double minX, double maxX, double minY, double maxY);
499     * bboxFromCoords(west, east, south, north)
500     *
501     * @param west  most western value / minX
502     * @param east  most eastern value / maxX
503     * @param south most southern value / minY
504     * @param north most northern value / maxY
505     * @return the resulting bounding box
506     */
507   def bboxFromCoords(west: Double, east: Double, south: Double, north: Double): Rectangle = {
508     val (prunedWest, prunedEast) = pruneLongitudeValues(west, east)
509     val (prunedSouth, prunedNorth) = pruneLatitudeValues(south, north)
510 
511     val rect = ctx.getShapeFactory().rect(prunedWest, prunedEast, prunedSouth, prunedNorth)
512     logger.debug(s"parsed rect ${rect.toString}")
513     rect
514   }

530   /**
531     * tries to extract a valid and usable bounding box rectangle from xml
532     *
533     * @param nodeSeq xml nodeseq containing an extent element
534     * @return [[Rectangle]] the bounding box
535     */
536   def bboxFromXml(nodeSeq: NodeSeq): Rectangle = {
537     val bboxXml = (nodeSeq \\ "extent" \ "EX_Extent")
538     logger.trace(f"bboxFromXml: ${bboxXml}")
539     if (bboxXml.size == 1) {
540       val westText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "westBoundLongitude" \ "Decimal").text
541       val eastText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "eastBoundLongitude" \ "Decimal").text
542       val southText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "southBoundLatitude" \ "Decimal").text
543       val northText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "northBoundLatitude" \ "Decimal").text
544       val west = westText.toDoubleWithDefault(minLon)
545       val east = eastText.toDoubleWithDefault(maxLon)
546       val south = southText.toDoubleWithDefault(minLat)
547       val north = northText.toDoubleWithDefault(maxLat)
548       // Rectangle rect(double minX, double maxX, double minY, double maxY);
549       logger.debug(s"parsed bbox (${west}, ${east}, ${south}, ${north})")
550       bboxFromCoords(west, east, south, north)
551     }
552     else {
553       logger.warn(f"${(nodeSeq \\ "fileIdentifier" \ "CharacterString").text} has no BBOX")
554       bboxFromCoords(minLon, maxLon, minLat, maxLat)
555     }
556 //    bboxFromCoords(165, 360 - 175, -53.1, -28.8)
557   }

571   /**

574     * @param gmd MdMetadataSet
575     * @return JsValue with GeoJSON Polygon geometry

577   def getJsGeom(gmd: MdMetadataSet): JsValue = {
578     val geometry = jsWriter.toString(gmd.bbox)
579 
580     if (gmd.bbox.getCrossesDateLine) {
581       val newCoordinates = Json.arr(Json.arr(
582         Json.arr(gmd.bbox.getMinX, gmd.bbox.getMinY), //SW
583         Json.arr(gmd.bbox.getMinX, gmd.bbox.getMaxY), //NW
584         Json.arr(gmd.bbox.getMaxX+360, gmd.bbox.getMaxY), //NE
585         Json.arr(gmd.bbox.getMaxX+360, gmd.bbox.getMinY), //NW
586         Json.arr(gmd.bbox.getMinX, gmd.bbox.getMinY)  //SW
587       ))
588       Json.obj("type" -> "Polygon", "coordinates" -> newCoordinates)
589     }
590     else {
591       Json.parse(geometry)
592     }
