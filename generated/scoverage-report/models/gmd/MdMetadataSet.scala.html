37 /**
38   * Holds a subset of MDMetadata
39   *
40   * @param fileIdentifier the unique file identifier of the record
41   * @param dateStamp      a datestamp of the record
42   * @param title          title of the record
43   * @param abstrakt       abstract of the record
44   * @param keywords       list of keywords for the record
45   * @param topicCategory  list of ISO/ANZLIC topic categories of the record
46   * @param contactName    contact name for the record or dataset
47   * @param contactOrg     cantact organisation for the record or dataset
48   * @param contactEmail   contact email for the record or dataset
49   * @param license        licenses for the record or dataset

53 case class MdMetadataSet(fileIdentifier: String,
54                          dateStamp: LocalDate,
55                          title: String,
56                          abstrakt: String,
57                          keywords: List[String],
58                          topicCategory: List[String],
59                          contactName: String,
60                          contactOrg: String,
61                          contactEmail: String,
62                          license: String,
63                          bbox: Rectangle,
64                          origin: String) extends ClassnameLogger {
65   require(!fileIdentifier.trim.isEmpty, "FileIdentifier was empty")
66 
67   override def toString: String = {
68     f"""MdMetadataSet(
69         |$fileIdentifier,
70         |${dateStampAsIsoString},
71         |${title},
72         |${abstrakt},

74         |topicCategory(${topicCategory.mkString(", ")}),
75         |${contactName},
76         |${contactOrg},
77         |${contactEmail},
78         |${license},
79         |${bboxAsWkt},
80         |${origin})
81      """.stripMargin.replaceAll("\n", " ")
82   }

105     * returns the object as LuceneDocument

142     doc.add(new TextField("catch_all", keywords.mkString(" "), Field.Store.YES))
143     doc.add(new TextField("catch_all", topicCategory.mkString(" "), Field.Store.YES))
144     doc.add(new TextField("catch_all", contactName, Field.Store.YES))
145     doc.add(new TextField("catch_all", contactOrg, Field.Store.YES))
146     doc.add(new TextField("catch_all", contactEmail, Field.Store.YES))
147     doc.add(new TextField("catch_all", license, Field.Store.YES))
148     doc.add(new TextField("catch_all", origin, Field.Store.YES))

234   /**
235     * check if Time info in date/datestamp
236     * check decide dateStamp and/or/plus CI_Date date
237     * worst case create default date
238     *
239     * TODO AK ponder if precise ZoneInfo for parsedDate needed, alternatively either UTC or NZ TimeZone
240     *
241     * @param dateStrings
242     * @return
243     */
244   def dateFromStrings(dateStrings: List[String]): LocalDate = {
245     val yearMonthMatcher = """^\d\d\d\d-\d\d$""".r
246     val yearMatcher = """^\d\d\d\d$""".r
247     val isoInstantMatcher = """.*Z$""".r
248 
249     val acceptedDateFormats = List(
250       DateTimeFormatter.ISO_LOCAL_DATE_TIME, //2011-12-03T10:15:30
251       DateTimeFormatter.ISO_LOCAL_DATE, //2011-12-03
252       DateTimeFormatter.BASIC_ISO_DATE, //20111203
253       DateTimeFormatter.ISO_OFFSET_DATE, //2011-12-03+01:00
254       DateTimeFormatter.ISO_OFFSET_DATE_TIME //2011-12-03T10:15:30+01:00
255       // DateTimeFormatter.ISO_INSTANT //2011-12-03T10:15:30Z - this cannot be parsed into LocalDate but only LocalDateTime?
256     )
257 
258     val datesList = dateStrings.filter(!_.trim.isEmpty).flatMap(//iterate over parameter list
259       dateString => acceptedDateFormats.map(//try all parsers per date
260         df => {
261           try {
262             val correctedDateString = dateString match {
263               case yearMonthMatcher() => dateString.concat("-01")
264               case yearMatcher() => dateString.concat("-01-01")
265               case isoInstantMatcher() => dateString.dropRight(1)
266               case _ => dateString
267             }
268             Some(LocalDate.parse(correctedDateString, df))
269           }
270           catch {
271             case e: DateTimeParseException => None
272           }
273         }
274       ).filter(_.isDefined)).map(_.get) //filter None and remove the Option
275 
276     if (datesList.isEmpty) {
277       logger.warn(f"Could not parse and of the dates (${dateStrings.mkString(", ")})")
278       LocalDate.ofEpochDay(0) //1970-01-01
279     }
280     else {
281       if (datesList.size > 1) {
282         logger.warn(f"Could parse ${datesList.size} of (${dateStrings.mkString(", ")}) only returning first success")
283       }
284       datesList.head
285     }
286   }

371   /**
372     * tries to naively prune the provided coordinates into good shape for WSG84
373     * TODO DATE Line Wraps :-( ?
374     * Rectangle rect(double minX, double maxX, double minY, double maxY);
375     * bboxFromCoords(west, east, south, north)
376     *
377     * @param west most western value / minY
378     * @param east most eastern value / maxY
379     * @return tuple of viable coordinates in WSG84
380     */
381   def pruneLongitudeValues(west: Double, east: Double): (Double, Double) = {
382     val result = List(west, east).map({ (value: Double) =>
383       value match {
384         case n if (math.abs(n) - 180) > 0 => {
385           //if val > 180 (outside geo box)
386           val result = math.signum(n) * 180 //cut it down to +/-180
387           logger.warn(f"cutting value: $n to $result")
388           result
389         }
390         case _ => value
391       }
392     })
393     (result(0), result(1))
394   }
395 
396   /**
397     * tries to naively prune the provided coordinates into good shape for WSG84
398     *
399     * @param south most southern value / minY
400     * @param north most northern value / maxY
401     * @return tuple of viable coordinates in WSG84
402     */
403   def pruneLatitudeValues(south: Double, north: Double): (Double, Double) = {
404     val result = List(south, north).map({ (value: Double) =>
405       value match {
406         case n if (math.abs(n) - 90) > 0 => {
407           //if val > 90 (outside geo box)
408           val result = math.signum(n) * 90 //cut it down to +/-90
409           logger.warn(f"cutting value: $n to $result")
410           result
411         }
412         case _ => value
413       }
414     })
415     if (result(0) > result(1)) {
416       logger.warn(s"South ${result(0)} was bigger than North(${result(1)}). Swapping.")
417       (result(1), result(0))
418     }
419     else {
420       (result(0), result(1))
421     }
422   }

424   /**
425     * tries to build a bounding box rectangle as safely as possible from provided coordinates
426     * Rectangle rect(double minX, double maxX, double minY, double maxY);
427     * bboxFromCoords(west, east, south, north)
428     *
429     * @param west  most western value / minX
430     * @param east  most eastern value / maxX
431     * @param south most southern value / minY
432     * @param north most northern value / maxY
433     * @return the resulting bounding box
434     */
435   def bboxFromCoords(west: Double, east: Double, south: Double, north: Double): Rectangle = {
436     val (prunedWest, prunedEast) = pruneLongitudeValues(west, east)
437     val (prunedSouth, prunedNorth) = pruneLatitudeValues(south, north)
438 
439     val rect = ctx.getShapeFactory().rect(prunedWest, prunedEast, prunedSouth, prunedNorth)
440     logger.debug("PARSED RECT: " + rect.toString)
441     rect
442   }

458   /**
459     * tries to extract a valid and usable bounding box rectangle from xml
460     *
461     * @param nodeSeq xml nodeseq containing an extent element
462     * @return [[Rectangle]] the bounding box
463     */
464   def bboxFromXml(nodeSeq: NodeSeq): Rectangle = {
465     val bboxXml = (nodeSeq \\ "extent" \ "EX_Extent")
466     logger.trace(f"bboxFromXml: ${bboxXml}")
467     if (bboxXml.size == 1) {
468       val westText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "westBoundLongitude" \ "Decimal").text
469       val eastText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "eastBoundLongitude" \ "Decimal").text
470       val southText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "southBoundLatitude" \ "Decimal").text
471       val northText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "northBoundLatitude" \ "Decimal").text
472       val west = westText.toDoubleWithDefault(minLon)
473       val east = eastText.toDoubleWithDefault(maxLon)
474       val south = southText.toDoubleWithDefault(minLat)
475       val north = northText.toDoubleWithDefault(maxLat)
476       // Rectangle rect(double minX, double maxX, double minY, double maxY);
477       bboxFromCoords(west, east, south, north)
478     }
479     else {
480       logger.warn(f"${(nodeSeq \\ "fileIdentifier" \ "CharacterString").text} has no BBOX")
481       bboxFromCoords(minLon, maxLon, minLat, maxLat)
482     }
483 //    bboxFromCoords(165, 360 - 175, -53.1, -28.8)
484   }

553   lazy val WORLD = ctx.getShapeFactory().rect(-180, 180, -90, 90)

555   /**

558     * @param gmdList List [[MdMetadataSet]]

561   def getArrayOfFeatures(gmdList: List[MdMetadataSet]): JsValue = {

566   /**
567     * computes the overall bounding box for the featurecollection
568     *
569     * @param gmdList List [[MdMetadataSet]]
570     * @return JsArray with 4 bbox double values (e, w, s, n)
571     */
572   def getBoundingBox(gmdList: List[MdMetadataSet]): JsValue = {
573     import collection.JavaConverters._
574     import collection.mutable._
575 
576     logger.debug(f"gmdList.size() ${gmdList.size}")
577 
578     val shapeBuffer: scala.collection.mutable.Buffer[Rectangle] = Buffer[Rectangle]()
579     gmdList.foreach(gmd =>
580       shapeBuffer.append(gmd.bbox)
581     )
582     logger.debug(f"shapeBuffer.size() ${shapeBuffer.size}")
583 
584     val shapeList: java.util.List[Rectangle] = new util.ArrayList[Rectangle]
585     shapeList.addAll(shapeBuffer.asJava)
586     logger.debug(f"shapeList.size() ${shapeList.size}")
587     // shapes - Copied by reference! (make a defensive copy if caller modifies), also needs RandomAccess
588     // https://locationtech.github.io/spatial4j/apidocs/org/locationtech/spatial4j/shape/ShapeCollection.html
589     val shapeCollection = new ShapeCollection[Rectangle](shapeList, ctx)
590     logger.debug(f"shapeCollection.size() ${shapeCollection.size}")
591 
592     val envelope = if (shapeCollection.size > 0) {
593       shapeCollection.getBoundingBox
594     }
595                    else {
596                      WORLD
597                    }
598 
599     Json.arr(
600       JsNumber(envelope.getMinX),
601       JsNumber(envelope.getMaxX),
602       JsNumber(envelope.getMaxY),
603       JsNumber(envelope.getMinY)
604     )
605   }

607   /**
608     * Converts List of [[MdMetadataSet]] object into [[JsObject]] as GeoJSON
609     */
610   def writes(gmdList: List[MdMetadataSet]): JsObject = {
