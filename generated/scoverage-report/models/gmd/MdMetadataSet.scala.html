234   /**
235     * check if Time info in date/datestamp
236     * check decide dateStamp and/or/plus CI_Date date
237     * worst case create default date
238     *
239     * TODO AK ponder if precise ZoneInfo for parsedDate needed, alternatively either UTC or NZ TimeZone
240     *
241     * @param dateStrings
242     * @return
243     */
244   def dateFromStrings(dateStrings: List[String]): LocalDate = {
245     val yearMonthMatcher = """^\d\d\d\d-\d\d$""".r
246     val yearMatcher = """^\d\d\d\d$""".r
247     val isoInstantMatcher = """.*Z$""".r
248 
249     val acceptedDateFormats = List(
250       DateTimeFormatter.ISO_LOCAL_DATE_TIME, //2011-12-03T10:15:30
251       DateTimeFormatter.ISO_LOCAL_DATE, //2011-12-03
252       DateTimeFormatter.BASIC_ISO_DATE, //20111203
253       DateTimeFormatter.ISO_OFFSET_DATE, //2011-12-03+01:00
254       DateTimeFormatter.ISO_OFFSET_DATE_TIME //2011-12-03T10:15:30+01:00
255       // DateTimeFormatter.ISO_INSTANT //2011-12-03T10:15:30Z - this cannot be parsed into LocalDate but only LocalDateTime?
256     )
257 
258     val datesList = dateStrings.filter(!_.trim.isEmpty).flatMap(//iterate over parameter list
259       dateString => acceptedDateFormats.map(//try all parsers per date
260         df => {
261           try {
262             val correctedDateString = dateString match {
263               case yearMonthMatcher() => dateString.concat("-01")
264               case yearMatcher() => dateString.concat("-01-01")
265               case isoInstantMatcher() => dateString.dropRight(1)
266               case _ => dateString
267             }
268             Some(LocalDate.parse(correctedDateString, df))
269           }
270           catch {
271             case e: DateTimeParseException => None
272           }
273         }
274       ).filter(_.isDefined)).map(_.get) //filter None and remove the Option
275 
276     if (datesList.isEmpty) {
277       logger.warn(f"Could not parse and of the dates (${dateStrings.mkString(", ")})")
278       LocalDate.ofEpochDay(0) //1970-01-01
279     }
280     else {
281       if (datesList.size > 1) {
282         logger.warn(f"Could parse ${datesList.size} of (${dateStrings.mkString(", ")}) only returning first success")
283       }
284       datesList.head
285     }
286   }
