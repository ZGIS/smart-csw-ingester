40 /**
41   * Holds a subset of MDMetadata
42   *
43   * @param fileIdentifier the unique file identifier of the record
44   * @param dateStamp      a datestamp of the record
45   * @param title          title of the record
46   * @param abstrakt       abstract of the record
47   * @param keywords       list of keywords for the record
48   * @param topicCategory  list of ISO/ANZLIC topic categories of the record
49   * @param contactName    contact name for the record or dataset
50   * @param contactOrg     cantact organisation for the record or dataset
51   * @param contactEmail   contact email for the record or dataset
52   * @param license        licenses for the record or dataset
53   * @param bbox           bbox of the record
54   * @param origin         origin i.e. source catalogue where the record was loaded from
55   */
56 case class MdMetadataSet(fileIdentifier: String,
57                          dateStamp: LocalDate,
58                          title: String,
59                          abstrakt: String,
60                          keywords: List[String],
61                          topicCategory: List[String],
62                          contactName: String,
63                          contactOrg: String,
64                          contactEmail: String,
65                          license: String,
66                          bbox: Rectangle,
67                          origin: String) extends ClassnameLogger {
68   require(!fileIdentifier.trim.isEmpty, "FileIdentifier was empty")
69 
70   override def toString: String = {
71     f"""MdMetadataSet(
72         |$fileIdentifier,
73         |${dateStampAsIsoString},
74         |${title},
75         |${abstrakt},
76         |keywords(${keywords.mkString(", ")}),
77         |topicCategory(${topicCategory.mkString(", ")}),
78         |${contactName},
79         |${contactOrg},
80         |${contactEmail},
81         |${license},
82         |${bboxAsWkt},
83         |${origin})
84      """.stripMargin.replaceAll("\n", " ")
85   }

108     * returns the object as LuceneDocument
109     *
110     * @return LuceneDocument for index

128     doc.add(new TextField("contactOrg", contactOrg, Field.Store.YES))
129     doc.add(new TextField("contactEmail", contactEmail, Field.Store.YES))
130     doc.add(new TextField("license", license, Field.Store.YES))

136     val bboxFields = bboxStrategy.createIndexableFields(bbox)
137     bboxFields.foreach(field => doc.add(field))
138 
139     doc.add(new StringField("origin", origin, Field.Store.YES))
140 
141     //FIXME decide if use catch_all field or how to build a query that queries all fields
142     doc.add(new TextField("catch_all", fileIdentifier, Field.Store.YES))
143     doc.add(new TextField("catch_all", title, Field.Store.YES))
144     doc.add(new TextField("catch_all", abstrakt, Field.Store.YES))
145     doc.add(new TextField("catch_all", keywords.mkString(" "), Field.Store.YES))
146     doc.add(new TextField("catch_all", topicCategory.mkString(" "), Field.Store.YES))
147     doc.add(new TextField("catch_all", contactName, Field.Store.YES))
148     doc.add(new TextField("catch_all", contactOrg, Field.Store.YES))
149     doc.add(new TextField("catch_all", contactEmail, Field.Store.YES))
150     doc.add(new TextField("catch_all", license, Field.Store.YES))
151     doc.add(new TextField("catch_all", origin, Field.Store.YES))
152 
153     doc
154   }

237   /**
238     * check if Time info in date/datestamp
239     * check decide dateStamp and/or/plus CI_Date date
240     * worst case create default date
241     *
242     * TODO AK ponder if precise ZoneInfo for parsedDate needed, alternatively either UTC or NZ TimeZone
243     *
244     * @param dateStrings
245     * @return
246     */
247   def dateFromStrings(dateStrings: List[String]): LocalDate = {
248     val yearMonthMatcher = """^\d\d\d\d-\d\d$""".r
249     val yearMatcher = """^\d\d\d\d$""".r
250     val isoInstantMatcher = """.*Z$""".r
251 
252     val acceptedDateFormats = List(
253       DateTimeFormatter.ISO_LOCAL_DATE_TIME, //2011-12-03T10:15:30
254       DateTimeFormatter.ISO_LOCAL_DATE, //2011-12-03
255       DateTimeFormatter.BASIC_ISO_DATE, //20111203
256       DateTimeFormatter.ISO_OFFSET_DATE, //2011-12-03+01:00
257       DateTimeFormatter.ISO_OFFSET_DATE_TIME //2011-12-03T10:15:30+01:00
258       // DateTimeFormatter.ISO_INSTANT //2011-12-03T10:15:30Z - this cannot be parsed into LocalDate but only LocalDateTime?
259     )
260 
261     val datesList = dateStrings.filter(!_.trim.isEmpty).flatMap(//iterate over parameter list
262       dateString => acceptedDateFormats.map(//try all parsers per date
263         df => {
264           try {
265             val correctedDateString = dateString match {
266               case yearMonthMatcher() => dateString.concat("-01")
267               case yearMatcher() => dateString.concat("-01-01")
268               case isoInstantMatcher() => dateString.dropRight(1)
269               case _ => dateString
270             }
271             Some(LocalDate.parse(correctedDateString, df))
272           }
273           catch {
274             case e: DateTimeParseException => None
275           }
276         }
277       ).filter(_.isDefined)).map(_.get) //filter None and remove the Option
278 
279     if (datesList.isEmpty) {
280       logger.warn(f"Could not parse and of the dates (${dateStrings.mkString(", ")})")
281       LocalDate.ofEpochDay(0) //1970-01-01
282     }
283     else {
284       if (datesList.size > 1) {
285         logger.warn(f"Could parse ${datesList.size} of (${dateStrings.mkString(", ")}) only returning first success")
286       }
287       datesList.head
288     }
289   }

374   /**
375     * tries to naively prune the provided coordinates into good shape for WSG84
376     * TODO DATE Line Wraps :-( ?
377     * Rectangle rect(double minX, double maxX, double minY, double maxY);
378     * bboxFromCoords(west, east, south, north)
379     *
380     * @param west most western value / minY
381     * @param east most eastern value / maxY
382     * @return tuple of viable coordinates in WSG84
383     */
384   def pruneLongitudeValues(west: Double, east: Double): (Double, Double) = {
385     val x1 = west match {
386       case n: Double if (n < minLon) => {
387         logger.warn(f"cutting west value: $west to $minLon")
388         minLon
389       }
390       case n: Double if (n > maxLon) => {
391         logger.warn(f"cutting west value: $west to $maxLon")
392         maxLon
393       }
394       case _ => {
395         // if all good then return value
396         west
397       }
398     }
399     val x2 = east match {
400       case n: Double if (n < minLon) => {
401         logger.warn(f"cutting east value: $east to $minLon")
402         minLon
403       }
404       case n: Double if (n > maxLon) => {
405         logger.warn(f"cutting east value: $east to $maxLon")
406         maxLon
407       }
408       case _ => {
409         // if all good then return value
410         east
411       }
412     }
413     (x1, x2)
414   }

416   /**
417     * tries to naively prune the provided coordinates into good shape for WSG84
418     *
419     * @param south most southern value / minY
420     * @param north most northern value / maxY
421     * @return tuple of viable coordinates in WSG84
422     */
423   def pruneLatitudeValues(south: Double, north: Double): (Double, Double) = {
424     val x1 = south match {
425       case n: Double if (n < minLat) => {
426         logger.warn(f"cutting south value: $south to $minLat")
427         minLat
428       }
429       case n: Double if (n > maxLat) => {
430         logger.warn(f"cutting south value: $south to $maxLat")
431         maxLat
432       }
433       case n: Double if (n > north) => {
434         logger.warn(f"switching south value: $south to $north")
435         north
436       }
437       case _ => {
438         // if all good then return value
439         south
440       }
441     }
442     val x2 = north match {
443       case n: Double if (n < minLat) => {
444         logger.warn(f"cutting north value: $north to $minLat")
445         minLat
446       }
447       case n: Double if (n > maxLat) => {
448         logger.warn(f"cutting north value: $north to $maxLat")
449         maxLat
450       }
451       case n: Double if (n < south) => {
452         logger.warn(f"switching north value: $north to $south")
453         south
454       }
455       case _ => {
456         // if all good then return value
457         north
458       }
459     }
460     (x1, x2)
461   }

463   /**
464     * tries to build a bounding box rectangle as safely as possible from provided coordinates
465     * Rectangle rect(double minX, double maxX, double minY, double maxY);
466     * bboxFromCoords(west, east, south, north)
467     *
468     * @param west  most western value / minX
469     * @param east  most eastern value / maxX
470     * @param south most southern value / minY
471     * @param north most northern value / maxY
472     * @return the resulting bounding box
473     */
474   def bboxFromCoords(west: Double, east: Double, south: Double, north: Double): Rectangle = {
475     val (prunedWest, prunedEast) = pruneLongitudeValues(west, east)
476     val (prunedSouth, prunedNorth) = pruneLatitudeValues(south, north)
477 
478     val rect = ctx.getShapeFactory().rect(prunedWest, prunedEast, prunedSouth, prunedNorth)
479     logger.error("PARSED RECT: " + rect.toString)
480     rect
481   }

497   /**
498     * tries to extract a valid and usable bounding box rectangle from xml
499     *
500     * @param nodeSeq xml nodeseq containing an extent element
501     * @return [[Rectangle]] the bounding box
502     */
503   def bboxFromXml(nodeSeq: NodeSeq): Rectangle = {
504     val bboxXml = (nodeSeq \\ "extent" \ "EX_Extent")
505     logger.trace(f"bboxFromXml: ${bboxXml}")
506     if (bboxXml.size == 1) {
507       val westText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "westBoundLongitude" \ "Decimal").text
508       val eastText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "eastBoundLongitude" \ "Decimal").text
509       val southText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "southBoundLatitude" \ "Decimal").text
510       val northText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "northBoundLatitude" \ "Decimal").text
511       val west = westText.toDoubleWithDefault(minLon)
512       val east = eastText.toDoubleWithDefault(maxLon)
513       val south = southText.toDoubleWithDefault(minLat)
514       val north = northText.toDoubleWithDefault(maxLat)
515       // Rectangle rect(double minX, double maxX, double minY, double maxY);
516       bboxFromCoords(west, east, south, north)
517     }
518     else {
519       logger.warn(f"${(nodeSeq \\ "fileIdentifier" \ "CharacterString").text} has no BBOX")
520       bboxFromCoords(minLon, maxLon, minLat, maxLat)
521     }
522   }

610   def getBoundingBox(gmdList: List[MdMetadataSet]): JsValue = {
611     import collection.JavaConverters._

613 
614     logger.debug(f"gmdList.size() ${gmdList.size}")

616     val shapeBuffer: scala.collection.mutable.Buffer[Rectangle] = Buffer[Rectangle]()
