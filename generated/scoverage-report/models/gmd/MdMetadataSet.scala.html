259   /**
260     * check if Time info in date/datestamp
261     * check decide dateStamp and/or/plus CI_Date date
262     * worst case create default date
263     *
264     * TODO AK ponder if precise ZoneInfo for parsedDate needed, alternatively either UTC or NZ TimeZone
265     *
266     * @param dateStrings
267     * @return
268     */
269   def dateFromStrings(dateStrings: List[String]): LocalDate = {
270     val yearMonthMatcher = """^\d\d\d\d-\d\d$""".r
271     val yearMatcher = """^\d\d\d\d$""".r
272     val isoInstantMatcher = """.*Z$""".r
273 
274     val acceptedDateFormats = List(
275       DateTimeFormatter.ISO_LOCAL_DATE_TIME, //2011-12-03T10:15:30
276       DateTimeFormatter.ISO_LOCAL_DATE, //2011-12-03
277       DateTimeFormatter.BASIC_ISO_DATE, //20111203
278       DateTimeFormatter.ISO_OFFSET_DATE, //2011-12-03+01:00
279       DateTimeFormatter.ISO_OFFSET_DATE_TIME //2011-12-03T10:15:30+01:00
280       //TODO SR DateTimeFormatter.ISO_INSTANT cannot be parsed into LocalDate but only LocalDateTime?
281       // DateTimeFormatter.ISO_INSTANT //2011-12-03T10:15:30Z
282     )
283 
284     val datesList = dateStrings.filter(!_.trim.isEmpty).flatMap(//iterate over parameter list
285       dateString => acceptedDateFormats.map(//try all parsers per date
286         df => {
287           try {
288             val correctedDateString = dateString match {
289               case yearMonthMatcher() => dateString.concat("-01")
290               case yearMatcher() => dateString.concat("-01-01")
291               case isoInstantMatcher() => dateString.dropRight(1)
292               case _ => dateString
293             }
294             Some(LocalDate.parse(correctedDateString, df))
295           }
296           catch {
297             case e: DateTimeParseException => None
298           }
299         }
300       ).filter(_.isDefined)).map(_.get) //filter None and remove the Option
301 
302     if (datesList.isEmpty) {
303       logger.warn(f"Could not parse and of the dates (${dateStrings.mkString(", ")})")
304       LocalDate.ofEpochDay(0) //1970-01-01
305     }
306     else {
307       if (datesList.size > 1) {
308         logger.warn(f"Could parse ${datesList.size} of (${dateStrings.mkString(", ")}) only returning first success")
309       }
310       datesList.head
311     }
312   }

428   /**
429     * tries to naively prune the provided coordinates into good shape for WSG84
430     * TODO DATE Line Wraps :-( ?
431     * Rectangle rect(double minX, double maxX, double minY, double maxY);
432     * bboxFromCoords(west, east, south, north)
433     *
434     * @param west most western value / minY
435     * @param east most eastern value / maxY
436     * @return tuple of viable coordinates in WSG84
437     */
438   def pruneLongitudeValues(west: Double, east: Double): (Double, Double) = {
439     if (math.abs(west - east) > math.abs(minLon - maxLon)) {
440       (minLon, maxLon) //in case the rectangle spans more than 360 deg make it world
441     }
442     else {
443       val result = List(west, east).map({ (value: Double) =>
444         value match {
445           case n if value >= minLon && value <= maxLon => n
446           case n if math.abs(value % math.abs(minLon - maxLon)) < maxLon => {
447             val result = value % maxLon
448             logger.warn(f"changing longitude value $n to $result")
449             result
450           }
451           case _ => {
452             val result = math.signum(value) * minLon + (value % maxLon)
453             logger.warn(f"changing longitude value $value to $result")
454             result
455           }
456         }
457       })
458       (result(0), result(1))
459     }
460

464   /**
465     * Cuts off latitudes outside of minLax / maxLat and swaps if south > north
466     *
467     * @param south most southern value / minY
468     * @param north most northern value / maxY
469     * @return tuple of viable coordinates
470     */
471   def pruneLatitudeValues(south: Double, north: Double): (Double, Double) = {
472     //min/max in tuples swaps north/south if necessary,
473     (Math.max(minLat, Math.min(south, north)),
474       Math.min(maxLat, Math.max(south, north)))
475   }

477   /**
478     * tries to build a bounding box rectangle as safely as possible from provided coordinates
479     * Rectangle rect(double minX, double maxX, double minY, double maxY);
480     * bboxFromCoords(west, east, south, north)
481     *
482     * @param west  most western value / minX
483     * @param east  most eastern value / maxX
484     * @param south most southern value / minY
485     * @param north most northern value / maxY
486     * @return the resulting bounding box
487     */
488   def bboxFromCoords(west: Double, east: Double, south: Double, north: Double): Rectangle = {
489     val (prunedWest, prunedEast) = pruneLongitudeValues(west, east)
490     val (prunedSouth, prunedNorth) = pruneLatitudeValues(south, north)
491 
492     val rect = ctx.getShapeFactory().rect(prunedWest, prunedEast, prunedSouth, prunedNorth)
493     logger.debug(s"parsed rect ${rect.toString}")
494     rect
495   }

511   /**
512     * tries to extract a valid and usable bounding box rectangle from xml
513     *
514     * @param nodeSeq xml nodeseq containing an extent element
515     * @return [[Rectangle]] the bounding box
516     */
517   def bboxFromXml(nodeSeq: NodeSeq): Rectangle = {
518     val bboxXml = (nodeSeq \\ "extent" \ "EX_Extent")
519     logger.trace(f"bboxFromXml: ${bboxXml}")
520     if (bboxXml.size == 1) {
521       val westText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "westBoundLongitude" \ "Decimal").text
522       val eastText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "eastBoundLongitude" \ "Decimal").text
523       val southText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "southBoundLatitude" \ "Decimal").text
524       val northText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "northBoundLatitude" \ "Decimal").text
525       val west = westText.toDoubleWithDefault(minLon)
526       val east = eastText.toDoubleWithDefault(maxLon)
527       val south = southText.toDoubleWithDefault(minLat)
528       val north = northText.toDoubleWithDefault(maxLat)
529       // Rectangle rect(double minX, double maxX, double minY, double maxY);
530       logger.debug(s"parsed bbox (${west}, ${east}, ${south}, ${north})")
531       bboxFromCoords(west, east, south, north)
532     }
533     else {
534       logger.warn(f"${(nodeSeq \\ "fileIdentifier" \ "CharacterString").text} has no BBOX")
535       bboxFromCoords(minLon, maxLon, minLat, maxLat)
536     }
537 //    bboxFromCoords(165, 360 - 175, -53.1, -28.8)
538   }

558   def getJsGeom(gmd: MdMetadataSet): JsValue = {
559     val geometry = jsWriter.toString(gmd.bbox)

564         Json.arr(gmd.bbox.getMinX, gmd.bbox.getMaxY), //NW
565         Json.arr(gmd.bbox.getMaxX+360, gmd.bbox.getMaxY), //NE
566         Json.arr(gmd.bbox.getMaxX+360, gmd.bbox.getMinY), //NW
567         Json.arr(gmd.bbox.getMinX, gmd.bbox.getMinY)  //SW
568       ))

570     }

576   /**

582   def correctBbox(bbox: JsArray): JsArray = {
583     val west = bbox(0).asOpt[Double].get
584     val east = if (bbox(0).asOpt[Double].get > bbox(2).asOpt[Double].get) {
585                  bbox(2).asOpt[Double].get + 360

588                  bbox(2).asOpt[Double].get
589                }
