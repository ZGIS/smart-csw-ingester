234   /**
235     * check if Time info in date/datestamp
236     * check decide dateStamp and/or/plus CI_Date date
237     * worst case create default date
238     *
239     * TODO AK ponder if precise ZoneInfo for parsedDate needed, alternatively either UTC or NZ TimeZone
240     *
241     * @param dateStrings
242     * @return
243     */
244   def dateFromStrings(dateStrings: List[String]): LocalDate = {
245     val yearMonthMatcher = """^\d\d\d\d-\d\d$""".r
246     val yearMatcher = """^\d\d\d\d$""".r
247     val isoInstantMatcher = """.*Z$""".r
248 
249     val acceptedDateFormats = List(
250       DateTimeFormatter.ISO_LOCAL_DATE_TIME, //2011-12-03T10:15:30
251       DateTimeFormatter.ISO_LOCAL_DATE, //2011-12-03
252       DateTimeFormatter.BASIC_ISO_DATE, //20111203
253       DateTimeFormatter.ISO_OFFSET_DATE, //2011-12-03+01:00
254       DateTimeFormatter.ISO_OFFSET_DATE_TIME //2011-12-03T10:15:30+01:00
255       //TODO SR DateTimeFormatter.ISO_INSTANT cannot be parsed into LocalDate but only LocalDateTime?
256       // DateTimeFormatter.ISO_INSTANT //2011-12-03T10:15:30Z
257     )
258 
259     val datesList = dateStrings.filter(!_.trim.isEmpty).flatMap(//iterate over parameter list
260       dateString => acceptedDateFormats.map(//try all parsers per date
261         df => {
262           try {
263             val correctedDateString = dateString match {
264               case yearMonthMatcher() => dateString.concat("-01")
265               case yearMatcher() => dateString.concat("-01-01")
266               case isoInstantMatcher() => dateString.dropRight(1)
267               case _ => dateString
268             }
269             Some(LocalDate.parse(correctedDateString, df))
270           }
271           catch {
272             case e: DateTimeParseException => None
273           }
274         }
275       ).filter(_.isDefined)).map(_.get) //filter None and remove the Option
276 
277     if (datesList.isEmpty) {
278       logger.warn(f"Could not parse and of the dates (${dateStrings.mkString(", ")})")
279       LocalDate.ofEpochDay(0) //1970-01-01
280     }
281     else {
282       if (datesList.size > 1) {
283         logger.warn(f"Could parse ${datesList.size} of (${dateStrings.mkString(", ")}) only returning first success")
284       }
285       datesList.head
286     }
287   }

372   /**
373     * tries to naively prune the provided coordinates into good shape for WSG84
374     * TODO DATE Line Wraps :-( ?
375     * Rectangle rect(double minX, double maxX, double minY, double maxY);
376     * bboxFromCoords(west, east, south, north)
377     *
378     * @param west most western value / minY
379     * @param east most eastern value / maxY
380     * @return tuple of viable coordinates in WSG84
381     */
382   def pruneLongitudeValues(west: Double, east: Double): (Double, Double) = {
383     if (math.abs(west - east) > math.abs(minLon - maxLon)) {
384       (minLon, maxLon) //in case the rectangle spans more than 360 deg make it world
385     }
386     else {
387       val result = List(west, east).map({ (value: Double) =>
388         value match {
389           case n if value >= minLon && value <= maxLon => n
390           case n if math.abs(value % math.abs(minLon - maxLon)) < maxLon => {
391             val result = value % maxLon
392             logger.warn(f"changing longitude value $n to $result")
393             result
394           }
395           case _ => {
396             val result = math.signum(value) * minLon + (value % maxLon)
397             logger.warn(f"changing longitude value $value to $result")
398             result
399           }
400         }
401       })
402       (result(0), result(1))
403     }
404 
405 
406   }

408   /**
409     * Cuts off latitudes outside of minLax / maxLat and swaps if south > north
410     *
411     * @param south most southern value / minY
412     * @param north most northern value / maxY
413     * @return tuple of viable coordinates
414     */
415   def pruneLatitudeValues(south: Double, north: Double): (Double, Double) = {
416     //min/max in tuples swaps north/south if necessary,
417     (Math.max(minLat, Math.min(south, north)),
418       Math.min(maxLat, Math.max(south, north)))
419   }

421   /**
422     * tries to build a bounding box rectangle as safely as possible from provided coordinates
423     * Rectangle rect(double minX, double maxX, double minY, double maxY);
424     * bboxFromCoords(west, east, south, north)
425     *
426     * @param west  most western value / minX
427     * @param east  most eastern value / maxX
428     * @param south most southern value / minY
429     * @param north most northern value / maxY
430     * @return the resulting bounding box
431     */
432   def bboxFromCoords(west: Double, east: Double, south: Double, north: Double): Rectangle = {
433     val (prunedWest, prunedEast) = pruneLongitudeValues(west, east)
434     val (prunedSouth, prunedNorth) = pruneLatitudeValues(south, north)
435 
436     val rect = ctx.getShapeFactory().rect(prunedWest, prunedEast, prunedSouth, prunedNorth)
437     logger.debug("PARSED RECT: " + rect.toString)
438     rect
439   }

455   /**
456     * tries to extract a valid and usable bounding box rectangle from xml
457     *
458     * @param nodeSeq xml nodeseq containing an extent element
459     * @return [[Rectangle]] the bounding box
460     */
461   def bboxFromXml(nodeSeq: NodeSeq): Rectangle = {
462     val bboxXml = (nodeSeq \\ "extent" \ "EX_Extent")
463     logger.trace(f"bboxFromXml: ${bboxXml}")
464     if (bboxXml.size == 1) {
465       val westText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "westBoundLongitude" \ "Decimal").text
466       val eastText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "eastBoundLongitude" \ "Decimal").text
467       val southText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "southBoundLatitude" \ "Decimal").text
468       val northText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "northBoundLatitude" \ "Decimal").text
469       val west = westText.toDoubleWithDefault(minLon)
470       val east = eastText.toDoubleWithDefault(maxLon)
471       val south = southText.toDoubleWithDefault(minLat)
472       val north = northText.toDoubleWithDefault(maxLat)
473       // Rectangle rect(double minX, double maxX, double minY, double maxY);
474       bboxFromCoords(west, east, south, north)
475     }
476     else {
477       logger.warn(f"${(nodeSeq \\ "fileIdentifier" \ "CharacterString").text} has no BBOX")
478       bboxFromCoords(minLon, maxLon, minLat, maxLat)
479     }
480 //    bboxFromCoords(165, 360 - 175, -53.1, -28.8)
481   }
