36 /**
37   * Holds a subset of MDMetadata
38   *
39   * @param fileIdentifier the unique file identifier of the record
40   * @param dateStamp a datestamp of the record
41   * @param title title of the record
42   * @param abstrakt abstract of the record
43   * @param keywords list of keywords for the record
44   * @param topicCategory list of ISO/ANZLIC topic categories of the record
45   * @param contactName contact name for the record or dataset
46   * @param contactOrg cantact organisation for the record or dataset
47   * @param contactEmail contact email for the record or dataset
48   * @param license licenses for the record or dataset
49   * @param bbox bbox of the record
50   * @param origin origin i.e. source catalogue where the record was loaded from
51   */
52 case class MdMetadataSet(fileIdentifier: String,
53                          dateStamp: LocalDate,
54                          title: String,
55                          abstrakt: String,
56                          keywords: List[String],
57                          topicCategory: List[String],
58                          contactName: String,
59                          contactOrg: String,
60                          contactEmail: String,
61                          license: String,
62                          bbox: Rectangle,
63                          origin: String) extends ClassnameLogger {
64   require(!fileIdentifier.trim.isEmpty, "FileIdentifier was empty")
65

67     f"""MdMetadataSet(
68         |$fileIdentifier,
69         |${dateStampAsIsoString},

73         |topicCategory(${topicCategory.mkString(", ")}),
74         |${contactName},
75         |${contactOrg},
76         |${contactEmail},
77         |${license},
78         |${bboxAsWkt},
79         |${origin})

103   /**
104     * returns the object as LuceneDocument
105     *
106     * @return LuceneDocument for index
107     */
108   def asLuceneDocument(): Document = {
109     val ctx = SpatialContext.GEO
110     val doc = new Document()
111 
112     val longDate = dateStamp.toEpochDay
113     val dateField = new LongPoint("dateStampCompare", 1)
114     dateField.setLongValue(longDate)
115     doc.add(dateField)
116 
117     val bboxStrategy: BBoxStrategy = BBoxStrategy.newInstance(ctx, "bboxStrategy")
118     val bboxFields = bboxStrategy.createIndexableFields(bbox)
119     bboxFields.foreach(field => doc.add(field))
120 
121     doc.add(new Field("fileIdentifier", fileIdentifier, TextField.TYPE_STORED))
122     doc.add(new Field("title", title, TextField.TYPE_STORED))
123     doc.add(new Field("abstrakt", abstrakt, TextField.TYPE_STORED))
124     // Range Query for Date as Long value, this field is to recreate the date object
125     doc.add(new Field("dateStampText", dateStampAsIsoString, TextField.TYPE_STORED))
126     doc.add(new Field("keywords", keywords.mkString(" "), TextField.TYPE_STORED))
127     doc.add(new Field("topicCategory", topicCategory.mkString(" "), TextField.TYPE_STORED))
128     doc.add(new Field("contactName", contactName, TextField.TYPE_STORED))
129     doc.add(new Field("contactOrg", contactOrg, TextField.TYPE_STORED))
130     doc.add(new Field("contactEmail", contactEmail, TextField.TYPE_STORED))
131     doc.add(new Field("license", license, TextField.TYPE_STORED))
132     // Bbox Query on spatial index, this textfield is to recreate the geometry
133     doc.add(new Field("bboxText", bboxAsWkt, TextField.TYPE_STORED))
134     doc.add(new Field("origin", origin, TextField.TYPE_STORED))
135 
136     //FIXME decide if use catch_all field or how to build a query that queries all fields
137     doc.add(new Field("catch_all", fileIdentifier, TextField.TYPE_STORED))
138     doc.add(new Field("catch_all", title, TextField.TYPE_STORED))
139     doc.add(new Field("catch_all", abstrakt, TextField.TYPE_STORED))
140     // Range Query for Date as Long value, this field is to recreate the date object
141     doc.add(new Field("catch_all", dateStampAsIsoString, TextField.TYPE_STORED))
142     doc.add(new Field("catch_all", keywords.mkString(" "), TextField.TYPE_STORED))
143     doc.add(new Field("catch_all", topicCategory.mkString(" "), TextField.TYPE_STORED))
144     doc.add(new Field("catch_all", contactName, TextField.TYPE_STORED))
145     doc.add(new Field("catch_all", contactOrg, TextField.TYPE_STORED))
146     doc.add(new Field("catch_all", contactEmail, TextField.TYPE_STORED))
147     doc.add(new Field("catch_all", license, TextField.TYPE_STORED))
148     // Bbox Query on spatial index, this textfield is to recreate the geometry
149     doc.add(new Field("catch_all", bboxAsWkt, TextField.TYPE_STORED))
150     doc.add(new Field("catch_all", origin, TextField.TYPE_STORED))
151 
152     doc
153   }

236   /**
237     * check if Time info in date/datestamp
238     * check decide dateStamp and/or/plus CI_Date date
239     * worst case create default date
240     *
241     * TODO AK ponder if precise ZoneInfo for parsedDate needed, alternatively either UTC or NZ TimeZone
242     *
243     * @param dateStrings
244     * @return
245     */
246   def dateFromStrings(dateStrings: List[String]): LocalDate = {
247     val yearMonthMatcher = """^\d\d\d\d-\d\d$""".r
248     val yearMatcher = """^\d\d\d\d$""".r
249     val isoInstantMatcher = """.*Z$""".r
250 
251     val acceptedDateFormats = List(
252       DateTimeFormatter.ISO_LOCAL_DATE_TIME, //2011-12-03T10:15:30
253       DateTimeFormatter.ISO_LOCAL_DATE, //2011-12-03
254       DateTimeFormatter.BASIC_ISO_DATE, //20111203
255       DateTimeFormatter.ISO_OFFSET_DATE, //2011-12-03+01:00
256       DateTimeFormatter.ISO_OFFSET_DATE_TIME //2011-12-03T10:15:30+01:00
257       // DateTimeFormatter.ISO_INSTANT //2011-12-03T10:15:30Z - this cannot be parsed into LocalDate but only LocalDateTime?
258     )
259 
260     val datesList = dateStrings.filter(!_.trim.isEmpty).flatMap(//iterate over parameter list
261       dateString => acceptedDateFormats.map(//try all parsers per date
262         df => {
263           try {
264             val correctedDateString = dateString match {
265               case yearMonthMatcher() => dateString.concat("-01")
266               case yearMatcher() => dateString.concat("-01-01")
267               case isoInstantMatcher() => dateString.dropRight(1)
268               case _ => dateString
269             }
270             Some(LocalDate.parse(correctedDateString, df))
271           }
272           catch {
273             case e: DateTimeParseException => None
274           }
275         }
276       ).filter(_.isDefined)).map(_.get) //filter None and remove the Option
277 
278     if (datesList.isEmpty) {
279       logger.warn(f"Could not parse and of the dates (${dateStrings.mkString(", ")})")
280       LocalDate.ofEpochDay(0) //1970-01-01
281     }
282     else {
283       if (datesList.size > 1) {
284         logger.warn(f"Could parse ${datesList.size} of (${dateStrings.mkString(", ")}) only returning first success")
285       }
286       datesList.head
287     }
288   }

373   /**
374     * tries to naively prune the provided coordinates into good shape for WSG84
375     * TODO DATE Line Wraps :-( ?
376     *
377     * @param east most eastern value / minY
378     * @param west most western value / maxY
379     * @return tuple of viable coordinates in WSG84
380     */
381   def pruneLongitudeValues(east: Double, west: Double) : (Double, Double) = {
382     val x1 = east match {
383       case n: Double if (n < minLon) => {
384         logger.warn(f"cutting east value: $east to $minLon")
385         minLon
386       }
387       case n: Double if (n > maxLon) => {
388         logger.warn(f"cutting east value: $east to $maxLon")
389         maxLon
390       }
391       case _ => {
392         // if all good then return value
393         east
394       }
395     }
396     val x2 = west match {
397       case n: Double if (n < minLon) => {
398         logger.warn(f"cutting west value: $west to $minLon")
399         minLon
400       }
401       case n: Double if (n > maxLon) => {
402         logger.warn(f"cutting west value: $west to $maxLon")
403         maxLon
404       }
405       case _ => {
406         // if all good then return value
407         west
408       }
409     }
410     (x1, x2)
411   }

453       }
454       case _ => {
455         // if all good then return value

457       }
458     }

462   /**
463     * tries to build a bounding box rectangle as safely as possible from provided coordinates
464     *
465     * @param east most eastern value / minX
466     * @param west most western value / maxX
467     * @param south most southern value / minY
468     * @param north most northern value / maxY
469     * @return the resulting bounding box
470     */
471   def bboxFromCoords(east: Double, west: Double, south: Double, north: Double): Rectangle = {
472     val (prunedEast, prunedWest) = pruneLongitudeValues(east, west)
473     val (prunedSouth, prunedNorth) = pruneLatitudeValues(south, north)
474     ctx.getShapeFactory().rect(prunedEast, prunedWest, prunedSouth, prunedNorth)
475   }

523     * tries to extract a valid and usable bounding box rectangle from xml
524     *
525     * @param nodeSeq xml nodeseq containing an extent element
526     * @return [[Rectangle]] the bounding box
527     */
528   def bboxFromXml(nodeSeq: NodeSeq): Rectangle = {
529     val bboxXml = (nodeSeq \\ "extent" \ "EX_Extent")
530     logger.trace(f"bboxFromXml: ${bboxXml}")
531     if (bboxXml.size == 1) {
532       val eastText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "eastBoundLongitude" \ "Decimal").text
533       val westText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "westBoundLongitude" \ "Decimal").text
534       val southText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "southBoundLatitude" \ "Decimal").text
535       val northText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "northBoundLatitude" \ "Decimal").text
536       val east = extractLatLonNumber(eastText, minLon)
537       val west = extractLatLonNumber(westText, maxLon)
538       val south = extractLatLonNumber(southText, minLat)
539       val north = extractLatLonNumber(northText, maxLat)
540       // Rectangle rect(double minX, double maxX, double minY, double maxY);
541       bboxFromCoords(east, west, south, north)
542     }
543     else {
544       logger.warn(f"${(nodeSeq \\ "fileIdentifier" \ "CharacterString").text} has no BBOX")
545       bboxFromCoords(minLon, maxLon, minLat, maxLat)
546     }
547   }

561   /**
562     * provides the GeoJSON Polygon geometry from the feature's Rectangle bbox
563     * @param gmd MdMetadataSet
564     * @return JsValue with GeoJSON Polygon geometry
565     */
566   def getJsGeom(gmd: MdMetadataSet) : JsValue = {
567     val geometry = jsWriter.toString(gmd.bbox)
568     Json.parse(geometry)

571   /**
572     * Converts [[MdMetadataSet]] object into [[JsObject]] as GeoJSON
573     */
574   def writes(gmd: MdMetadataSet) : JsObject = {
575 
576     val properties = Json.obj(
577       "fileIdentifier" -> gmd.fileIdentifier,
578       "dateStamp" -> gmd.dateStampAsIsoString,
579       "title" -> gmd.title,
580       "abstrakt" -> gmd.abstrakt,
581       "keywords" -> gmd.keywords,
582       "topicCategory" -> gmd.topicCategory,
583       "contactName" -> gmd.contactName,
584       "contactOrg" -> gmd.contactOrg,
585       "contactEmail" -> gmd.contactEmail,
586       "license" -> gmd.license,
587       // extent is an array [10, 10, 40, 40] minX, maxX, maxY, minY
588       "bbox" -> Json.arr(
589         JsNumber(gmd.bbox.getMinX()),
590         JsNumber(gmd.bbox.getMaxX()),
591         JsNumber(gmd.bbox.getMaxY()),
592         JsNumber(gmd.bbox.getMinY())
593       ),
594       "origin" -> gmd.origin
595     )
596 
597     Json.obj(
598       "type" -> "Feature",
599       "geometry" -> getJsGeom(gmd),
600       "properties" -> properties
601     )
602   }
603 }
