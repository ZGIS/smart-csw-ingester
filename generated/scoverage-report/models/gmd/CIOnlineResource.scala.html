49 /**
50   * type of an online resource
51   */
52 object ResourceType {
53 
54   case object WEBSITE extends ResourceType {
55     val name = "website"
56   }
57 
58   case object DOWNLOAD extends ResourceType {
59     val name = "download"
60   }
61 
62   case object MAP extends ResourceType {
63     val name = "map"
64   }
65 
66   case object DATA extends ResourceType {
67     val name = "data"
68   }
69 
70   case object METADATA extends ResourceType {
71     val name = "metadata"
72   }
73 
74   def fromString(str: String): ResourceType = str match {
75     case "website" => WEBSITE
76     case "download" => DOWNLOAD
77     case "map" => MAP
78     case "data" => DATA
79     case "metadata" => METADATA
80     case str => UnknownResourceType(str)
81   }
82

84     * JSON reader
85     *
86     * @see For more explanation on this little magic: [[https://groups.google.com/forum/?fromgroups=#!starred/play-framework/hGrveOkbJ6U]]
87     */
88   implicit val reads = Reads[ResourceType](r => JsSuccess(ResourceType.fromString(r.as[String])))
89   implicit val writes = Writes[ResourceType](r => JsString(r.name))
90 }

92 /**
93   * Representation of an CI_OnlineResource

98 case class CIOnlineResource(linkage: URL,
99                             name: Option[String],
100                             description: Option[String],
101                             protocol: Option[String],
102                             resourceType: ResourceType) extends ClassnameLogger {

134       (JsPath \ "description").readNullable[String] and
135       (JsPath \ "protocol").readNullable[String] and
136       (JsPath \ "resourceType").read[ResourceType]
137     ) (CIOnlineResource.apply _)
138 
139   /**
140     * parse CIOnlineResource from XML
141     *
142     * @param nodeSeq that represents a gmd:CI_OnlineResource
143     * @param origin  name of the cataloue origin (used to determine the link type in some cases)
144     * @see [[https://geo-ide.noaa.gov/wiki/index.php?title=CI_OnlineResource NOAA EDM Wiki - CI_OnlineResource]]
145     * @return
146     */
147   def fromXml(nodeSeq: NodeSeq, origin: String): CIOnlineResource = {
148     val linkage = new URL((nodeSeq \ "linkage" \ "URL").text.trim)
149     logger.debug(s"Linkage: ${linkage}")
150 
151     import utils.StringUtils._
152     val name = (nodeSeq \ "name" \ "CharacterString").text.toOption()
153     logger.debug(s"Name: ${name}")
154     val description = (nodeSeq \ "description" \ "CharacterString").text.toOption()
155     logger.debug(s"Description: ${description}")
156 
157     val protocol = linkage.toString match {
158       // the GNS CSW has a little bit of a stupid way of assigning protocol to the linkages...
159       case r"https?:\/\/data.gns.cri.nz\/rgmad\/(?:thumbs|images|layers)\/.*" => "WWW:LINK-1.0-http--download".toOption()
160       case _ => (nodeSeq \ "protocol" \ "CharacterString").text.toOption()
161     }
162     logger.debug(s"Protocol ${protocol}")
163 
164     import utils.StringUtils.Regex
165     val resourceType = protocol match {
166       case Some("WWW:LINK-1.0-http--metadata-URL") => ResourceType.METADATA
167       case Some("WWW:LINK-1.0-http--link") => linkage.toString match {
168         case r"https?:\/\/data.gns.cri.nz\/rgmad\/(?:thumbs|images|layers)\/.*" => ResourceType.DOWNLOAD
169         case _ => ResourceType.WEBSITE
170       }
171       case Some("WWW:LINK-1.0-http--downloaddata") => ResourceType.DOWNLOAD
172       case Some("OGC:WCS-1.1.0-http-get-capabilities") => ResourceType.METADATA
173       case _ => linkage.toString match {
174         //from here we start some magic by looking at URLs
175         case r"https?:\/\/geoportal\.doc\.govt\.nz\/(?i:ArcGIS)\/.*\/MapServer" => ResourceType.METADATA
176         case r"https?:\/\/data.linz.govt.nz\/layer\/.*" => ResourceType.MAP
177         case _ => ResourceType.WEBSITE
178       }
179     }
180     logger.debug(s"ResourceType: ${resourceType}")
181 
182     CIOnlineResource(linkage, name, description, protocol, resourceType)
183   }
