31   /**
32     * Converts to a Option[OwcLink].
33     * OWC:Context specReference attribute: atom rel="profile" - geojson links.profiles[] array
34     * OWC:Context contextMetadata attribute: atom rel="via" - geojson links.via[] array
35     *
36     * OWC:Resource contentDescription attribute: atom rel="alternate" - geojson links.alternates[] array
37     * OWC:Resource preview attribute: atom rel="icon" - geojson links.previews[] array
38     * OWC:Resource contentByRef attribute: atom rel="enclosure" - geojson links.data[] array
39     * OWC:Resource resourceMetadata attribute: atom rel="via" - geojson links.via[] array
40     *
41     * links for data and previews (aka rels enclosure and icon should have length attributes set)
42     *
43     * @return Some[OwcLink] or None if CIOnlineResource could not be converted sensibly
44     */
45   def asOwcLink(cIOnlineResource: CIOnlineResource): Option[OwcLink] = {
46     import utils.StringUtils._
47 
48     cIOnlineResource.protocol match {
49       case Some("WWW:LINK-1.0-http--metadata-URL") =>
50         Some(OwcLink(rel = "via",
51           mimeType = Some("application/xml"),
52           href = cIOnlineResource.linkage,
53           title = cIOnlineResource.name))
54       case Some("WWW:LINK-1.0-http--link") =>
55         Some(OwcLink(rel = "alternate",
56           mimeType = Some("text/html"),
57           href = cIOnlineResource.linkage,
58           title = cIOnlineResource.name))
59       case Some(r"WWW:LINK-1.0-http--download(?:data)?") => {
60         val mimeType = cIOnlineResource.linkage.getFile.toLowerCase match {
61           case r".*\.jpe?g" => "image/jpeg"
62           case r".*\.png" => "image/png"
63           case r".*\.pdf" => "application/pdf"
64           case r".*\.xlsx?" => "application/excel"
65           case r".*\.txt" => "text/plain"
66           case r".*\.csv" => "text/csv"
67           case _ => "application/octet-stream"
68         }
69         Some(OwcLink(rel = "enclosure",
70           mimeType = Some(mimeType),
71           href = cIOnlineResource.linkage,
72           title = cIOnlineResource.name))
73       }
74       case _ => cIOnlineResource.linkage match {
75         case r"https?:\/\/data.linz.govt.nz\/layer\/.*" =>
76           Some(OwcLink(rel = "alternates",
77             mimeType = Some("text/html"),
78             href = cIOnlineResource.linkage,
79             title = cIOnlineResource.name))
80         case r"https?:\/\/lris.scinfo.org.nz\/layer\/.*" =>
81           Some(OwcLink(rel = "alternates",
82             mimeType = Some("text/html"),
83             href = cIOnlineResource.linkage,
84             title = cIOnlineResource.name))
85         case r"https?:\/\/geoportal\.doc\.govt\.nz\/(?i:ArcGIS)\/.*\/MapServer" =>
86           Some(OwcLink(rel = "alternates",
87             mimeType = Some("text/html"),
88             href = cIOnlineResource.linkage,
89             title = cIOnlineResource.name))
90         case _ => cIOnlineResource.resourceType match {
91           case ResourceType.WEBSITE => Some(OwcLink(rel = "alternate",
92             mimeType = Some("text/html"),
93             href = cIOnlineResource.linkage,
94             title = cIOnlineResource.name))
95           case _ => None
96         }
97       }
98     }
99   }

101   /**
102     * Converts OnlineResource to a List of OwcOffering. List might be empty, if the CIOnlineResource cannot be
103     * converted to offerings in a sensible way.
104     *
105     * @return
106     */
107   def asOwcOfferings(cIOnlineResource: CIOnlineResource): List[OwcOffering] = {
108     import utils.StringUtils._

127                 mimeType = Some("application/xml"),
128                 requestUrl = new URL(s"${cIOnlineResource.linkage}/WFSServer?request=GetCapabilities&service=WFS"))

131         )
132       case _ => Nil
133     }

136   /**
137     * Converts MdMetadataSet to [[OwcResource]]
138     *
139     * @return
140     */
141   def asOwcResource(mdMetadataSet: MdMetadataSet, urlBaseForIds: String): OwcResource = {
142     import utils.StringUtils._
143 
144     val offerings =
145     // convert linkages to offerings (ist might be empty if there are no linkages that can be converted to offerings
146       mdMetadataSet.linkage.flatMap(ci => asOwcOfferings(ci)) :::
147         //Offering to get the original metadata document. Every document in the index should have that
148         List(
149           OwcOffering(code = CSW.code,
150             operations = List(
151               OwcOperation(
152                 code = "GetCapabilities",
153                 method = "GET",
154                 mimeType = Some("application/xml"),
155                 requestUrl = new URL(s"${mdMetadataSet.originUrl}?request=GetCapabilities&service=CSW")),
156               OwcOperation(
157                 code = "GetRecordById",
158                 method = "GET",
159                 mimeType = Some("application/xml"),
160                 requestUrl = new URL(s"${mdMetadataSet.originUrl}?request=GetRecordById&version=2.0.2&service=CSW&elementSetName=full" +
161                   "&outputSchema=http%3A%2F%2Fwww.isotc211.org%2F2005%2Fgmd" +
162                   s"&Id=${URLEncoder.encode(mdMetadataSet.fileIdentifier, "UTF-8")}")))
163           )
164         )
165 
166     // ATTENTION: While id in OwcResource is of Type CharacterString, it still expects an IRI/URI type String, that'S why URL
167     val baseUrl = new URL(urlBaseForIds)
168     OwcResource(
169       id = new URL(s"${baseUrl.getProtocol}://${baseUrl.getHost}/context/resource/${URLEncoder.encode(mdMetadataSet.fileIdentifier, "UTF-8")}"),
170       geospatialExtent = Some(mdMetadataSet.bbox),
171       title = mdMetadataSet.title,
172       subtitle = mdMetadataSet.abstrakt.toOption(),
173       updateDate = OffsetDateTime.of(mdMetadataSet.dateStamp.atStartOfDay(), ZoneId.systemDefault().getRules.getOffset(mdMetadataSet.dateStamp.atStartOfDay())),
174       author = List(OwcAuthor(name = mdMetadataSet.contactName.toOption(), email = MdMetadataSet.parseEmailStringtoEmailAddress(mdMetadataSet.contactEmail), uri = None)),
175       publisher = None,
176       rights = mdMetadataSet.license.toOption(),
177       temporalExtent = None,
178 
179       // links.alternates[] and rel=alternate
180       contentDescription = mdMetadataSet.linkage.map(asOwcLink(_)).filter(_.isDefined)
181         .map(_.get).filter(link => link.rel.equalsIgnoreCase("alternate")),
182 
183       // aka links.previews[] and rel=icon (atom)
184       preview = mdMetadataSet.linkage.map(asOwcLink(_)).filter(_.isDefined)
185         .map(_.get).filter(link => link.rel.equalsIgnoreCase("icon")),
186 
187       // aka links.data[] and rel=enclosure (atom)
188       contentByRef = mdMetadataSet.linkage.map(asOwcLink(_)).filter(_.isDefined)
189         .map(_.get).filter(link => link.rel.equalsIgnoreCase("enclosure")),
190 
191       // aka links.via[] & rel=via
192       resourceMetadata = mdMetadataSet.linkage.map(asOwcLink(_)).filter(_.isDefined)
193         .map(_.get).filter(link => link.rel.equalsIgnoreCase("via")),
194       offering = offerings,
195       minScaleDenominator = None,
196       maxScaleDenominator = None,
197       keyword = List(),
198       folder = None)
199   }

201   /**
202     * Convert List[MdMetadataSet] to OwcContext
203     *
204     * @param metadataList  List[MdMetadataSet] containing the entries
205     * @param idAsUrlString String containing the ID for the OwcContext document
206     */
207   def toOwcContext(metadataList: List[MdMetadataSet], idAsUrlString: String): OwcContext = {
208     val bbox = Some(GeoJSONFeatureCollectionWriter.getBoundingBoxAsRect(metadataList))
209 
210     val owcCreatorApplication = OwcCreatorApplication(
211       title = Some(utils.BuildInfo.name),
212       uri = Some(new URL(idAsUrlString)),
213       version = Some(s"${utils.BuildInfo.version}-${utils.BuildInfo.buildNumber}")
214     )
215 
216     implicit val localDateOrdering: Ordering[LocalDate] = Ordering.by(_.toEpochDay)
217     val dateRange = metadataList.map(md => md.dateStamp).sorted
218     val optDateInterval: Option[List[OffsetDateTime]] = if (dateRange.nonEmpty) {
219       if (dateRange.tail.isEmpty) {
220         Some(List(OffsetDateTime.of(dateRange.head, LocalTime.of(12, 0), ZoneOffset.UTC)))
221       } else {
222         Some(List(OffsetDateTime.of(dateRange.last, LocalTime.of(12, 0), ZoneOffset.UTC)))
223       }
224     } else {
225       None
226     }
227 
228     val owcResources = metadataList.map(md => asOwcResource(md, idAsUrlString))
229     OwcContext(
230       id = new URL(idAsUrlString),
231       areaOfInterest = bbox,
232 
233       // aka links.profiles[] and rel=profile
234       specReference = List(OwcProfile.CORE.newOf),
235 
236       // e.g. links.via[] and rel=via
237       contextMetadata = List(),
238       language = "en",
239       title = "Search result list CSW ingester",
240       subtitle = Some("Originating query: "),
241       updateDate = OffsetDateTime.now(),
242       author = List(),
243       publisher = None,
244       creatorApplication = Some(owcCreatorApplication),
245       creatorDisplay = None,
246       rights = None,
247       timeIntervalOfInterest = optDateInterval,
248       keyword = List(),
249       resource = owcResources)
250   }
