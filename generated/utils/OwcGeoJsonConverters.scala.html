35   /**
36     * Converts to a Option[OwcLink].
37     * OWC:Context specReference attribute: atom rel="profile" - geojson links.profiles[] array
38     * OWC:Context contextMetadata attribute: atom rel="via" - geojson links.via[] array
39     *
40     * OWC:Resource contentDescription attribute: atom rel="alternate" - geojson links.alternates[] array
41     * OWC:Resource preview attribute: atom rel="icon" - geojson links.previews[] array
42     * OWC:Resource contentByRef attribute: atom rel="enclosure" - geojson links.data[] array
43     * OWC:Resource resourceMetadata attribute: atom rel="via" - geojson links.via[] array
44     *
45     * links for data and previews (aka rels enclosure and icon should have length attributes set)
46     *
47     * @return Some[OwcLink] or None if CIOnlineResource could not be converted sensibly
48     */
49   def asOwcLink(cIOnlineResource: CIOnlineResource): Option[OwcLink] = {
50     import utils.StringUtils._
51 
52     val filename = Try(cIOnlineResource.linkage.getFile.toLowerCase).toOption
53     val mimetypeOption = filename.flatMap(fn => MimeTypes.forFileName(fn))
54 
55     cIOnlineResource.protocol match {
56       case Some("WWW:LINK-1.0-http--metadata-URL") =>
57         Some(OwcLink(rel = "via",
58           mimeType = mimetypeOption.orElse(Some("application/xml")),
59           href = cIOnlineResource.linkage,
60           title = cIOnlineResource.name))
61       case Some("WWW:LINK-1.0-http--link") =>
62         Some(OwcLink(rel = "alternate",
63           mimeType = mimetypeOption.orElse(Some("text/html")),
64           href = cIOnlineResource.linkage,
65           title = cIOnlineResource.name))
66       case Some(r"WWW:LINK-1.0-http--download(?:data)?") =>
67         Some(OwcLink(rel = "enclosure",
68           mimeType = mimetypeOption.orElse(Some("application/octet-stream")),
69           href = cIOnlineResource.linkage,
70           title = cIOnlineResource.name))
71       case Some("WWW:LINK-1.0-http--image-thumbnail") =>
72         Some(OwcLink(rel = "icon",
73           mimeType = mimetypeOption.orElse(Some("image/png")),
74           href = cIOnlineResource.linkage,
75           title = cIOnlineResource.name))
76 
77       case _ => cIOnlineResource.linkage.toString match {
78         case r"https?:\/\/data.linz.govt.nz\/layer\/.*" =>
79           Some(OwcLink(rel = "via",
80             mimeType = Some("text/html"),
81             href = cIOnlineResource.linkage,

207               OwcOperation(

212                   "&outputSchema=http%3A%2F%2Fwww.isotc211.org%2F2005%2Fgmd" +
213                   s"&Id=${URLEncoder.encode(mdMetadataSet.fileIdentifier, "UTF-8")}")))
214           )
215         )
216 
217     // ATTENTION: While id in OwcResource is of Type CharacterString, it still expects an IRI/URI type String, that'S why URL
218     val baseUrl = new URL(urlBaseForIds)
219     OwcResource(
220       id = new URL(s"${baseUrl.getProtocol}://${baseUrl.getHost}/context/resource/${URLEncoder.encode(mdMetadataSet.fileIdentifier, "UTF-8")}"),
221       geospatialExtent = Some(mdMetadataSet.bbox),
222       title = mdMetadataSet.title,
223       subtitle = mdMetadataSet.abstrakt.toOption(),
224       updateDate = OffsetDateTime.of(mdMetadataSet.dateStamp.atStartOfDay(), ZoneId.systemDefault().getRules.getOffset(mdMetadataSet.dateStamp.atStartOfDay())),
225       author = List(OwcAuthor(name = mdMetadataSet.contactName.toOption(), email = MdMetadataSet.parseEmailStringtoEmailAddress(mdMetadataSet.contactEmail), uri = None)),
226       publisher = mdMetadataSet.contactOrg.toOption(),
227       rights = mdMetadataSet.license.toOption(),
228       temporalExtent = None,
229 
230       // links.alternates[] and rel=alternate
231       contentDescription = mdMetadataSet.linkage.map(asOwcLink(_)).filter(_.isDefined)
232         .map(_.get).filter(link => link.rel.equalsIgnoreCase("alternate")),
233 
234       // aka links.previews[] and rel=icon (atom)
235       preview = mdMetadataSet.linkage.map(asOwcLink(_)).filter(_.isDefined)
236         .map(_.get).filter(link => link.rel.equalsIgnoreCase("icon")),
237 
238       // aka links.data[] and rel=enclosure (atom)
239       contentByRef = mdMetadataSet.linkage.map(asOwcLink(_)).filter(_.isDefined)
240         .map(_.get).filter(link => link.rel.equalsIgnoreCase("enclosure")),
241 
242       // aka links.via[] & rel=via
243       resourceMetadata = mdMetadataSet.linkage.map(asOwcLink(_)).filter(_.isDefined)
244         .map(_.get).filter(link => link.rel.equalsIgnoreCase("via")),
245       offering = offerings,
246       minScaleDenominator = None,
247       maxScaleDenominator = None,
248       keyword = asOwcCategory(mdMetadataSet),
249       folder = None)
250   }

253     * Convert List[MdMetadataSet] to OwcContext
254     *
255     * @param metadataList  List[MdMetadataSet] containing the entries
256     * @param idAsUrlString String containing the ID for the OwcContext document
257     */
258   def toOwcContext(metadataList: List[MdMetadataSet], idAsUrlString: String): OwcContext = {
259     val bbox = Some(GeoJSONFeatureCollectionWriter.getBoundingBoxAsRect(metadataList))
260 
261     val owcCreatorApplication = OwcCreatorApplication(
262       title = Some(utils.BuildInfo.name),
263       uri = Some(new URL(idAsUrlString)),
264       version = Some(s"${utils.BuildInfo.version}-${utils.BuildInfo.buildNumber}")
265     )
266 
267     implicit val localDateOrdering: Ordering[LocalDate] = Ordering.by(_.toEpochDay)
268     val dateRange = metadataList.map(md => md.dateStamp).sorted
269     val optDateInterval: Option[List[OffsetDateTime]] = if (dateRange.nonEmpty) {
270       if (dateRange.tail.isEmpty) {
271         Some(List(OffsetDateTime.of(dateRange.head, LocalTime.of(12, 0), ZoneOffset.UTC)))
272       } else {
273         Some(List(OffsetDateTime.of(dateRange.last, LocalTime.of(12, 0), ZoneOffset.UTC)))
274       }
275     } else {
276       None
277     }
278 
279     val owcResources = metadataList.map(md => asOwcResource(md, idAsUrlString))
280     OwcContext(
281       id = new URL(idAsUrlString),
282       areaOfInterest = bbox,
283 
284       // aka links.profiles[] and rel=profile
285       specReference = List(OwcProfile.CORE.newOf),
286 
287       // e.g. links.via[] and rel=via
288       contextMetadata = List(),
289       language = "en",
290       title = "Search result list CSW ingester",
291       subtitle = Some("Originating from query"),
292       updateDate = OffsetDateTime.now(),
293       author = List(),
294       publisher = None,
295       creatorApplication = Some(owcCreatorApplication),
296       creatorDisplay = None,
297       rights = None,
298       timeIntervalOfInterest = optDateInterval,
299       keyword = List(),
300       resource = owcResources)
301   }

303   /**
304     * tries to guess SERVICE based OGC offerings
305     *
306     * @param protocolString
307     * @param url
308     * @return
309     */
310   def generateOwcOfferingFromServiceProto(protocolString: String, url: URL): List[OwcOffering] = {
311 
312     // OGC:WCS-1.0.0-http-get-capabilities
313     // OGC:WCS-1.0.0-http-get-coverage
314     // OGC:WMS-1.3.0-http-get-map
315     // OGC:WFS-2.0.0-http-get-feature
316     // OGC:SOS-1.0.0-http-get-observation
317     val protoInfo = protocolString.split(":")
318     if (protoInfo.length > 1) {
319       if (protoInfo(0).equalsIgnoreCase("OGC")) {
320         val operationInfo = protoInfo(1).split("-")
321         val offeringCode = operationInfo(0)
322         val offeringType = Try(applyOfferingType(offeringCode)).toOption
323         if (operationInfo.length > 1) {
324           val version = operationInfo(1)
325           val http = operationInfo(2)
326           val part1 = operationInfo(3)
327           val part2 = operationInfo(4)
328           val operationName = part1.toLowerCase.capitalize + part2.toLowerCase.capitalize
329           offeringType.map {
330             ot =>
331               OwcOffering(code = ot.code,
332                 operations = List(
333                   OwcOperation(
334                     code = operationName,
335                     method = "GET",
336                     mimeType = Some("application/xml"),
337                     requestUrl = new URL(s"${url.toString}?service=${offeringCode.toUpperCase}&request=$operationName&version=$version"))
338                 )
339               )
340           }.toList
341         } else {
342 
343           offeringType.map {
344             ot =>
345               OwcOffering(code = ot.code,
346                 operations = List(
347                   OwcOperation(
348                     code = "GetCapabilities",
349                     method = "GET",
350                     mimeType = Some("application/xml"),
351                     requestUrl = new URL(s"${url.toString}?request=GetCapabilities&service=${offeringCode.toUpperCase}"))
352                 )
353               )
354           }.toList
355         }
356       } else {
357         List()
358       }
359     } else {
360       List()
361     }
362   }

370   def applyOfferingType(v: String): OwcOfferingType = {
371     v.toUpperCase match {

373       case "WFS" => WFS
374       case "WCS" => WCS
375       case "SOS" => SOS

377       case "CSW" => CSW
378       case _ =>
379         logger.warn(s"Value $v is not a supported OwcOfferingType value")
380         throw new IllegalArgumentException(s"Value $v is not a supported OwcOfferingType value")
381     }

384   def generateOwcOfferingForArcServer(url: URL): List[OwcOffering] = {
385 
386     // GeoPortals ArcGIS server offers WMS/WFS for all layers I have seen. So we generate offerings for that.
387     // from http://geoportal.doc.govt.nz/ArcGIS/rest/services/GeoportalServices/DOC_BDIPEST_HimalayanThar_2007/MapServer
388     // to http://geoportal.doc.govt.nz/arcgis/services/GeoportalServices/DOC_Operations_Regions/MapServer/WFSServer?request=GetCapabilities&service=WFS
389 
390     val urlParts = url.getPath.split("/")
391     logger.debug(urlParts.length.toString)
392 
393     if (urlParts.length == 7) {
394       logger.debug(urlParts.mkString("::"))
395       if (urlParts(0).isEmpty &&
396         urlParts(1).equalsIgnoreCase("ArcGIS") &&
397         urlParts(2).equalsIgnoreCase("rest") &&
398         urlParts(3).equalsIgnoreCase("services")
399       ) {
400         logger.debug(s"${url.getProtocol}://${url.getHost}/arcgis/services/${urlParts(4)}/${urlParts(5)}/${urlParts(6)}")
401         val owcBaseUrl = new URL(s"${url.getProtocol}://${url.getHost}/arcgis/services/${urlParts(4)}/${urlParts(5)}/${urlParts(6)}")
402         List(
403           OwcOffering(code = WMS.code,
404             operations = List(
405               OwcOperation(
406                 code = "GetCapabilities",
407                 method = "GET",
408                 mimeType = Some("application/xml"),
409                 requestUrl = new URL(s"${owcBaseUrl.toString}/WMSServer?request=GetCapabilities&service=WMS"))
410             )
411           ),
412           OwcOffering(code = WFS.code,
413             operations = List(
414               OwcOperation(
415                 code = "GetCapabilities",
416                 method = "GET",
417                 mimeType = Some("application/xml"),
418                 requestUrl = new URL(s"${owcBaseUrl.toString}/WFSServer?request=GetCapabilities&service=WFS"))
419             )

421         )
