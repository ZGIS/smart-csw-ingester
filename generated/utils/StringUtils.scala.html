29   /**
30     * Safe number converters that provide default values on [[NumberFormatException]]
31     *
32     * @param s
33     */
34   implicit class NumberConverters(val s: String) {
35 
36     import scala.util.control.Exception._
37     /** catcher for [[NumberFormatException]] */
38     private lazy val nfeCatcher = catching(classOf[NumberFormatException])
39 
40     /**
41       * Converts [[String]] to Int with default value on [[NumberFormatException]].
42       *
43       * @param default default value on NumberFormatException. Defaults to 0 if omitted.
44       * @return
45       */
46     def toIntWithDefault(default: Int = 0): Int = {
47       nfeCatcher.opt(s.toInt).getOrElse(default)
48     }
49 
50     /**
51       * Converts [[String]] to [[Double]] with default value on [[NumberFormatException]].
52       *
53       * @param default default value on NumberFormatException. Defaults to 0.0 if omitted.
54       * @return
55       */
56     def toDoubleWithDefault(default: Double = 0.0): Double = {
57       nfeCatcher.opt(s.toDouble).getOrElse(default)
58     }
59   }
60 
61   /**
62     * Converts String to Option
63     * @param s
64     */
65   implicit class OptionConverters(val s: String) {
66 
67     /**
68       * Converts String to Option. None when trimmed String is empty.
69       *
70       * @return Given String encapsulated in Option or None if empty after trimming
71       */
72     def toOption(): Option[String] = {
73       if (s.trim.isEmpty) {
74         None
75       }
76       else {
77         Some(s)
78       }
79     }
80   }
