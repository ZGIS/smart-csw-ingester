281   /**
282     * check if Time info in date/datestamp
283     * check decide dateStamp and/or/plus CI_Date date
284     * worst case create default date
285     *
286     * TODO AK ponder if precise ZoneInfo for parsedDate needed, alternatively either UTC or NZ TimeZone
287     *
288     * @param dateStrings
289     * @return
290     */
291   def dateFromStrings(dateStrings: List[String]): LocalDate = {
292     val yearMonthMatcher = """^\d\d\d\d-\d\d$""".r
293     val yearMatcher = """^\d\d\d\d$""".r
294     val isoInstantMatcher = """.*Z$""".r
295 
296     val acceptedDateFormats = List(
297       DateTimeFormatter.ISO_LOCAL_DATE_TIME, //2011-12-03T10:15:30
298       DateTimeFormatter.ISO_LOCAL_DATE, //2011-12-03
299       DateTimeFormatter.BASIC_ISO_DATE, //20111203
300       DateTimeFormatter.ISO_OFFSET_DATE, //2011-12-03+01:00
301       DateTimeFormatter.ISO_OFFSET_DATE_TIME //2011-12-03T10:15:30+01:00
302       //TODO SR DateTimeFormatter.ISO_INSTANT cannot be parsed into LocalDate but only LocalDateTime?
303       // DateTimeFormatter.ISO_INSTANT //2011-12-03T10:15:30Z
304     )
305 
306     val datesList = dateStrings.filter(!_.trim.isEmpty).flatMap(//iterate over parameter list
307       dateString => acceptedDateFormats.map(//try all parsers per date
308         df => {
309           try {
310             val correctedDateString = dateString match {
311               case yearMonthMatcher() => dateString.concat("-01")
312               case yearMatcher() => dateString.concat("-01-01")
313               case isoInstantMatcher() => dateString.dropRight(1)
314               case _ => dateString
315             }
316             Some(LocalDate.parse(correctedDateString, df))
317           }
318           catch {
319             case e: DateTimeParseException => None
320           }
321         }
322       ).filter(_.isDefined)).map(_.get) //filter None and remove the Option
323 
324     if (datesList.isEmpty) {
325       logger.warn(f"Could not parse any of the dates (${dateStrings.mkString(", ")})")
326       LocalDate.ofEpochDay(0) //1970-01-01
327     }
328     else {
329       if (datesList.size > 1) {
330         logger.info(f"Could parse ${datesList.size} of (${dateStrings.mkString(", ")}) only returning first success")
331       }
332       datesList.head
333     }
334   }

379   /**
380     * extracts hierarchyLevel fields from the provided XML,
381     * there are three places, listed in preferential order:
382     * . gmd:hierarchyLevel \ gmd:MD_ScopeCode \ codeListValue="nonGeographicDataset"
383     * . gmd:hierarchyLevel \ gmd:MD_ScopeCode \ text
384     * . gmd:hierarchyLevelName \ gmd:CharacterString \ text
385     * and can be of typically dataset, service, software, model, sensor, series, nonGeographicDatset
386     *
387     * @param nodeSeq the provided xml
388     * @return a String of the decided hierarchyLevel
389     */
390   def hierarchyLevelFromXml(nodeSeq: NodeSeq): String = {
391     val hierarchyLevelScopeCodeListValueSeq = (nodeSeq \\ "hierarchyLevel" \ "MD_ScopeCode" \ "@codeListValue")
392     val hierarchyLevelScopeTextSeq = (nodeSeq \\ "hierarchyLevel" \ "MD_ScopeCode")
393     val hierarchyLevelNameOptSeq = (nodeSeq \\ "hierarchyLevelName" \ "CharacterString")
394 
395     val levels = (hierarchyLevelScopeCodeListValueSeq ++ hierarchyLevelScopeTextSeq ++ hierarchyLevelNameOptSeq)
396       .map(elem => elem.text.trim).filter(hl => hl.length > 0)
397 
398     levels.headOption.getOrElse("dataset")
399   }

505   /**
506     * tries to extract a valid and usable bounding box rectangle from xml
507     *
508     * @param nodeSeq xml nodeseq containing an extent element
509     * @return [[Rectangle]] the bounding box
510     */
511   def bboxFromXml(nodeSeq: NodeSeq): Rectangle = {
512     val bboxXml = (nodeSeq \\ "extent" \ "EX_Extent")
513     logger.trace(f"bboxFromXml: ${bboxXml}")
514     if (bboxXml.size == 1) {
515       val westText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "westBoundLongitude" \ "Decimal").text
516       val eastText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "eastBoundLongitude" \ "Decimal").text
517       val southText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "southBoundLatitude" \ "Decimal").text
518       val northText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "northBoundLatitude" \ "Decimal").text
519       val west = westText.toDoubleWithDefault(minLon)
520       val east = eastText.toDoubleWithDefault(maxLon)
521       val south = southText.toDoubleWithDefault(minLat)
522       val north = northText.toDoubleWithDefault(maxLat)
523       // Rectangle rect(double minX, double maxX, double minY, double maxY);
524       logger.debug(s"parsed bbox (${west}, ${east}, ${south}, ${north})")
525       bboxFromCoords(west, east, south, north)
526     }
527     else {
528       logger.warn(f"${(nodeSeq \\ "fileIdentifier" \ "CharacterString").text} has no BBOX")
529       bboxFromCoords(minLon, maxLon, minLat, maxLat)
530     }
531     //    bboxFromCoords(165, 360 - 175, -53.1, -28.8)
532   }

534   /**
535     * tries to build a bounding box rectangle as safely as possible from provided coordinates
536     * Rectangle rect(double minX, double maxX, double minY, double maxY);
537     * bboxFromCoords(west, east, south, north)
538     *
539     * @param west  most western value / minX
540     * @param east  most eastern value / maxX
541     * @param south most southern value / minY
542     * @param north most northern value / maxY
543     * @return the resulting bounding box
544     */
545   def bboxFromCoords(west: Double, east: Double, south: Double, north: Double): Rectangle = {
546     val (prunedWest, prunedEast) = pruneLongitudeValues(west, east)
547     val (prunedSouth, prunedNorth) = pruneLatitudeValues(south, north)
548 
549     val rect = ctx.getShapeFactory().rect(prunedWest, prunedEast, prunedSouth, prunedNorth)
550     logger.debug(s"parsed rect ${rect.toString}")
551     rect
552   }
