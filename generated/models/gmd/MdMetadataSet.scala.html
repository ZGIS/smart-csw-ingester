277   /**
278     * check if Time info in date/datestamp
279     * check decide dateStamp and/or/plus CI_Date date
280     * worst case create default date
281     *
282     * TODO AK ponder if precise ZoneInfo for parsedDate needed, alternatively either UTC or NZ TimeZone
283     *
284     * @param dateStrings
285     * @return
286     */
287   def dateFromStrings(dateStrings: List[String]): LocalDate = {
288     val yearMonthMatcher = """^\d\d\d\d-\d\d$""".r
289     val yearMatcher = """^\d\d\d\d$""".r
290     val isoInstantMatcher = """.*Z$""".r
291 
292     val acceptedDateFormats = List(
293       DateTimeFormatter.ISO_LOCAL_DATE_TIME, //2011-12-03T10:15:30
294       DateTimeFormatter.ISO_LOCAL_DATE, //2011-12-03
295       DateTimeFormatter.BASIC_ISO_DATE, //20111203
296       DateTimeFormatter.ISO_OFFSET_DATE, //2011-12-03+01:00
297       DateTimeFormatter.ISO_OFFSET_DATE_TIME //2011-12-03T10:15:30+01:00
298       //TODO SR DateTimeFormatter.ISO_INSTANT cannot be parsed into LocalDate but only LocalDateTime?
299       // DateTimeFormatter.ISO_INSTANT //2011-12-03T10:15:30Z
300     )
301 
302     val datesList = dateStrings.filter(!_.trim.isEmpty).flatMap(//iterate over parameter list
303       dateString => acceptedDateFormats.map(//try all parsers per date
304         df => {
305           try {
306             val correctedDateString = dateString match {
307               case yearMonthMatcher() => dateString.concat("-01")
308               case yearMatcher() => dateString.concat("-01-01")
309               case isoInstantMatcher() => dateString.dropRight(1)
310               case _ => dateString
311             }
312             Some(LocalDate.parse(correctedDateString, df))
313           }
314           catch {
315             case e: DateTimeParseException => None
316           }
317         }
318       ).filter(_.isDefined)).map(_.get) //filter None and remove the Option
319 
320     if (datesList.isEmpty) {
321       logger.warn(f"Could not parse any of the dates (${dateStrings.mkString(", ")})")
322       LocalDate.ofEpochDay(0) //1970-01-01
323     }
324     else {
325       if (datesList.size > 1) {
326         logger.info(f"Could parse ${datesList.size} of (${dateStrings.mkString(", ")}) only returning first success")
327       }
328       datesList.head
329     }
330   }

392 
393     levels.headOption.getOrElse("dataset")

491   def bboxFromXml(nodeSeq: NodeSeq): Rectangle = {
492     val bboxXml = (nodeSeq \\ "extent" \ "EX_Extent")
493     logger.trace(f"bboxFromXml: ${bboxXml}")
494     if (bboxXml.size == 1) {
495       val westText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "westBoundLongitude" \ "Decimal").text
496       val eastText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "eastBoundLongitude" \ "Decimal").text

499       val west = westText.toDoubleWithDefault(minLon)
500       val east = eastText.toDoubleWithDefault(maxLon)
501       val south = southText.toDoubleWithDefault(minLat)
502       val north = northText.toDoubleWithDefault(maxLat)

504       logger.debug(s"parsed bbox (${west}, ${east}, ${south}, ${north})")
505       bboxFromCoords(west, east, south, north)
506     }
507     else {
508       logger.warn(f"${(nodeSeq \\ "fileIdentifier" \ "CharacterString").text} has no BBOX")
509       bboxFromCoords(minLon, maxLon, minLat, maxLat)
510     }
511     //    bboxFromCoords(165, 360 - 175, -53.1, -28.8)

514   /**
515     * tries to build a bounding box rectangle as safely as possible from provided coordinates
516     * Rectangle rect(double minX, double maxX, double minY, double maxY);
517     * bboxFromCoords(west, east, south, north)
518     *
519     * @param west  most western value / minX
520     * @param east  most eastern value / maxX
521     * @param south most southern value / minY
522     * @param north most northern value / maxY
523     * @return the resulting bounding box
524     */
525   def bboxFromCoords(west: Double, east: Double, south: Double, north: Double): Rectangle = {
526     val (prunedWest, prunedEast) = pruneLongitudeValues(west, east)
527     val (prunedSouth, prunedNorth) = pruneLatitudeValues(south, north)
528 
529     val rect = ctx.getShapeFactory().rect(prunedWest, prunedEast, prunedSouth, prunedNorth)
530     logger.debug(s"parsed rect ${rect.toString}")
531     rect
532   }

534   /**
535     * tries to naively prune the provided coordinates into good shape for WSG84
536     * TODO DATE Line Wraps :-( ?
537     * Rectangle rect(double minX, double maxX, double minY, double maxY);
538     * bboxFromCoords(west, east, south, north)
539     *
540     * @param west most western value / minY
541     * @param east most eastern value / maxY
542     * @return tuple of viable coordinates in WSG84
543     */
544   def pruneLongitudeValues(west: Double, east: Double): (Double, Double) = {
545     if (math.abs(west - east) > math.abs(minLon - maxLon)) {
546       (minLon, maxLon) //in case the rectangle spans more than 360 deg make it world
547     }
548     else {
549       val result = List(west, east).map({ (value: Double) =>
550         value match {
551           case n if value >= minLon && value <= maxLon => n
552           case n if math.abs(value % math.abs(minLon - maxLon)) < maxLon => {
553             val result = value % maxLon
554             logger.info(f"changing longitude value $n to $result")
555             result
556           }
557           case _ => {
558             val result = math.signum(value) * minLon + (value % maxLon)
559             logger.info(f"changing longitude value $value to $result")
560             result
561           }
562         }
563       })
