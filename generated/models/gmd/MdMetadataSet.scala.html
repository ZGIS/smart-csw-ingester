238     * extracts a LocalDate from the provided XML
239     *
240     * @param nodeSeq the provided xml
241     * @return a LoclDate from the list of extracted fields
242     */
243   def dateFromXml(nodeSeq: NodeSeq): LocalDate = {
244     val dateNodes = (nodeSeq \\ "dateStamp" \ "Date") ++
245       (nodeSeq \\ "dateStamp" \ "DateTime") ++
246       (nodeSeq \\ "date" \ "CI_Date" \ "date" \ "Date") ++
247       (nodeSeq \\ "date" \ "CI_Date" \ "date" \ "DateTime")
248     dateFromStrings(dateNodes.map(node => node.text).toList)

251   /**
252     * check if Time info in date/datestamp
253     * check decide dateStamp and/or/plus CI_Date date
254     * worst case create default date
255     *
256     * TODO AK ponder if precise ZoneInfo for parsedDate needed, alternatively either UTC or NZ TimeZone
257     *
258     * @param dateStrings
259     * @return
260     */
261   def dateFromStrings(dateStrings: List[String]): LocalDate = {
262     val yearMonthMatcher = """^\d\d\d\d-\d\d$""".r
263     val yearMatcher = """^\d\d\d\d$""".r
264     val isoInstantMatcher = """.*Z$""".r
265 
266     val acceptedDateFormats = List(
267       DateTimeFormatter.ISO_LOCAL_DATE_TIME, //2011-12-03T10:15:30
268       DateTimeFormatter.ISO_LOCAL_DATE, //2011-12-03
269       DateTimeFormatter.BASIC_ISO_DATE, //20111203
270       DateTimeFormatter.ISO_OFFSET_DATE, //2011-12-03+01:00
271       DateTimeFormatter.ISO_OFFSET_DATE_TIME //2011-12-03T10:15:30+01:00
272       //TODO SR DateTimeFormatter.ISO_INSTANT cannot be parsed into LocalDate but only LocalDateTime?
273       // DateTimeFormatter.ISO_INSTANT //2011-12-03T10:15:30Z
274     )
275 
276     val datesList = dateStrings.filter(!_.trim.isEmpty).flatMap(//iterate over parameter list
277       dateString => acceptedDateFormats.map(//try all parsers per date
278         df => {
279           try {
280             val correctedDateString = dateString match {
281               case yearMonthMatcher() => dateString.concat("-01")
282               case yearMatcher() => dateString.concat("-01-01")
283               case isoInstantMatcher() => dateString.dropRight(1)
284               case _ => dateString
285             }
286             Some(LocalDate.parse(correctedDateString, df))
287           }
288           catch {
289             case e: DateTimeParseException => None
290           }
291         }
292       ).filter(_.isDefined)).map(_.get) //filter None and remove the Option
293 
294     if (datesList.isEmpty) {
295       logger.warn(f"Could not parse any of the dates (${dateStrings.mkString(", ")})")
296       LocalDate.ofEpochDay(0) //1970-01-01
297     }
298     else {
299       if (datesList.size > 1) {
300         logger.info(f"Could parse ${datesList.size} of (${dateStrings.mkString(", ")}) only returning first success")
301       }
302       datesList.head
303     }
304   }

441     * @param nodeSeq xml nodeseq containing an extent element
442     * @return [[Rectangle]] the bounding box

444   def bboxFromXml(nodeSeq: NodeSeq): Rectangle = {
445     val bboxXml = (nodeSeq \\ "extent" \ "EX_Extent")
446     logger.trace(f"bboxFromXml: ${bboxXml}")
447     if (bboxXml.size == 1) {
448       val westText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "westBoundLongitude" \ "Decimal").text
449       val eastText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "eastBoundLongitude" \ "Decimal").text
450       val southText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "southBoundLatitude" \ "Decimal").text
451       val northText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "northBoundLatitude" \ "Decimal").text
452       val west = westText.toDoubleWithDefault(minLon)
453       val east = eastText.toDoubleWithDefault(maxLon)
454       val south = southText.toDoubleWithDefault(minLat)
455       val north = northText.toDoubleWithDefault(maxLat)
456       // Rectangle rect(double minX, double maxX, double minY, double maxY);
457       logger.debug(s"parsed bbox (${west}, ${east}, ${south}, ${north})")
458       bboxFromCoords(west, east, south, north)
459     }
460     else {
461       logger.warn(f"${(nodeSeq \\ "fileIdentifier" \ "CharacterString").text} has no BBOX")
462       bboxFromCoords(minLon, maxLon, minLat, maxLat)
463     }
464     //    bboxFromCoords(165, 360 - 175, -53.1, -28.8)

467   /**
468     * tries to build a bounding box rectangle as safely as possible from provided coordinates
469     * Rectangle rect(double minX, double maxX, double minY, double maxY);
470     * bboxFromCoords(west, east, south, north)
471     *
472     * @param west  most western value / minX
473     * @param east  most eastern value / maxX
474     * @param south most southern value / minY
475     * @param north most northern value / maxY
476     * @return the resulting bounding box
477     */
478   def bboxFromCoords(west: Double, east: Double, south: Double, north: Double): Rectangle = {
479     val (prunedWest, prunedEast) = pruneLongitudeValues(west, east)
480     val (prunedSouth, prunedNorth) = pruneLatitudeValues(south, north)
481 
482     val rect = ctx.getShapeFactory().rect(prunedWest, prunedEast, prunedSouth, prunedNorth)
483     logger.debug(s"parsed rect ${rect.toString}")
484     rect
485   }

487   /**
488     * tries to naively prune the provided coordinates into good shape for WSG84
489     * TODO DATE Line Wraps :-( ?
490     * Rectangle rect(double minX, double maxX, double minY, double maxY);
491     * bboxFromCoords(west, east, south, north)
492     *
493     * @param west most western value / minY
494     * @param east most eastern value / maxY
495     * @return tuple of viable coordinates in WSG84
496     */
497   def pruneLongitudeValues(west: Double, east: Double): (Double, Double) = {
498     if (math.abs(west - east) > math.abs(minLon - maxLon)) {
499       (minLon, maxLon) //in case the rectangle spans more than 360 deg make it world
500     }
501     else {
502       val result = List(west, east).map({ (value: Double) =>
503         value match {
504           case n if value >= minLon && value <= maxLon => n
505           case n if math.abs(value % math.abs(minLon - maxLon)) < maxLon => {
506             val result = value % maxLon
507             logger.info(f"changing longitude value $n to $result")
508             result
509           }
510           case _ => {
511             val result = math.signum(value) * minLon + (value % maxLon)
512             logger.info(f"changing longitude value $value to $result")
513             result
514           }
515         }
516       })
517       (result(0), result(1))
518     }
519   }

521   /**
522     * Cuts off latitudes outside of minLax / maxLat and swaps if south > north
523     *
524     * @param south most southern value / minY
525     * @param north most northern value / maxY
526     * @return tuple of viable coordinates
527     */
528   def pruneLatitudeValues(south: Double, north: Double): (Double, Double) = {
529     //min/max in tuples swaps north/south if necessary,
530     (Math.max(minLat, Math.min(south, north)),
531       Math.min(maxLat, Math.max(south, north)))
532   }

546   /**
547     * returns an Option[EmailAddress] object if parsing is successful
548     *
549     * @param emailString the string that might comprise of or contain an email address
550     * @return Option[EmailAddress]
551     */
552   def parseEmailStringtoEmailAddress(emailString: String): Option[EmailAddress] = {
553 
554     if (EmailAddress.isValid(emailString)) {
555       Some(EmailAddress(emailString))
556     } else {
557       if (emailString.contains(",")) {
558         val commaFree = emailString.split(",").head.trim
559 
560         Try {
