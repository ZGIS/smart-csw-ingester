277   /**
278     * check if Time info in date/datestamp
279     * check decide dateStamp and/or/plus CI_Date date
280     * worst case create default date
281     *
282     * TODO AK ponder if precise ZoneInfo for parsedDate needed, alternatively either UTC or NZ TimeZone
283     *
284     * @param dateStrings
285     * @return
286     */
287   def dateFromStrings(dateStrings: List[String]): LocalDate = {
288     val yearMonthMatcher = """^\d\d\d\d-\d\d$""".r
289     val yearMatcher = """^\d\d\d\d$""".r
290     val isoInstantMatcher = """.*Z$""".r
291 
292     val acceptedDateFormats = List(
293       DateTimeFormatter.ISO_LOCAL_DATE_TIME, //2011-12-03T10:15:30
294       DateTimeFormatter.ISO_LOCAL_DATE, //2011-12-03
295       DateTimeFormatter.BASIC_ISO_DATE, //20111203
296       DateTimeFormatter.ISO_OFFSET_DATE, //2011-12-03+01:00
297       DateTimeFormatter.ISO_OFFSET_DATE_TIME //2011-12-03T10:15:30+01:00
298       //TODO SR DateTimeFormatter.ISO_INSTANT cannot be parsed into LocalDate but only LocalDateTime?
299       // DateTimeFormatter.ISO_INSTANT //2011-12-03T10:15:30Z
300     )
301 
302     val datesList = dateStrings.filter(!_.trim.isEmpty).flatMap(//iterate over parameter list
303       dateString => acceptedDateFormats.map(//try all parsers per date
304         df => {
305           try {
306             val correctedDateString = dateString match {
307               case yearMonthMatcher() => dateString.concat("-01")
308               case yearMatcher() => dateString.concat("-01-01")
309               case isoInstantMatcher() => dateString.dropRight(1)
310               case _ => dateString
311             }
312             Some(LocalDate.parse(correctedDateString, df))
313           }
314           catch {
315             case e: DateTimeParseException => None
316           }
317         }
318       ).filter(_.isDefined)).map(_.get) //filter None and remove the Option
319 
320     if (datesList.isEmpty) {
321       logger.warn(f"Could not parse any of the dates (${dateStrings.mkString(", ")})")
322       LocalDate.ofEpochDay(0) //1970-01-01
323     }
324     else {
325       if (datesList.size > 1) {
326         logger.info(f"Could parse ${datesList.size} of (${dateStrings.mkString(", ")}) only returning first success")
327       }
328       datesList.head
329     }
330   }

467     * @param nodeSeq xml nodeseq containing an extent element
468     * @return [[Rectangle]] the bounding box
469     */
470   def bboxFromXml(nodeSeq: NodeSeq): Rectangle = {
471     val bboxXml = (nodeSeq \\ "extent" \ "EX_Extent")
472     logger.trace(f"bboxFromXml: ${bboxXml}")
473     if (bboxXml.size == 1) {
474       val westText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "westBoundLongitude" \ "Decimal").text
475       val eastText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "eastBoundLongitude" \ "Decimal").text
476       val southText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "southBoundLatitude" \ "Decimal").text
477       val northText = (bboxXml \\ "geographicElement" \ "EX_GeographicBoundingBox" \ "northBoundLatitude" \ "Decimal").text
478       val west = westText.toDoubleWithDefault(minLon)
479       val east = eastText.toDoubleWithDefault(maxLon)
480       val south = southText.toDoubleWithDefault(minLat)
481       val north = northText.toDoubleWithDefault(maxLat)
482       // Rectangle rect(double minX, double maxX, double minY, double maxY);
483       logger.debug(s"parsed bbox (${west}, ${east}, ${south}, ${north})")
484       bboxFromCoords(west, east, south, north)
485     }
486     else {
487       logger.warn(f"${(nodeSeq \\ "fileIdentifier" \ "CharacterString").text} has no BBOX")
488       bboxFromCoords(minLon, maxLon, minLat, maxLat)
489     }
490     //    bboxFromCoords(165, 360 - 175, -53.1, -28.8)
491   }

493   /**
494     * tries to build a bounding box rectangle as safely as possible from provided coordinates
495     * Rectangle rect(double minX, double maxX, double minY, double maxY);
496     * bboxFromCoords(west, east, south, north)
497     *
498     * @param west  most western value / minX
499     * @param east  most eastern value / maxX
500     * @param south most southern value / minY
501     * @param north most northern value / maxY
502     * @return the resulting bounding box
503     */
504   def bboxFromCoords(west: Double, east: Double, south: Double, north: Double): Rectangle = {
505     val (prunedWest, prunedEast) = pruneLongitudeValues(west, east)
506     val (prunedSouth, prunedNorth) = pruneLatitudeValues(south, north)
507 
508     val rect = ctx.getShapeFactory().rect(prunedWest, prunedEast, prunedSouth, prunedNorth)
509     logger.debug(s"parsed rect ${rect.toString}")
510     rect
511   }

513   /**
514     * tries to naively prune the provided coordinates into good shape for WSG84
515     * TODO DATE Line Wraps :-( ?
516     * Rectangle rect(double minX, double maxX, double minY, double maxY);
517     * bboxFromCoords(west, east, south, north)
518     *
519     * @param west most western value / minY
520     * @param east most eastern value / maxY
521     * @return tuple of viable coordinates in WSG84
522     */
523   def pruneLongitudeValues(west: Double, east: Double): (Double, Double) = {
524     if (math.abs(west - east) > math.abs(minLon - maxLon)) {
525       (minLon, maxLon) //in case the rectangle spans more than 360 deg make it world
526     }
527     else {
528       val result = List(west, east).map({ (value: Double) =>
529         value match {
530           case n if value >= minLon && value <= maxLon => n
531           case n if math.abs(value % math.abs(minLon - maxLon)) < maxLon => {
532             val result = value % maxLon
533             logger.info(f"changing longitude value $n to $result")
534             result
535           }
536           case _ => {
537             val result = math.signum(value) * minLon + (value % maxLon)
538             logger.info(f"changing longitude value $value to $result")
539             result
540           }
541         }
542       })
543       (result(0), result(1))
544     }
545   }

547   /**
548     * Cuts off latitudes outside of minLax / maxLat and swaps if south > north
549     *
550     * @param south most southern value / minY
551     * @param north most northern value / maxY
552     * @return tuple of viable coordinates
553     */
554   def pruneLatitudeValues(south: Double, north: Double): (Double, Double) = {
555     //min/max in tuples swaps north/south if necessary,
556     (Math.max(minLat, Math.min(south, north)),
557       Math.min(maxLat, Math.max(south, north)))
558   }
