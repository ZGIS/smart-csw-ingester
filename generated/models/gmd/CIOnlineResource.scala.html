67     val name = "data"
68   }

70   case object SERVICE extends ResourceType {
71     val name = "service"
72   }

74   case object METADATA extends ResourceType {
75     val name = "metadata"
76   }
77 
78   case object IMAGE extends ResourceType {
79     val name = "image"

82   def fromString(str: String): ResourceType = str match {
83     case "website" => WEBSITE
84     case "download" => DOWNLOAD
85     case "map" => MAP
86     case "data" => DATA
87     case "service" => SERVICE
88     case "metadata" => METADATA
89     case "image" => IMAGE
90     case str => UnknownResourceType(str)
91   }

99   implicit val writes = Writes[ResourceType](r => JsString(r.name))

108 case class CIOnlineResource(linkage: URL,

149   /**
150     * parse CIOnlineResource from XML
151     *
152     * @param nodeSeq that represents a gmd:CI_OnlineResource
153     * @param origin  name of the cataloue origin (used to determine the link type in some cases)
154     * @see [[https://geo-ide.noaa.gov/wiki/index.php?title=CI_OnlineResource NOAA EDM Wiki - CI_OnlineResource]]
155     * @return
156     */
157   def fromXml(nodeSeq: NodeSeq, origin: String): CIOnlineResource = {
158 
159     import utils.StringUtils._
160     import utils.StringUtils.Regex
161 
162     val urlString = new URL((nodeSeq \ "linkage" \ "URL").text.trim)
163     logger.debug(s"Linkage: ${urlString}")
164 
165     val name = (nodeSeq \ "name" \ "CharacterString").text.trim.toOption()
166     logger.debug(s"Name: ${name}")
167     val description = (nodeSeq \ "description" \ "CharacterString").text.trim.toOption()
168     logger.debug(s"Description: ${description}")
169 
170     val protocol = urlString.toString match {
171       // the GNS CSW has a little bit of a stupid way of assigning protocol to the linkages...
172       case r"https?:\/\/data.gns.cri.nz\/rgmad\/thumbs\/.*" => "WWW:LINK-1.0-http--image-thumbnail".toOption()
173       case r"https?:\/\/data.gns.cri.nz\/rgmad\/(?:services|tiles)\/.*" => "WWW:LINK-1.0-http--link".toOption()
174       case r"https?:\/\/data.gns.cri.nz\/rgmad\/(?:layers|images)\/.*" => "WWW:LINK-1.0-http--download".toOption()
175       case _ => (nodeSeq \ "protocol" \ "CharacterString").text.trim.toOption()
176     }
177     logger.debug(s"Protocol ${protocol}")
178 
179 
180     val resourceType = protocol match {
181       case Some("WWW:LINK-1.0-http--download") => ResourceType.DOWNLOAD
182       case Some("WWW:LINK-1.0-http--downloaddata") => ResourceType.DATA
183       case Some("WWW:LINK-1.0-http--metadata-URL") => ResourceType.METADATA
184       case Some("WWW:LINK-1.0-http--image-thumbnail") => ResourceType.IMAGE
185       case Some("WWW:LINK-1.0-http--link") => urlString.toString match {
186         case r"https?:\/\/data.gns.cri.nz\/rgmad\/thumbs\/.*" => ResourceType.IMAGE
187         case r"https?:\/\/data.gns.cri.nz\/rgmad\/(?:services|tiles)\/.*" => ResourceType.WEBSITE
188         case r"https?:\/\/data.gns.cri.nz\/rgmad\/(?:layers|images)\/.*" => ResourceType.DOWNLOAD
189         case _ => ResourceType.WEBSITE
190       }
191       case Some(proto) if proto.contains("OGC:WMS") => ResourceType.SERVICE
192       case Some(proto) if proto.contains("OGC:WFS") => ResourceType.SERVICE
193       case Some(proto) if proto.contains("OGC:WCS") => ResourceType.SERVICE
194       case Some(proto) if proto.contains("OGC:WPS") => ResourceType.SERVICE
195       case Some(proto) if proto.contains("OGC:CSW") => ResourceType.SERVICE
196       case Some(proto) if proto.contains("OGC:SOS") => ResourceType.SERVICE
197 
198       case _ => urlString.toString match {
199         case r"https?:\/\/geoportal\.doc\.govt\.nz\/(?i:ArcGIS)\/.*\/MapServer" => ResourceType.METADATA
200         case r"https?:\/\/data.linz.govt.nz\/layer\/.*" => ResourceType.MAP
201         case r"https?:\/\/data.mfe.govt.nz\/layer\/.*" => ResourceType.MAP
202         case r"https?:\/\/lris.scinfo.org.nz\/layer\/.*" => ResourceType.MAP
203         case _ => ResourceType.WEBSITE
204       }
205     }
206     logger.debug(s"ResourceType: ${resourceType}")
207 
208     CIOnlineResource(urlString, name, description, protocol, resourceType)
209   }
