157   /**
158     * Builds a seq of Lucene Documents from a given CswGetRecordsResponse object
159     *
160     * @param cswGetRecordsResponse
161     * @param catalogueName
162     * @return
163     */
164   private def extractLuceneDocsFromResponse(cswGetRecordsResponse: CswGetRecordsResponse, catalogueName: String): Map[String, Document] = {
165     logger.info(s"Start building (partial) index for ${catalogueName}...")
166     val mdMetadataSet = (cswGetRecordsResponse.xml \\ "MD_Metadata").map(mdMetadataNode => {
167       logger.debug(f"Preparing($catalogueName): ${(mdMetadataNode \ "fileIdentifier" \ "CharacterString").text}")
168       logger.trace(mdMetadataNode.toString)
169       MdMetadataSet.fromXml(mdMetadataNode, catalogueName, luceneService.getCatalogueUrl(catalogueName))
170     }).filter(item => item.isDefined).map(item => item.get) //filter out all None values
171 
172     mdMetadataSet.map(md => (md.fileIdentifier, md.asLuceneDocument)).toMap
173   }

175   /**
176     * Builds a Lucene Index from a given CswGetRecordsResponse object
177     *
178     * @param intermediateSetofDocuments Set[Document] to build an index from
179     * @param catalogueName         String containing the catalogue name
180     * @return [[Directory]] containing the index
181     */
182   @deprecated
183   private def buildIndexFor(intermediateSetofDocuments: Map[String, Document], catalogueName: String): Directory = {
184 
185     val directory = new RAMDirectory()
186     val config = new IndexWriterConfig()
187     config.setCommitOnClose(true)
188 
189     //FIXME SR use SCALA_ARM (automated resource management)?
190     val iwriter = new IndexWriter(directory, config)
191     try {
192       iwriter.deleteAll()
193       iwriter.commit()
194       intermediateSetofDocuments.foreach(tup => iwriter.addDocument(tup._2))
195       iwriter.commit()
196       logger.info(s"Partial index ready for $catalogueName")
197     }
198     finally {
199       iwriter.close()
200     }
201     directory
202   }

204   /**
205     * it all happens here, we could have tried tailrec annotation to work in constant heap?
206     *
207     * @param catalogueName
208     * @param catalogueUrl
209     * @param startDocument
210     * @param documentsToFetch

212   private def queryCatalogue(catalogueName: String, catalogueUrl: String, startDocument: Int,
213                              documentsToFetch: Int): Unit = {
214     val wsClientResponseFuture =
215       wsClient.url(catalogueUrl)
216 //        .withRequestTimeout(20.seconds)
217         .withHeaders("Content-Type" -> "application/xml")
218         .post(CswGetRecordsRequest(startDocument, documentsToFetch))
219 
220     wsClientResponseFuture.onComplete {
221       case Success(wsClientResponse) => {
222         wsClientResponse.status match {
223           case Status.OK => {
224             logger.debug(f"Response Content Type: ${wsClientResponse.allHeaders.getOrElse("Content-Type", "Unknown")}")
225             logger.debug(f"Response-Length: ${wsClientResponse.body.length}")
226             logger.trace(f"Response-Body: ${wsClientResponse.body.toString}")
227             wsClientResponse.xml.label match {
228               case "ExceptionReport" => {
229                 logger.warn(
230                   f"Got XML Exception Response. Text: ${(wsClientResponse.xml \ "Exception" \ "ExceptionText").text}")
231               }
232               case "GetRecordsResponse" => {
233                 val cswGetRecResp = CswGetRecordsResponse(wsClientResponse.xml)
234                 logger.info(
235                   f"c: $catalogueName nextRecord: ${cswGetRecResp.nextRecord}, " +
236                     f"numberOfRec ${cswGetRecResp.numberOfRecordsMatched}, " +
237                     f"recordsReturned ${cswGetRecResp.numberOfRecordsReturned}")
238                 if ((cswGetRecResp.nextRecord > cswGetRecResp.numberOfRecordsMatched) ||
239                   (cswGetRecResp.nextRecord == 0) ||
240                   (cswGetRecResp.numberOfRecordsReturned == 0)) {
241                   logger.info(s"Sending IndexResponseDocument - is last! ($catalogueName)")
242                   self ! IndexResponseDocument(cswGetRecResp, catalogueName, true)
243                 }
244                 else {
245                   logger.info(s"Sending IndexResponseDocument - and start another query round. ($catalogueName)")
246                   self ! IndexResponseDocument(cswGetRecResp, catalogueName, false)
247                   self ! QueryCatalogueNext(catalogueName, catalogueUrl, cswGetRecResp.nextRecord, documentsToFetch)
248                 }
249               }
250               case _ => {
251                 logger.warn(f"Unknown response content. Body: ${wsClientResponse.xml.toString} ($catalogueName)")
252                 self ! ShutdownActor(catalogueName)
253               }
254             }
255           }
256           case _ => {
257             self ! ShutdownActor(catalogueName)
258           }
259         }
260       }
261       case Failure(ex) => {
262         logger.warn(s"Exception while querying CSW $catalogueName (${ex.getClass.getCanonicalName}): ${ex.getMessage}", ex)
263         self ! ShutdownActor(catalogueName)
264       }
265     }
266   }
