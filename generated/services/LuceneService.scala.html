199

201     val filename = s"$catalogueName.avro"
202     val keys = docs.keys.toSet

221     config.setCommitOnClose(true)
222     config.setOpenMode(OpenMode.CREATE_OR_APPEND)

224     try {
225       docs.foreach(

227         tup => iwriter.updateDocument(new Term("id", tup._1), tup._2))
228 
229       iwriter.commit()
230       // iwriter.forceMerge(1, true)
231       logger.info(s"Update-merging index $catalogueName into main, ready ")

234         diffSet.map(id => try {

237 
238       iwriter.commit()
239       // iwriter.forceMerge(1, true)
240       logger.info(s"Update-merging remove old ids from $catalogueName into main, ready ")

242     }
243     finally {

245     }
246 
247     // persist the new state for the catalogue
248     val avroOutTry = Try(AvroOutputStream.data[CatalogueIdState](avroFile))
249     avroOutTry.failed.map(ex => logger.warn(s"persisting catalogue state not successful: ${ex.getLocalizedMessage}"))
250     avroOutTry.toOption.foreach(avroOut =>

252         avroOut.write(CatalogueIdState(catalogueName, keys.toSeq))

299   def query(query: String,
300             bboxWtk: Option[String] = None,
301             fromDate: Option[LocalDate] = None,
302             toDate: Option[LocalDate] = None,
303             maxNumberOfResults: Option[Int] = None): SearchResult = {
304 
305     val textQuery = parseQueryString(query)
306 
307     val dateQuery = LongPoint.newRangeQuery("dateStamp",
308       fromDate.getOrElse(LocalDate.ofEpochDay(0)).toEpochDay(),
309       toDate.getOrElse(LocalDate.now()).toEpochDay())
310 
311     val bboxQuery = parseBboxQuery(bboxWtk.filterNot(_.isEmpty).getOrElse(WORLD_WKT))
312 
313     searcherManager.maybeRefreshBlocking()
314     val isearcher = searcherManager.acquire()
315 
316     // TODO TBD AK The way you built it, could omit dates or BBOX query constraint in this
317     // this builder if not provided through query params (but particularly date should NOT be limited to 1970 to now?)
318     val booleanQueryBuilder = new BooleanQuery.Builder()
319     booleanQueryBuilder.add(textQuery, BooleanClause.Occur.MUST)
320     booleanQueryBuilder.add(dateQuery, BooleanClause.Occur.MUST)
321     booleanQueryBuilder.add(bboxQuery, BooleanClause.Occur.MUST)
322     val finalLuceneQuery = booleanQueryBuilder.build()
323 
324     val numOfMatchingDocuments = isearcher.count(finalLuceneQuery)
325 
326     // TODO SR is "all" a good default, when queried for 0 or less documents?
327     val maxDocuments = maxNumberOfResults match {
328       case Some(x) if maxNumberOfResults.get <= 0 => numOfMatchingDocuments + 1 //+1 just in case 0 docs match
329       case _ => maxNumberOfResults.getOrElse(defaultMaxDocuments)
330     }
331     val search = isearcher.search(finalLuceneQuery, maxDocuments)
332     val scoreDocs = search.scoreDocs
333 
334     val results = scoreDocs.map(scoreDoc => {
335       val doc = isearcher.doc(scoreDoc.doc)
336       MdMetadataSet.fromLuceneDoc(doc, scoreDoc.score)
337     }).toList
338 
339     //FIXME SR use ARM --> possible mem leak
340     searcherManager.release(isearcher)
341 
342     new SearchResult {
343       override val luceneQuery: Query = finalLuceneQuery
344       override val numberOfMatchingDocuments: Int = numOfMatchingDocuments
345       override val documents: List[MdMetadataSet] = results
346     }
347   }

376   /**
377     * creates a query, that should find anything inside a BBOX
378     *
379     * @param bboxWkt
380     * @return
381     */
382   private def parseBboxQuery(bboxWkt: String) = {
383     logger.debug(s"create query for $bboxWkt")
384 
385     val envelopeWktRegex =
386       "ENVELOPE\\(([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+)\\)".r
387 
388     val shape = bboxWkt.replaceAll("\\s", "") match {
389       case envelopeWktRegex(minX, maxX, minY, maxY) => {
390         MdMetadataSet.bboxFromCoords(minX.toFloat, maxX.toFloat, minY.toFloat, maxY.toFloat)
391       }
392       case _ => {
393         logger.warn(s"Could not parse WKT '$bboxWkt'. Using WORLD as query BBOX.");
394         MdMetadataSet.bboxFromCoords(-180, 180, -90, 90);
395       }
396     }
397 
398     logger.debug(s"parsed shape is ${shape.toString}")
399     val bboxStrategy: BBoxStrategy = BBoxStrategy.newInstance(SPATIAL_CONTEXT, "bbox")
400     // bboxStrategy.makeQuery(new SpatialArgs(SpatialOperation.IsWithin, shape))
401     // FIXME AK intersects vs iswithin, should be selectable
402     bboxStrategy.makeQuery(new SpatialArgs(SpatialOperation.Intersects, shape))
403   }

430   /**
431     * create a filesystem based lucene directory
432     *
433     * @param indexBaseFolder
434     * @return
435     */
436   private def openOrCreateFileSystemIndex(indexBaseFolder: String): Directory = {
437 
438     val baseIndexDir = Paths.get(indexBaseFolder)
439     if (!Files.isWritable(baseIndexDir)) {
440       val errmsg = s"Main index storage directory ${baseIndexDir.toAbsolutePath} does not exist or is not writable, please check the path"
441       logger.error(errmsg)
442       throw new IOException(errmsg)
443     }
444     val jointIndexDir = new File(Paths.get(indexBaseFolder) + File.separator + "main")
445 
446     val preExistent = if (Files.isWritable(jointIndexDir.toPath)) {
447       true
448     } else {
449       if (jointIndexDir.mkdir()) {
450         false
451       } else {
452         val errmsg = s"Joint index sub-directory ${jointIndexDir.toPath.toString} could not be created"
453         logger.error(errmsg)
454         throw new IOException(errmsg)
455       }
456     }
457 
458     val directory = FSDirectory.open(jointIndexDir.toPath)
459     val analyzer = new StandardAnalyzer()
460     val config = new IndexWriterConfig(analyzer)
461     config.setCommitOnClose(true)
462     if (preExistent) {
463       // Add new documents to an existing index:
464       config.setOpenMode(OpenMode.CREATE_OR_APPEND)
465       val iwriter = new IndexWriter(directory, config)
466 
467       try {
468         val emptyDoc = new Document()
469         // id field not used in our search semantics, only to provide lucene uniqueness
470         emptyDoc.add(new StringField("id", "00000000-0000-0000-0000-000000000000", Field.Store.NO))
471         iwriter.updateDocument(new Term("fileIdentifier", "00000000-0000-0000-0000-000000000000"), emptyDoc)
472         iwriter.commit()
473         logger.info("Re-opened main index ready")
474       }
475       finally {
476         iwriter.close()
477       }
478     } else {
479       // Create a new index in the directory, removing any
480       // previously indexed documents:
481       config.setOpenMode(OpenMode.CREATE)
482       val iwriter = new IndexWriter(directory, config)
483 
484       try {
485         iwriter.deleteAll()
486         iwriter.addDocument(new Document)
487         iwriter.commit()
488         logger.info("Empty main index ready")
489       }
490       finally {
491         iwriter.close()
492       }
493     }
494 
495     directory
496   }
