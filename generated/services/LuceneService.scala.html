141         logger.info(s"Schedule for next build index of $catalogueName is secDelay: $secDelay and hoursRepeat: $hoursRepeat")

143           buildIndex(catalogueName)

244   /**
245     * Queries the Search Index
246     *
247     * @param query
248     * @return
249     */
250   def query(query: String,
251             bboxWtk: Option[String] = None,
252             fromDate: Option[LocalDate] = None,
253             toDate: Option[LocalDate] = None,
254             maxNumberOfResults: Option[Int] = None): SearchResult = {
255 
256     val textQuery = parseQueryString(query)
257 
258     val dateQuery = LongPoint.newRangeQuery("dateStamp",
259       fromDate.getOrElse(LocalDate.ofEpochDay(0)).toEpochDay(),
260       toDate.getOrElse(LocalDate.now()).toEpochDay())
261 
262     val bboxQuery = parseBboxQuery(bboxWtk.filterNot(_.isEmpty).getOrElse(WORLD_WKT))
263 
264     searcherManager.maybeRefreshBlocking()
265     val isearcher = searcherManager.acquire()
266 
267     // TODO TBD AK The way you built it, could omit dates or BBOX query constraint in this
268     // this builder if not provided through query params (but particularly date should NOT be limited to 1970 to now?)
269     val booleanQueryBuilder = new BooleanQuery.Builder()
270     booleanQueryBuilder.add(textQuery, BooleanClause.Occur.MUST)
271     booleanQueryBuilder.add(dateQuery, BooleanClause.Occur.MUST)
272     booleanQueryBuilder.add(bboxQuery, BooleanClause.Occur.MUST)
273     val finalLuceneQuery = booleanQueryBuilder.build()
274 
275     val numOfMatchingDocuments = isearcher.count(finalLuceneQuery)
276 
277     // TODO SR is "all" a good default, when queried for 0 or less documents?
278     val maxDocuments = maxNumberOfResults match {
279       case Some(x) if maxNumberOfResults.get <= 0 => numOfMatchingDocuments + 1 //+1 just in case 0 docs match
280       case _ => maxNumberOfResults.getOrElse(defaultMaxDocuments)
281     }
282     val search = isearcher.search(finalLuceneQuery, maxDocuments)
283     val scoreDocs = search.scoreDocs
284 
285     val results = scoreDocs.map(scoreDoc => {
286       val doc = isearcher.doc(scoreDoc.doc)
287       MdMetadataSet.fromLuceneDoc(doc, scoreDoc.score)
288     }).toList
289 
290     //FIXME SR use ARM --> possible mem leak
291     searcherManager.release(isearcher)
292 
293     new SearchResult {
294       override val luceneQuery: Query = finalLuceneQuery
295       override val numberOfMatchingDocuments: Int = numOfMatchingDocuments
296       override val documents: List[MdMetadataSet] = results
297     }
298   }

327   /**
328     * creates a query, that should find anything inside a BBOX
329     *
330     * @param bboxWkt
331     * @return
332     */
333   private def parseBboxQuery(bboxWkt: String) = {
334     logger.debug(s"create query for $bboxWkt")
335 
336     val envelopeWktRegex =
337       "ENVELOPE\\(([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+)\\)".r
338 
339     val shape = bboxWkt.replaceAll("\\s", "") match {
340       case envelopeWktRegex(minX, maxX, minY, maxY) => {
341         MdMetadataSet.bboxFromCoords(minX.toFloat, maxX.toFloat, minY.toFloat, maxY.toFloat)
342       }
343       case _ => {
344         logger.warn(s"Could not parse WKT '$bboxWkt'. Using WORLD as query BBOX.");
345         MdMetadataSet.bboxFromCoords(-180, 180, -90, 90);
346       }
347     }
348 
349     logger.debug(s"parsed shape is ${shape.toString}")
350     val bboxStrategy: BBoxStrategy = BBoxStrategy.newInstance(SPATIAL_CONTEXT, "bbox")
351     // bboxStrategy.makeQuery(new SpatialArgs(SpatialOperation.IsWithin, shape))
352     // FIXME AK intersects vs iswithin, should be selectable
353     bboxStrategy.makeQuery(new SpatialArgs(SpatialOperation.Intersects, shape))
354   }

381   /**
382     * create a filesystem based lucene directory
383     *
384     * @param indexBaseFolder
385     * @return
386     */
387   private def openOrCreateFileSystemIndex(indexBaseFolder: String): Directory = {
388 
389     val baseIndexDir = Paths.get(indexBaseFolder)
390     if (!Files.isWritable(baseIndexDir)) {
391       val errmsg = s"Main index storage directory ${baseIndexDir.toAbsolutePath} does not exist or is not writable, please check the path"
392       logger.error(errmsg)
393       throw new IOException(errmsg)
394     }
395     val jointIndexDir = new File(Paths.get(indexBaseFolder) + File.separator + "main")
396 
397     val preExistent = if (Files.isWritable(jointIndexDir.toPath)) {
398       true
399     } else {
400       if (jointIndexDir.mkdir()) {
401         false
402       } else {
403         val errmsg = s"Joint index sub-directory ${jointIndexDir.toPath.toString} could not be created"
404         logger.error(errmsg)
405         throw new IOException(errmsg)
406       }
407     }
408 
409     val directory = FSDirectory.open(jointIndexDir.toPath)
410     val analyzer = new StandardAnalyzer()
411     val config = new IndexWriterConfig(analyzer)
412     config.setCommitOnClose(true)
413     if (preExistent) {
414       // Add new documents to an existing index:
415       config.setOpenMode(OpenMode.CREATE_OR_APPEND)
416       val iwriter = new IndexWriter(directory, config)
417 
418       try {
419         val emptyDoc = new Document()
420         // id field not used in our search semantics, only to provide lucene uniqueness
421         emptyDoc.add(new StringField("id", "00000000-0000-0000-0000-000000000000", Field.Store.NO))
422         iwriter.updateDocument(new Term("fileIdentifier", "00000000-0000-0000-0000-000000000000"), emptyDoc)
423         iwriter.commit()
424         logger.info("Re-opened main index ready")
425       }
426       finally {
427         iwriter.close()
428       }
429     } else {
430       // Create a new index in the directory, removing any
431       // previously indexed documents:
432       config.setOpenMode(OpenMode.CREATE)
433       val iwriter = new IndexWriter(directory, config)
434 
435       try {
436         iwriter.deleteAll()
437         iwriter.addDocument(new Document)
438         iwriter.commit()
439         logger.info("Empty main index ready")
440       }
441       finally {
442         iwriter.close()
443       }
444     }
445 
446     directory
447   }
