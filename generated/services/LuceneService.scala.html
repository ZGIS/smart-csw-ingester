136   def buildIndex(): Unit = {
137     catalogues.map({
138       case (catalogueName, catalogueUrl) =>
139         val secDelay = if (environment.mode.equals(Mode.Prod)) 5 + scala.util.Random.nextInt(56) else 1
140         val hoursRepeat = 24 + scala.util.Random.nextInt(49)
141         logger.info(s"Schedule for next build index of $catalogueName is secDelay: $secDelay and hoursRepeat: $hoursRepeat")
142         system.scheduler.schedule(secDelay.second, hoursRepeat.hours){
143           buildIndex(catalogueName)
144         }

211   /**
212     * Queries the Search Index
213     *
214     * @param query
215     * @return
216     */
217   def query(query: String,
218             bboxWtk: Option[String] = None,
219             fromDate: Option[LocalDate] = None,
220             toDate: Option[LocalDate] = None,
221             maxNumberOfResults: Option[Int] = None): SearchResult = {
222 
223     val textQuery = parseQueryString(query)
224 
225     val dateQuery = LongPoint.newRangeQuery("dateStamp",
226       fromDate.getOrElse(LocalDate.ofEpochDay(0)).toEpochDay(),
227       toDate.getOrElse(LocalDate.now()).toEpochDay())
228 
229     val bboxQuery = parseBboxQuery(bboxWtk.filterNot(_.isEmpty).getOrElse(WORLD_WKT))
230 
231     searcherManager.maybeRefreshBlocking()
232     val isearcher = searcherManager.acquire()
233 
234     // TODO TBD AK The way you built it, could omit dates or BBOX query constraint in this
235     // this builder if not provided through query params (but particularly date should NOT be limited to 1970 to now?)
236     val booleanQueryBuilder = new BooleanQuery.Builder()
237     booleanQueryBuilder.add(textQuery, BooleanClause.Occur.MUST)
238     booleanQueryBuilder.add(dateQuery, BooleanClause.Occur.MUST)
239     booleanQueryBuilder.add(bboxQuery, BooleanClause.Occur.MUST)
240     val finalLuceneQuery = booleanQueryBuilder.build()
241 
242     val numOfMatchingDocuments = isearcher.count(finalLuceneQuery)
243 
244     // TODO SR is "all" a good default, when queried for 0 or less documents?
245     val maxDocuments = maxNumberOfResults match {
246       case Some(x) if maxNumberOfResults.get <= 0 => numOfMatchingDocuments + 1 //+1 just in case 0 docs match
247       case _ => maxNumberOfResults.getOrElse(defaultMaxDocuments)
248     }
249     val search = isearcher.search(finalLuceneQuery, maxDocuments)
250     val scoreDocs = search.scoreDocs
251 
252     val results = scoreDocs.map(scoreDoc => {
253       val doc = isearcher.doc(scoreDoc.doc)
254       MdMetadataSet.fromLuceneDoc(doc, scoreDoc.score)
255     }).toList
256 
257     //FIXME SR use ARM --> possible mem leak
258     searcherManager.release(isearcher)
259 
260     new SearchResult {
261       override val luceneQuery: Query = finalLuceneQuery
262       override val numberOfMatchingDocuments: Int = numOfMatchingDocuments
263       override val documents: List[MdMetadataSet] = results
264     }
265   }

293   /**
294     * creates a query, that should find anything inside a BBOX
295     *
296     * @param bboxWkt
297     * @return
298     */
299   private def parseBboxQuery(bboxWkt: String) = {
300     logger.debug(s"create query for $bboxWkt")
301 
302     val envelopeWktRegex =
303       "ENVELOPE\\(([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+)\\)".r
304 
305     val shape = bboxWkt.replaceAll("\\s", "") match {
306       case envelopeWktRegex(minX, maxX, minY, maxY) => {
307         MdMetadataSet.bboxFromCoords(minX.toFloat, maxX.toFloat, minY.toFloat, maxY.toFloat)
308       }
309       case _ => {
310         logger.warn(s"Could not parse WKT '$bboxWkt'. Using WORLD as query BBOX.");
311         MdMetadataSet.bboxFromCoords(-180, 180, -90, 90);
312       }
313     }
314 
315     logger.debug(s"parsed shape is ${shape.toString}")
316     val bboxStrategy: BBoxStrategy = BBoxStrategy.newInstance(SPATIAL_CONTEXT, "bbox")
317     // bboxStrategy.makeQuery(new SpatialArgs(SpatialOperation.IsWithin, shape))
318     // FIXME AK intersects vs iswithin, should be selectable
319     bboxStrategy.makeQuery(new SpatialArgs(SpatialOperation.Intersects, shape))
320   }

347   /**
348     * create a filesystem based lucene directory
349     *
350     * @param indexBaseFolder
351     * @return
352     */
353   private def openOrCreateFileSystemIndex(indexBaseFolder: String): Directory = {
354 
355     val baseIndexDir = Paths.get(indexBaseFolder)
356     if (!Files.isWritable(baseIndexDir)) {
357       val errmsg = s"Main index storage directory ${baseIndexDir.toAbsolutePath} does not exist or is not writable, please check the path"
358       logger.error(errmsg)
359       throw new IOException(errmsg)
360     }
361     val jointIndexDir = new File(Paths.get(indexBaseFolder) + File.separator + "main")
362 
363     val preExistent = if (Files.isWritable(jointIndexDir.toPath)) {
364       true
365     } else {
366       if (jointIndexDir.mkdir()) {
367         false
368       } else {
369         val errmsg = s"Joint index sub-directory ${jointIndexDir.toPath.toString} could not be created"
370         logger.error(errmsg)
371         throw new IOException(errmsg)
372       }
373     }
374 
375     val directory = FSDirectory.open(jointIndexDir.toPath)
376     val analyzer = new StandardAnalyzer()
377     val config = new IndexWriterConfig(analyzer)
378     config.setCommitOnClose(true)
379     if (preExistent) {
380       // Add new documents to an existing index:
381       config.setOpenMode(OpenMode.CREATE_OR_APPEND)
382       val iwriter = new IndexWriter(directory, config)
383 
384       try {
385         val emptyDoc = new Document()
386         // id field not used in our search semantics, only to provide lucene uniqueness
387         emptyDoc.add(new StringField("id", "00000000-0000-0000-0000-000000000000", Field.Store.NO))
388         iwriter.updateDocument(new Term("fileIdentifier","00000000-0000-0000-0000-000000000000"), emptyDoc)
389         iwriter.commit()
390         logger.info("Re-opened main index ready")
391       }
392       finally {
393         iwriter.close()
394       }
395     } else {
396       // Create a new index in the directory, removing any
397       // previously indexed documents:
398       config.setOpenMode(OpenMode.CREATE)
399       val iwriter = new IndexWriter(directory, config)
400 
401       try {
402         iwriter.deleteAll()
403         iwriter.addDocument(new Document)
404         iwriter.commit()
405         logger.info("Empty main index ready")
406       }
407       finally {
408         iwriter.close()
409       }
410     }
411 
412     directory
413   }
