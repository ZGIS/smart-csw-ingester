199   def mergeUpdateDocsIndex(docs: Map[String, Document], catalogueName: String): Unit = {
200 
201     // try load persisted catalogue state id list
202     val filename = s"$catalogueName.avro"
203     val keys = docs.keys.toSet
204     val avroFile = new File(catalogueStateDirectory + File.separator + filename)
205     val avroInTry = Try(AvroInputStream.data[CatalogueIdState](avroFile))
206     avroInTry.failed.map(ex => logger.warn(s"reading catalogue state not successful: ${ex.getLocalizedMessage}"))
207     val oldCatalogueStateOption = avroInTry.toOption.flatMap { avroIn =>
208       val oldCatalogueState = avroIn.iterator.toSeq.headOption
209       avroIn.close()
210       oldCatalogueState.foreach(cs => logger.info(s"restored list of ${cs.fileIdentifiers.size} for ${cs.catalogueName}"))
211       oldCatalogueState
212     }
213 
214     val inOldButnotInNew: Option[Set[String]] = oldCatalogueStateOption.map { cs =>
215       cs.fileIdentifiers.toSet.diff(keys)
216     }
217     // if existent calcluate the diff of file ids in the old loaded list vs the file ids in the current merge opearation
218 
219     // if ids in old, but not in new, go through the lucene directory and try to delete these by id from th ecurrent index
220 
221     val config = new IndexWriterConfig()
222     config.setCommitOnClose(true)
223     config.setOpenMode(OpenMode.CREATE_OR_APPEND)
224     val iwriter = new IndexWriter(indexDirectory, config)
225     try {
226       docs.foreach(
227         // don't use, only for index uniqueness and doc update
228         tup => iwriter.updateDocument(new Term("id", tup._1), tup._2))
229 
230       iwriter.commit()
231       // iwriter.forceMerge(1, true)
232       logger.info(s"Update-merging index $catalogueName into main, ready ")
233 
234       inOldButnotInNew.map(diffSet =>
235         diffSet.map(id => try {
236           iwriter.deleteDocuments(new Term("id", id))
237         }))
238 
239       iwriter.commit()
240       // iwriter.forceMerge(1, true)
241       logger.info(s"Update-merging remove old ids from $catalogueName into main, ready ")
242 
243     }
244     finally {
245       iwriter.close()
246     }
247 
248     // persist the new state for the catalogue
249     val avroOutTry = Try(AvroOutputStream.data[CatalogueIdState](avroFile))
250     avroOutTry.failed.map(ex => logger.warn(s"persisting catalogue state not successful: ${ex.getLocalizedMessage}"))
251     avroOutTry.toOption.foreach(avroOut =>
252       try {
253         avroOut.write(CatalogueIdState(catalogueName, keys.toSeq))
254         avroOut.flush()
255         logger.info(s"Updating catalogue state persistence for $catalogueName, ready ")
256       } finally {
257         avroOut.close()
258       })

295     * Queries the Search Index
296     *
297     * @param query
298     * @return
299     */
300   def query(query: String,
301             bboxWtk: Option[String] = None,
302             fromDate: Option[LocalDate] = None,
303             toDate: Option[LocalDate] = None,
304             maxNumberOfResults: Option[Int] = None): SearchResult = {
305 
306     val textQuery = parseQueryString(query)
307 
308     val dateQuery = LongPoint.newRangeQuery("dateStamp",
309       fromDate.getOrElse(LocalDate.ofEpochDay(0)).toEpochDay(),
310       toDate.getOrElse(LocalDate.now()).toEpochDay())
311 
312     val bboxQuery = parseBboxQuery(bboxWtk.filterNot(_.isEmpty).getOrElse(WORLD_WKT))
313 
314     searcherManager.maybeRefreshBlocking()
315     val isearcher = searcherManager.acquire()
316 
317     // TODO TBD AK The way you built it, could omit dates or BBOX query constraint in this
318     // this builder if not provided through query params (but particularly date should NOT be limited to 1970 to now?)
319     val booleanQueryBuilder = new BooleanQuery.Builder()
320     booleanQueryBuilder.add(textQuery, BooleanClause.Occur.MUST)
321     booleanQueryBuilder.add(dateQuery, BooleanClause.Occur.MUST)
322     booleanQueryBuilder.add(bboxQuery, BooleanClause.Occur.MUST)
323     val finalLuceneQuery = booleanQueryBuilder.build()
324 
325     val numOfMatchingDocuments = isearcher.count(finalLuceneQuery)
326 
327     // TODO SR is "all" a good default, when queried for 0 or less documents?
328     val maxDocuments = maxNumberOfResults match {
329       case Some(x) if maxNumberOfResults.get <= 0 => numOfMatchingDocuments + 1 //+1 just in case 0 docs match
330       case _ => maxNumberOfResults.getOrElse(defaultMaxDocuments)
331     }
332     val search = isearcher.search(finalLuceneQuery, maxDocuments)
333     val scoreDocs = search.scoreDocs
334 
335     val results = scoreDocs.map(scoreDoc => {
336       val doc = isearcher.doc(scoreDoc.doc)
337       MdMetadataSet.fromLuceneDoc(doc, scoreDoc.score)
338     }).toList
339 
340     //FIXME SR use ARM --> possible mem leak
341     searcherManager.release(isearcher)
342 
343     new SearchResult {
344       override val luceneQuery: Query = finalLuceneQuery
345       override val numberOfMatchingDocuments: Int = numOfMatchingDocuments
346       override val documents: List[MdMetadataSet] = results
347     }
348   }

377   /**
378     * creates a query, that should find anything inside a BBOX
379     *
380     * @param bboxWkt
381     * @return
382     */
383   private def parseBboxQuery(bboxWkt: String) = {
384     logger.debug(s"create query for $bboxWkt")
385 
386     val envelopeWktRegex =
387       "ENVELOPE\\(([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+),([-+]?[0-9]*\\.?[0-9]+)\\)".r
388 
389     val shape = bboxWkt.replaceAll("\\s", "") match {
390       case envelopeWktRegex(minX, maxX, minY, maxY) => {
391         MdMetadataSet.bboxFromCoords(minX.toFloat, maxX.toFloat, minY.toFloat, maxY.toFloat)
392       }
393       case _ => {
394         logger.warn(s"Could not parse WKT '$bboxWkt'. Using WORLD as query BBOX.");
395         MdMetadataSet.bboxFromCoords(-180, 180, -90, 90);
396       }
397     }
398 
399     logger.debug(s"parsed shape is ${shape.toString}")
400     val bboxStrategy: BBoxStrategy = BBoxStrategy.newInstance(SPATIAL_CONTEXT, "bbox")
401     // bboxStrategy.makeQuery(new SpatialArgs(SpatialOperation.IsWithin, shape))
402     // FIXME AK intersects vs iswithin, should be selectable
403     bboxStrategy.makeQuery(new SpatialArgs(SpatialOperation.Intersects, shape))
404   }

431   /**
432     * create a filesystem based lucene directory
433     *
434     * @param indexBaseFolder
435     * @return
436     */
437   private def openOrCreateFileSystemIndex(indexBaseFolder: String): Directory = {
438 
439     val baseIndexDir = Paths.get(indexBaseFolder)
440     if (!Files.isWritable(baseIndexDir)) {
441       val errmsg = s"Main index storage directory ${baseIndexDir.toAbsolutePath} does not exist or is not writable, please check the path"
442       logger.error(errmsg)
443       throw new IOException(errmsg)
444     }
445     val jointIndexDir = new File(Paths.get(indexBaseFolder) + File.separator + "main")
446 
447     val preExistent = if (Files.isWritable(jointIndexDir.toPath)) {
448       true
449     } else {
450       if (jointIndexDir.mkdir()) {
451         false
452       } else {
453         val errmsg = s"Joint index sub-directory ${jointIndexDir.toPath.toString} could not be created"
454         logger.error(errmsg)
455         throw new IOException(errmsg)
456       }
457     }
458 
459     val directory = FSDirectory.open(jointIndexDir.toPath)
460     val analyzer = new StandardAnalyzer()
461     val config = new IndexWriterConfig(analyzer)
462     config.setCommitOnClose(true)
463     if (preExistent) {
464       // Add new documents to an existing index:
465       config.setOpenMode(OpenMode.CREATE_OR_APPEND)
466       val iwriter = new IndexWriter(directory, config)
467 
468       try {
469         val emptyDoc = new Document()
470         // id field not used in our search semantics, only to provide lucene uniqueness
471         emptyDoc.add(new StringField("id", "00000000-0000-0000-0000-000000000000", Field.Store.NO))
472         iwriter.updateDocument(new Term("fileIdentifier", "00000000-0000-0000-0000-000000000000"), emptyDoc)
473         iwriter.commit()
474         logger.info("Re-opened main index ready")
475       }
476       finally {
477         iwriter.close()
478       }
479     } else {
480       // Create a new index in the directory, removing any
481       // previously indexed documents:
482       config.setOpenMode(OpenMode.CREATE)
483       val iwriter = new IndexWriter(directory, config)
484 
485       try {
486         iwriter.deleteAll()
487         iwriter.addDocument(new Document)
488         iwriter.commit()
489         logger.info("Empty main index ready")
490       }
491       finally {
492         iwriter.close()
493       }
494     }
495 
496     directory
497   }
